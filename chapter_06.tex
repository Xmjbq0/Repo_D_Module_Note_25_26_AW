\section{Holonomic $\mathcal{D}$-modules}
Let $X = \text{spec} R$ be a smooth affine scheme with algebraic coordinates $(x_1, \ldots, x_n)$ and let $M$ be a finite generated $\mathcal{D}_X$-module. We already know $\Ch(M) \subset T^*X$ is a conic involutive (or coisotropic) subvariety. In particular, $\dim \Ch(M) \geq n$. 

\begin{definition}
    $M$ is called $\textbf{holonomic}$ if $\dim \Ch(M) = n$.
\end{definition}

\begin{corollary}
    If $M$ is a holonomic $\mathcal{D}_X$-module, then $\Ch(M) \subset T^*X$ is a Lagrangian subvariety, and $$\mathrm{CC}(M) = \sum_{E_i \subset X\atop \text{irr subvarieties}} n_i T^*_{E_i} X$$
\end{corollary}

\begin{remark}
    Recall that for a linear subspace $W \subseteq \Tv_{X,p}$, Lagrangian means $W=W^{\perp}\iff \mathrm{dim}_{k}W=n$, and we define $V\subseteq T^{\ast}X$ is Lagrangian if for any smooth closed point $p\in V$, we have $\mathcal{T}_{V,p}\subseteq \mathcal{T}_{T^{\ast}X,p}$ is a Lagrangian linear subspace. Here $T^{\ast}_{E_{i}}X$ is the conormal bundle of irreducible subvariety $E_{i}$. 
\end{remark}

\begin{example}
    \quad
    \begin{enumerate}
        \item $R$ is a $\mathcal{D}_X$-module. $\mathrm{CC}(R) = T^*_X X\simeq X \Rightarrow R$ is holonomic.
        \item Let $M$ be an integrable connection on $X$, we obtain that $\Ch(M)=T^{\ast}_{X}X$ and $\mathrm{dim}_{k}V(\xi_{1},\dots,\xi_{n})=n$, we have $M$ is holonomic. 
    \end{enumerate}
\end{example}

\section{The de Rham Complexes and Analytification}

Let $M$ be a $\mathcal{D}_X$-module, and denote $\partial_i = \partial_{x_i}$ be the vector field along $x_i$.  
Define a complex of $\mathbb{C}$-vector spaces (since $\partial_1$ is only $\mathbb{C}$-linear): $$K(M; \partial_1) = [M \xrightarrow{\partial_1} M].$$Here $[M\xrightarrow{\partial_{1}} M]$ means a complex concentrated on degree 0 and degree 1, with $\partial_{1}$ as coboundary map. 

\quad

\begin{note}[The cone of a morphism of complexes]\label{note 6.2}
Let $f: A^\bullet \longrightarrow B^\bullet$ be a morphism of complexes:

\begin{center}
 \begin{tikzcd}
 \cdots \arrow[r] & A^{n-1} \arrow[r, "\mathrm{d}_A^{n-1}"] \arrow[d, "f^{n-1}"] & A^n \arrow[r, "\mathrm{d}_A^n"] \arrow[d, "f^n"] & A^{n+1} \arrow[r] \arrow[d, "f^{n+1}"] & \cdots \\
 \cdots \arrow[r] & B^{n-1} \arrow[r, "\mathrm{d}_B^{n-1}"] & B^n \arrow[r, "\mathrm{d}_B^n"] & B^{n+1} \arrow[r] & \cdots
 \end{tikzcd}
\end{center}

then $\mathrm{\mathrm{Cone}}(f)=C^{\bullet}$ defined by $C^n = A^n \oplus B^{n-1}$ with differential $$\mathrm{d}_C^n = \begin{pmatrix}
 \mathrm{d}_A^n & 0 \\
 f^n & -\mathrm{d}_B^{n-1}
 \end{pmatrix}: A^n \oplus B^{n-1} \longrightarrow A^{n+1} \oplus B^n.$$The cone construction gives a short exact sequence of complexes:
\begin{center}
\begin{tikzcd}
0 \arrow[r] & B^\bullet[-1] \arrow[r] & \mathrm{Cone}(f)^\bullet \arrow[r] & A^\bullet \arrow[r] & 0.
\end{tikzcd}
\end{center}
\end{note}


With Note \ref{note 6.2}, for $k \geq 2$, we inductively define     $$K(M; \partial_1, \cdots, \partial_k) = \mathrm{Cone}\left(K(M; \partial_1, \cdots, \partial_{k-1}) \xrightarrow{\partial_k} K(M; \partial_1, \cdots, \partial_{k-1})\right).$$ Then the $\textbf{de Rham complex}$ of $M$ is   $$K(M; \partial_1, \cdots, \partial_n),$$ denoted as $\mathrm{DR}(M).$

Meanwhile, we have another way to define $\mathrm{DR}(M)$: 
$$M \to \Omega^{1}_{X} \otimes_{R} M \to \Omega^{2}_{X} \otimes_{R} M \to \cdots \to \Omega^{n}_{X} \otimes_{R} M.$$
With $M \to \Omega^{1}_{X}: m\rightarrow \sum\limits_{i}dx_{i}\otimes \partial_{i}(m)$, and $\Omega^{i}_{X} \otimes_{R} M \to \Omega^{i+1}_{X} \otimes_{R} M:\eta \otimes m \mapsto \sum\limits_{i} dx_i \wedge \eta \otimes \partial_i(m)$ for $i\ge 1$.
This definition of $\mathrm{DR}(M)$ is independent of algebraic coordinates. 

\begin{remark}
    Here $\sum\limits_{i} dx_i \otimes \partial_i \in \Omega^1 \otimes_R \mathcal{T}_X$ is canonical (i.e., independent of coordinates).
\end{remark}

\begin{lemma}
    We have isomorphic complexes: $$\mathrm{DR}(M) \cong K(M; \partial_1, \ldots, \partial_n).$$The coordinate-free de Rham complex is isomorphic to the Koszul complex on the partial derivatives.
\end{lemma}
\begin{proof}[Sketch of proof]
    When $n = 2$:

% 1. Koszul Complex
Koszul Complex $K(M; \partial_{1}, \partial_{2})$ is:
\[
\begin{tikzcd}[row sep=tiny, column sep=4em]
    M \arrow[r] 
    & M \oplus M \arrow[r] 
    & M \\
    m \arrow[r, mapsto] 
    & (\partial_{1} m, \partial_{2} m) 
    & \\
    & (m_1, m_2) \arrow[r, mapsto] 
    & \partial_2(m_1) - \partial_1(m_2)
\end{tikzcd}
\]

% 2. de Rham Complex
De Rham Complex $\mathrm{DR}(M)$ is:
\[
\begin{tikzcd}[row sep=tiny, column sep=large]
    M \arrow[r] 
    & \Omega^{1}_{X}\otimes_{R} M \arrow[r] 
    & \Omega_{X}^{2}\otimes_{R}M \\
    m \arrow[r, mapsto] 
    & dx_{1}\otimes\partial_{1} m + dx_{2}\otimes\partial_{2}m 
    & \\
    & dx_{1}\otimes m_{1} + dx_{2}\otimes m_{2} \arrow[r, mapsto] 
    & dx_{1}\wedge dx_{2}\otimes(\partial_{1} m_{2}-\partial_{2} m_{1})
\end{tikzcd}
\]

The general case is similar, and we can construct an explicit isomorphism between the two complexes.
\end{proof}

Now, for $\alpha \notin \mathbb{Z}$, consider the de Rham complex of $M_{\alpha}$: $$M_{\alpha} \xrightarrow{\partial_x} M_{\alpha}$$
Here $\partial_x$ acts on the left, so it is only a $\C$-linear map, not $\D_X$-linear.

But in order to well define this problem, we need to take localization at $x \neq 0$, i.e. work on $ U := X \backslash \{0\}$. That is to calculate $\mathrm{DR}(M_{\alpha})|_{X \backslash \{0\}}$ rather than $\mathrm{DR}(M_{\alpha})$ itself.

We need several facts to give the true de Rham complex on $X \backslash \{0\}$ that makes sense.

\begin{lemma}[Quasi-coherent sheaf and localization of $\D$-modules]
Assume the base field is algebraically closed field of characteristic $0$.
\begin{enumerate}
    \item For an arbitrary smooth (not necessarily affine) variety $X$, $\D_X$ is a quasi-coherent sheaf of $\mathcal{O}_X$-module \cite[Cor.~4.9.]{chen2024dmodules} or \cite[Beginning of Sec.~1.4.]{hotta2008d}.
    \item The definition of a \textbf{quasi-coherent $\D_X$-module} means it is quasi-coherent as an $\mathcal{O}_X$-module \cite[Def.~6.3.]{chen2024dmodules} or \cite[Notation~1.4.1.]{hotta2008d}.
    \item So the localization of a quasi-coherent $\D_X$-module $M$ at an open subset $U \subseteq X$ is the localization as an $\mathcal{O}_X$-module, that is, given by $\mathcal{O}_U \otimes_{\mathcal{O}_X} M$.
    \item The $\Gamma(U,\D_X)$-action on $M_U$ is completely determined by $\Gamma(V,\D_X)$-action on $M_V$ for open sets $U\subseteq V$ \cite[\href{https://stacks.math.columbia.edu/tag/0G36}{Tag 0G36}]{stacks-project} or \cite[Lem.~4.1., Lem.~4.2.]{chen2024dmodules}.
    \item In particular assume $X=\Spec R$ and $U=\Spec R_{f}$, for a derivation (order 1 differential operator) $\theta\in \Tv_X \subseteq \D_X$, we have: $$\theta|_U\left(\frac{m}{f}\right) = \frac{\theta(m)}{f} + \frac{m}{f^2}\cdot \theta(f)$$
\end{enumerate}
\end{lemma}


So we have $$M_{\alpha,U} := \mathcal{O}_{U} \otimes_{\mathcal{O}_X} M_{\alpha} = \mathcal{O}_{U} \otimes_{\mathcal{O}_X} \frac{\D_X}{\D_X \cdot (x\partial_x - \alpha)} = \frac{\D_U}{\D_U \cdot (x\partial_x - \alpha)}$$

And

$$\DR(M_{\alpha})|_U =\frac{\D_U}{\D_U \cdot (x\partial_x - \alpha)}\xrightarrow{\partial_x|_U} \frac{\D_U}{\D_U \cdot (x\partial_x - \alpha)}$$

This time since $\frac{1}{x}$ is a regular function on $U=X \backslash \{0\}$, the element
$x^{-k}x^\alpha$ becomes \emph{canonically defined} as $x^{-k}\otimes \overline{1}$.
In particular, there is no longer any ``up to a constant'' ambiguity. With $\partial_x \cdot \overline{1} = \frac{\alpha}{x} \cdot \overline{1}$ on $U$, we get $M_{\alpha,U}$ is a $\C[x, \frac{1}{x}]$-module with one generator.

With $\text{Ker}(\partial_x|_U) = \{f \cdot x^{\alpha} \mid f\in \C[x,\frac{1}{x}],\,\partial_x(f \cdot x^{\alpha}) = 0\}\simeq \{f \in \mathbb{C}[x, \frac{1}{x}] \mid xf' + \alpha f= 0\} = 0$, know that $x^{-\alpha}$ should be the solution of $xf'+\alpha f$, but $\alpha\not\in \mathbb{Z}$ makes $x^{-\alpha}\not\in \mathbb{C}\left[ x, \frac{1}{x} \right]$, thus has no solution in $\mathbb{C}[x, \frac{1}{x}]$. 

Yet, from ODE \cite[Chap.~XIX, n\textsuperscript{o}~384]{Goursat1933CoursAnalyseT2},
we know that the differential equation $xf'+\alpha f=0$ has unique (given initial value at some point) holomorphic solutions at least locally, which suggests that the current definition of de Rham complex is not enough, and we need to extend scalars.

Let $Y = \text{Spec } S$ be an affine variety of finite type over $\mathbb{C}$ with $\dim Y = n$. Define a close immersion $Y \longrightarrow \mathbb{A}^{m}_{\mathbb{C}}$ by a $\C$-algebra map:
\begin{align*}
    \mathbb{C}[z_1, \ldots, z_m] &\longrightarrow S \\
    z_i &\longmapsto \text{generators of } S
\end{align*}
And denote $I = I(Y) = \langle f_1, \ldots, f_k \rangle, \, f_i \in \mathbb{C}[z_1, \ldots, z_m]$. 

In particular, $f_i$ are holomorphic functions on $\mathbb{C}^m$, then $f_1 = \cdots = f_k = 0$ defines an analytic subvariety $Y^{\mathrm{an}}$ of the complex manifold $\mathbb{C}^m$. We have a continuous map: 
$$ \lambda: Y^{\mathrm{an}} \longrightarrow Y $$For all$\ y \in Y^{\text{an}}$, by Implicit Function Theorem, $Y^{\text{an}}$ is smooth around $y$ if and only if the Jacobian matrix $J(f_1, \ldots, f_k)(y)$ at $y$ has rank $m-n$. 

\begin{definition} We define

\begin{enumerate}
    \item $Y$ is $\textbf{smooth}$ at a closed point $p$ if $$\dim_{\mathbb{C}} \mathfrak{m}_p/\mathfrak{m}_p^2 = n,$$ where $\mathfrak{m}_p$ is the maximal ideal of $p$.
    \item $Y$ is $\textbf{smooth}$ if $Y$ is smooth at every closed point.
\end{enumerate}
\end{definition}
\begin{remark}
    For other equivalent definitions of smoothness, see .
\end{remark}
\begin{lemma}
    $Y$ is smooth $\iff$ $Y^{\text{an}}$ is a complex manifold.
\end{lemma}
\begin{proof}
    Enough to prove for closed point $p$ of $Y$ we have: $$\text{rank}(J(f_1, \ldots, f_k)(p)) = m-n \quad (*)\iff \quad \dim_{\mathbb{C}} \mathfrak{m}_{p}/\mathfrak{m}_p^2 = n.$$Know that $p$ is also a closed point of $\mathbb{A}^{m}_{\mathbb{C}}$ with maximal ideal $\mathfrak{m} = (z_1 - \alpha_1, \cdots, z_m - \alpha_m)$. Define $$\mathbb{C}[z_1, \cdots, z_m] \to V = \mathbb{C}^m\atop f \quad \mapsto \quad \left( \frac{\partial f}{\partial z_1}(p), \cdots, \frac{\partial f}{\partial z_m}(p) \right)$$By HW, this induces an iso:$$
\mathfrak{\mathfrak{m}} /\mathfrak{m}^2 \xrightarrow[\varphi]{\simeq} V\simeq \Omega_{V,p}=\Omega_{V}\otimes R/\mathfrak{m}_{p}$$Since $I\subseteq \mathfrak{m}$ is generated by $f_{1},f_{2},\dots,f_{k}$, and for $h\in \mathbb{C}[z_{1},\dots,z_{m}]$, we can write $h=c+\sum(z_{i}-\alpha_{i})+\sum(z_{i}-\alpha_{i})(z_{j}-\alpha_{j})+\dots$, then we get $[hf_{i}]=c[f]$ in $\mathfrak{m} /\mathfrak{m}^{2}$ for $\mathfrak{m}^{2}$ makes only monomial with $\mathrm{deg}\leq 1$ matter and $f=\sum(z_{i}-\alpha_{i})+(\text{higher deg})$. So we have $\varphi(f_{1}),\dots,\varphi(f_{k})$ span $\varphi(I)=I+\mathfrak{m} /\mathfrak{m}^{2}$, and further, we obtain: $$
\dim(\varphi(I)) = \operatorname{rank} \left( J(f_1, \cdots, f_k)(p) \right).
$$
On the other hand, we have short exact sequence $$0\rightarrow\varphi(I)=I+\mathfrak{m}^{2} /\mathfrak{m}^{2}\rightarrow \mathfrak{m} /\mathfrak{m}^{2}\rightarrow \mathfrak{m}/\mathfrak{m}^{2}+I\rightarrow 0.$$
Know that $\mathfrak{m}_{p}=\mathfrak{m} /I$, $\mathfrak{m}_{p} /\mathfrak{m}_{p}^{2}=\frac{\mathfrak{m}/I}{\mathfrak{m}^{2}+I /I}\simeq \mathfrak{m} /\mathfrak{m}^{2}+I$,  with $\mathrm{dim}_{\mathbb{C}} \mathfrak{m} /\mathfrak{m}^{2}=m$, we get $\operatorname{rank} \left( J(f_1, \cdots, f_k)(p) \right)+\mathrm{dim}_{\mathbb{C}}\mathfrak{m}_{p} /\mathfrak{m}_{p}^{2}=m$, done. 
\end{proof}

\section{Basic version Riemman-Hilbert correspondence}

Let $Y$ be a complex manifold, $\mathcal{O}_Y$: the sheaf of holomorphic functions on $Y$, $\mathcal{L}$: a sheaf of $\mathbb{C}$-vector spaces. 

\begin{definition}
    $\mathcal{L}$ is called a $\textbf{local system}$ if $\forall y \in Y$, $\exists$ open neighborhood $U_y$ of $y$ s.t. $\mathcal{L}|_{U_y} \simeq\mathbb{C}^r_{U_y}$, where $\mathbb{C}^{r}_{U_y}$ is the constant sheaf on $U_y$.
\end{definition}

\begin{theorem}
    Let $X = \text{Spec } R$ be a smooth affine algebraic variety over $\mathbb{C}$ and $M$ an integrable connection on $X$, then $$M^{\text{an}} \simeq \mathcal{L} \otimes \mathcal{O}_{X^{\text{an}}},$$with $\mathcal{L}$ a local system.
\end{theorem}
\begin{proof}
   \quad

    Goal: $M$ is locally free, we want to show for small enough neighborhood, we can pick a basis $s_{1},\dots,s_{r}$ makes $\partial_{x_{i}}s_{j}=0$, thus
	We have known $M$ is a locally free $R$-module, $\forall x \in X$, take an open neighborhood $U \subset X$ s.t. $M|_U$ is free, then $M^{\text{an}}|_{U^{\text{an}}}$ is a free $\mathcal{O}_{U^{\text{an}}}$-module, generated by $\underline{\mathfrak{m}} = (m_1, \cdots, m_r)$.

	Denote $(x_1, \cdots, x_n)$ coordinates of $U^{\text{an}}$, then we have $\partial_{n} \underline{\mathfrak{m}} = X \cdot\underline{ \mathfrak{m}}$, where $X$ is a matrix of holomorphic functions on $U^{\mathrm{an}}$. Expand $X$ w.r.t. $x_n$ in a small enough neighborhood of $x\in V\subseteq U^{\mathrm{an}}$, we have $$X = \sum_{i\ge0} x_n^i \cdot X_i,$$where $X_i$ is a matrix of holomorphic functions in $(x_1, \cdots, x_{n-1})$. 

	Take $\Psi=\sum\limits_{i\ge 0} \frac{1}{i+1} x_{n}^{i+1}X_{i}$, so $\partial_{x}(\Psi)=X$, then $\Phi=\mathrm{\exp}(-\Psi)$, which is an invertible matrix. Set $\underline{S}=(s_{1},\dots,s_{r})=\Phi(\underline{\mathfrak{m}})$, so $\partial_{x}\underline{S}=\partial_{n}(\Phi)\cdot \underline{\mathfrak{m}}+\Phi \cdot(\partial_{n} \underline{\mathfrak{m}})=(\partial_{n}(\Phi)+\Phi \cdot X)\underline{\mathfrak{m}}=0$. 

	Since $\Phi$ is invertible matrix, we obtain $s_{1},\dots,s_{r}$ also a basis of $U^{\mathrm{an}}|_{V}$, so we have $M^{\mathrm{an}}|_{V}=\bigoplus\limits_{i=1}^{r}\mathcal{O}_{V}s_{i}$ s.t. $\partial_{n}s_{i}=0$. 

	Want to clarify the relation of $s_{i}$ and other $\partial_{x_{j}},j<n$, then Take $M_1 = \ker[M^{\text{an}}|_V \xrightarrow{\partial_n} M^{\text{an}}|_V]=\left\{ \partial_{n}\left( \sum f_{i}s_{i} \right)=0:f_{i}\in \mathcal{O}_{V} \right\}=\left\{ \sum\limits_{i=1}^{n}\partial_{n}(f_{i})s_{i}=0 \right\}.$ So we get $\sum_{i=1}^r \partial_n(f_i) s_i = 0$ for all $i=1,\dots,r$, i.e. $f_{i}$ is independent of $x_{n}$. 

	So we have $M_{1}=\bigoplus\limits_{i=1}^{r}\mathcal{O}_{V}^{n-1}s_{i}$, $\mathcal{O}_{V}^{n-1}$ are holomorphic functions that are independent of $x_{n}$, which suggests that we do not need the var $x_{n}$. And do the same thing for $M_{1}$,we get a new basis with $\partial_{n}s_{i}=\partial_{n-1}s_{i}=0$ for all $i=1,\dots,r$, and the change of basis is via an invertible matrix of homomorphic functions in $(x_{1},\dots,x_{n-1})$. 

	Continue the process, by induction, we finally obtain:  $$ M_n = \bigoplus_{i=1}^r \mathbb{C} \cdot s_i $$with $\partial_j(s_i) = 0 \quad \forall i, j$. And since every change of basis is by an invertible matrix, we know that for a small enough neighborhood $V \subseteq U^{\mathrm{an}}$, we get $M^{\mathrm{an}}|_{V}=\bigoplus\limits_{i=1}^{r}\mathcal{O}_{V}s_{i}$, with $\partial_{j}(s_{i})=0$ . 
\end{proof}

\begin{corollary}
    $\mathrm{DR}(M)|_{V}\stackrel{\text{quasi-isomorphism}}{\simeq}H^{0}\mathrm{DR}(M)|_{V}\simeq \mathbb{C}^{r}_{V}$. 
\end{corollary}
\begin{example}[Algebraic vs Analytic]
    Let $M = \mathbb{C}[x, \frac{1}{x}]$, $M' = \mathbb{C}[x, \frac{1}{x}] \cdot e^{\frac{1}{x}}$. 
    
    Know that $M$ and $M'$ are different as algebraic $D$-modules, but $M^{\text{an}}|_{\mathbb{C}^*} = \mathcal{O}_{\mathbb{C}^*}$, $M'^{\mathrm{an}}|_{\mathbb{C}^{\ast}}=\mathcal{O}_{\mathbb{C}^{\ast}}\cdot e^{1/x}=\mathcal{O}_{\mathbb{C}^{\ast}}$, where $\mathbb{C}^{\ast}$ is the punctured complex plane. So $M^{\text{an}}|_{\mathbb{C}^*} \cong M'^{\text{an}}|_{\mathbb{C}^*}$ as analytic $\mathcal{D}$-modules and $DR(M^{\text{an}}) = DR(M'^{\text{an}}) \cong \mathbb{C}_{\mathbb{C}^*}^*$ Moreover, $M^{\text{an}} = \mathcal{O}_{\mathbb{C}}\cdot \frac{1}{x},M'^{\mathrm{an}}=\mathcal{O}_{\mathbb{C}}\cdot e^{1/x}$ so $M^{\mathrm{an}}$ and $M'^{\mathrm{an}}$ are different as analytic $\mathcal{D}$-module. 
\end{example}

\section{Regular holonomic $\mathcal{D}$-module}

Let $X = \text{Spec } R$ be a smooth affine algebraic variety over $\mathbb{C}$ and $M$ a holonomic $\mathcal{D}_X$-mod. 
For any point $x \in X^{\text{an}}$, we consider the embedding:$$
i: \Delta \hookrightarrow X^{\text{an}}, \quad 0 \mapsto x.
$$where $\Delta$ is a small disk in the complex plane.
Then we have a morphism of sheaves of rings:$$R \to \mathcal{O}_{\Delta}, \quad f \mapsto f|_{\Delta}.$$We define $i^*M = M \otimes_R \mathcal{O}_{\Delta}$. 
\begin{definition}
    $M$ is called $\textbf{regular holonomic}$, if $i^{\ast}M$ is generated by $M_{\alpha}^{\mathrm{an}}$ and $N^{\mathrm{an}}|_{\Delta}$ in $\mathbf{K}_{0}(\mathcal{D}_{\Delta})$ for every small enough disk embedding $i:\Delta \hookrightarrow X^{\mathrm{an}}$. 
\end{definition}

\begin{example}
    $M=\mathbb{C}\left[ x, \frac{1}{x} \right]\cdot e^{1/x}$ is not regular because $M$ is simple and $M\simeq \mathcal{D}_{X} /\mathcal{D}_{X}(x^{2}\partial_{x}+1)$. 
\end{example}
