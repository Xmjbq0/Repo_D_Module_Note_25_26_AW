\section{Holonomic $\mathcal{D}$-modules}
Let $X = \text{spec} R$ be a smooth affine scheme with algebraic coordinates $(x_1, \ldots, x_n)$ and let $M$ be a finite generated $\mathcal{D}_X$-module. We already know $\Ch(M) \subset T^*X$ is a conic involutive (or coisotropic) subvariety. In particular, $\dim \Ch(M) \geq n$.

\begin{definition}
    $M$ is called $\textbf{holonomic}$ if $\dim \Ch(M) = n$.
\end{definition}

\begin{corollary}
    If $M$ is a holonomic $\mathcal{D}_X$-module, then $\Ch(M) \subset T^*X$ is a Lagrangian subvariety, and $$\mathrm{CC}(M) = \sum_{E_i \subset X\atop \text{irr subvarieties}} n_i T^*_{E_i} X$$
\end{corollary}

\begin{remark}
    Recall that for a linear subspace $W \subseteq \Tv_{X,p}$, Lagrangian means $W=W^{\perp}\iff \mathrm{dim}_{k}W=n$, and we define $V\subseteq T^{\ast}X$ is Lagrangian if for any smooth closed point $p\in V$, we have $\mathcal{T}_{V,p}\subseteq \mathcal{T}_{T^{\ast}X,p}$ is a Lagrangian linear subspace. Here $T^{\ast}_{E_{i}}X$ is the conormal bundle of irreducible subvariety $E_{i}$.
\end{remark}

\begin{example}
    \quad
    \begin{enumerate}
        \item $R$ is a $\mathcal{D}_X$-module. $\mathrm{CC}(R) = T^*_X X\simeq X \Rightarrow R$ is holonomic.
        \item Let $M$ be an integrable connection on $X$, we obtain that $\Ch(M)=T^{\ast}_{X}X$ and $\mathrm{dim}_{k}V(\xi_{1},\dots,\xi_{n})=n$, we have $M$ is holonomic.
    \end{enumerate}
\end{example}

\section{The de Rham Complexes and Analytification}

Let $M$ be a $\mathcal{D}_X$-module, and denote $\partial_i = \partial_{x_i}$ be the vector field along $x_i$.
Define a complex of $\mathbb{C}$-vector spaces (since $\partial_1$ is only $\mathbb{C}$-linear): $$K(M; \partial_1) = [M \xrightarrow{\partial_1} M].$$Here $[M\xrightarrow{\partial_{1}} M]$ means a complex concentrated on degree 0 and degree 1, with $\partial_{1}$ as coboundary map.

\quad

\begin{note}[The cone of a morphism of complexes]\label{note 6.2}
    Let $f: A^\bullet \longrightarrow B^\bullet$ be a morphism of complexes:

    \begin{center}
        \begin{tikzcd}
            \cdots \arrow[r] & A^{n-1} \arrow[r, "\mathrm{d}_A^{n-1}"] \arrow[d, "f^{n-1}"] & A^n \arrow[r, "\mathrm{d}_A^n"] \arrow[d, "f^n"] & A^{n+1} \arrow[r] \arrow[d, "f^{n+1}"] & \cdots \\
            \cdots \arrow[r] & B^{n-1} \arrow[r, "\mathrm{d}_B^{n-1}"] & B^n \arrow[r, "\mathrm{d}_B^n"] & B^{n+1} \arrow[r] & \cdots
        \end{tikzcd}
    \end{center}

    then $\mathrm{\mathrm{Cone}}(f)=C^{\bullet}$ defined by $C^n = A^n \oplus B^{n-1}$ with differential $$\mathrm{d}_C^n = \begin{pmatrix}
            \mathrm{d}_A^n & 0                   \\
            f^n            & -\mathrm{d}_B^{n-1}
        \end{pmatrix}: A^n \oplus B^{n-1} \longrightarrow A^{n+1} \oplus B^n.$$The cone construction gives a short exact sequence of complexes:
    \begin{center}
        \begin{tikzcd}
            0 \arrow[r] & B^\bullet[-1] \arrow[r] & \mathrm{Cone}(f)^\bullet \arrow[r] & A^\bullet \arrow[r] & 0.
        \end{tikzcd}
    \end{center}
\end{note}


With Note \ref{note 6.2}, for $k \geq 2$, we inductively define     $$K(M; \partial_1, \cdots, \partial_k) = \mathrm{Cone}\left(K(M; \partial_1, \cdots, \partial_{k-1}) \xrightarrow{\partial_k} K(M; \partial_1, \cdots, \partial_{k-1})\right).$$ Then the $\textbf{de Rham complex}$ of $M$ is   $$K(M; \partial_1, \cdots, \partial_n),$$ denoted as $\mathrm{DR}(M).$

Meanwhile, we have another way to define $\mathrm{DR}(M)$:
$$M \to \Omega^{1}_{X} \otimes_{R} M \to \Omega^{2}_{X} \otimes_{R} M \to \cdots \to \Omega^{n}_{X} \otimes_{R} M.$$
With $M \to \Omega^{1}_{X}: m\rightarrow \sum\limits_{i}dx_{i}\otimes \partial_{i}(m)$, and $\Omega^{i}_{X} \otimes_{R} M \to \Omega^{i+1}_{X} \otimes_{R} M:\eta \otimes m \mapsto \sum\limits_{i} dx_i \wedge \eta \otimes \partial_i(m)$ for $i\ge 1$.
This definition of $\mathrm{DR}(M)$ is independent of algebraic coordinates.

\begin{remark}
    Here $\sum\limits_{i} dx_i \otimes \partial_i \in \Omega^1 \otimes_R \mathcal{T}_X$ is canonical (i.e., independent of coordinates).
\end{remark}

\begin{lemma}
    We have isomorphic complexes: $$\mathrm{DR}(M) \cong K(M; \partial_1, \ldots, \partial_n).$$The coordinate-free de Rham complex is isomorphic to the Koszul complex on the partial derivatives.
\end{lemma}
\begin{proof}[Sketch of proof]
    When $n = 2$:

    % 1. Koszul Complex
    Koszul Complex $K(M; \partial_{1}, \partial_{2})$ is:
    \[
        \begin{tikzcd}[row sep=tiny, column sep=4em]
            M \arrow[r]
            & M \oplus M \arrow[r]
            & M \\
            m \arrow[r, mapsto]
            & (\partial_{1} m, \partial_{2} m)
            & \\
            & (m_1, m_2) \arrow[r, mapsto]
            & \partial_2(m_1) - \partial_1(m_2)
        \end{tikzcd}
    \]

    % 2. de Rham Complex
    De Rham Complex $\mathrm{DR}(M)$ is:
    \[
        \begin{tikzcd}[row sep=tiny, column sep=large]
            M \arrow[r]
            & \Omega^{1}_{X}\otimes_{R} M \arrow[r]
            & \Omega_{X}^{2}\otimes_{R}M \\
            m \arrow[r, mapsto]
            & dx_{1}\otimes\partial_{1} m + dx_{2}\otimes\partial_{2}m
            & \\
            & dx_{1}\otimes m_{1} + dx_{2}\otimes m_{2} \arrow[r, mapsto]
            & dx_{1}\wedge dx_{2}\otimes(\partial_{1} m_{2}-\partial_{2} m_{1})
        \end{tikzcd}
    \]

    The general case is similar, and we can construct an explicit isomorphism between the two complexes.
\end{proof}

\end{example}

Now, for $\alpha \notin \mathbb{Z}$, consider the de Rham complex of $M_{\alpha}$: $$M_{\alpha} \xrightarrow{\partial_x} M_{\alpha}$$
Here $\partial_x$ acts on the left, so it is only a $\C$-linear map, not $\D_X$-linear.

But in order to well define this problem, we need to take localization at $x \neq 0$, i.e. work on $ U := X \backslash \{0\}$. That is to calculate $\mathrm{DR}(M_{\alpha})|_{X \backslash \{0\}}$ rather than $\mathrm{DR}(M_{\alpha})$ itself.

We need several facts to give the true de Rham complex on $X \backslash \{0\}$ that makes sense.

\begin{lemma}[Quasi-coherent sheaf and localization of $\D$-modules]\label{lem:quasi-coherent-sheaf-and-localization-of-D-modules}
    Assume the base field is algebraically closed field of characteristic $0$.
    \begin{enumerate}
        \item For an arbitrary smooth (not necessarily affine) variety $X$, $\D_X$ is a quasi-coherent sheaf of $\mathcal{O}_X$-module \cite[Cor.~4.9. or more exactly Construction~4.4.]{chen2024dmodules} or \cite[Beginning of Sec.~1.4.]{hotta2008d}.
        \item For a remark of the first point, $\D_X$ is a $(\Ox,\Ox)$-bimodule, so the \textbf{commutativity constraint}
              \[
                  \D_X\mathrm{(right)}\otimes_{\Ox} M\stackrel{?}{\simeq} M\otimes_{\Ox}\mathrm{(left)}\D_X
              \]
              may not tautologically true, such map may even not exist.
        \item The definition of a \textbf{quasi-coherent $\D_X$-module} means it is quasi-coherent as an $\mathcal{O}_X$-module \cite[Def.~6.3.]{chen2024dmodules} or \cite[Notation~1.4.1.]{hotta2008d}.
        \item So the localization of a quasi-coherent $\D_X$-module $M$ at an open subset $U \subseteq X$ is the localization as an $\mathcal{O}_X$-module, that is, given by $\mathcal{O}_U \otimes_{\mathcal{O}_X} M$.
        \item The $\Gamma(U,\D_X)$-action on $M_U$ is completely determined by $\Gamma(V,\D_X)$-action on $M_V$ for open sets $U\subseteq V$ \cite[\href{https://stacks.math.columbia.edu/tag/0G36}{Tag 0G36}]{stacks-project} or \cite[Lem.~4.1., Lem.~4.2.]{chen2024dmodules}.
        \item In particular assume $X=\Spec R$ and $U=\Spec R_{f}$, for a derivation (order 1 differential operator) $\theta\in \Tv_X \subseteq \D_X$, we have: $$\theta|_U\left(\frac{m}{f}\right) = \frac{\theta(m)}{f} + \frac{m}{f^2}\cdot \theta(f)$$
    \end{enumerate}
\end{lemma}


So we have $$M_{\alpha,U} := \mathcal{O}_{U} \otimes_{\mathcal{O}_X} M_{\alpha} = \mathcal{O}_{U} \otimes_{\mathcal{O}_X} \frac{\D_X}{\D_X \cdot (x\partial_x - \alpha)} = \frac{\D_U}{\D_U \cdot (x\partial_x - \alpha)}$$

And
$$\DR(M_{\alpha})|_U =\frac{\D_U}{\D_U \cdot (x\partial_x - \alpha)}\xrightarrow{\partial_x|_U} \frac{\D_U}{\D_U \cdot (x\partial_x - \alpha)}$$

\begin{remark}
    So we get, for inclusion $j: U:=X\setminus \{0\} \to X$, $\D_X/\D_X(x\partial_x-\alpha) = j_*(O_U\cdot x^\alpha)$ as sheaves of $\D_X$-modules. Since we are working on an affine open $X$ with quasi-cohrent $\Ox$-modules, the correctly defined $\D$-module direct image functor \cite[$Rf_*$ in Sec.~1.5.]{hotta2008d} is just $j_*$.
\end{remark}

$\text{Ker}(\partial_x|_U) = \{f \cdot x^{\alpha} \mid f\in \C[x,\frac{1}{x}],\,\partial_x(f \cdot x^{\alpha}) = 0\}\simeq \{f \in \mathbb{C}[x, \frac{1}{x}] \mid xf' + \alpha f= 0\} = 0$, know that $x^{-\alpha}$ should be the solution of $xf'+\alpha f$, but $\alpha\not\in \mathbb{Z}$ makes $x^{-\alpha}\not\in \mathbb{C}\left[ x, \frac{1}{x} \right]$, thus has no solution in $\mathbb{C}[x, \frac{1}{x}]$.

Yet, from ODE \cite[Chap.~XIX, n\textsuperscript{o}~384]{Goursat1933CoursAnalyseT2},
we know that the differential equation $xf'+\alpha f=0$ has unique (given initial value at some point) holomorphic solutions at least locally, which suggests that the current definition of de Rham complex is not enough, and we need to extend scalars.

Let $Y = \text{Spec } S$ be an affine variety of finite type over $\mathbb{C}$ with $\dim Y = n$. Define a close immersion $Y \longrightarrow \mathbb{A}^{m}_{\mathbb{C}}$ by a $\C$-algebra map:
\begin{align*}
    \mathbb{C}[z_1, \ldots, z_m] & \longrightarrow S                   \\
    z_i                          & \longmapsto \text{generators of } S
\end{align*}
And denote $I = I(Y) = \langle f_1, \ldots, f_k \rangle, \, f_i \in \mathbb{C}[z_1, \ldots, z_m]$.

In particular, $f_i$ are holomorphic functions on $\mathbb{C}^m$, then $f_1 = \cdots = f_k = 0$ defines an analytic subvariety $Y^{\mathrm{an}}$ of the complex manifold $\mathbb{C}^m$. We have a continuous map:
$$ \lambda: Y^{\mathrm{an}} \longrightarrow Y $$For all$\ y \in Y^{\mathrm{an}}$, by Implicit Function Theorem, $Y^{\mathrm{an}}$ is smooth around $y$ if and only if the Jacobian matrix $J(f_1, \ldots, f_k)(y)$ at $y$ has rank $m-n$.

\begin{definition} We define

    \begin{enumerate}
        \item $Y$ is $\textbf{smooth}$ at a closed point $p$ if $$\dim_{\mathbb{C}} \mathfrak{m}_p/\mathfrak{m}_p^2 = n,$$ where $\mathfrak{m}_p$ is the maximal ideal of $p$ in $\mathcal{O}_Y$.
        \item $Y$ is $\textbf{smooth}$ if $Y$ is smooth at every closed point.
    \end{enumerate}
\end{definition}
\begin{remark}
    For other equivalent definitions of smoothness, see Definition~\ref{def:smooth-locus} and Remark~\ref{rmk:smooth-locus}.
    Again, one should note that, if the base field is not perfect, regular local ring only implies regularity at that point, not smoothness.
\end{remark}
\begin{lemma}
    $Y$ is smooth $\iff$ $Y^{\mathrm{an}}$ is a complex manifold.
\end{lemma}
\begin{proof}
    Enough to prove for closed point $p$ of $Y$ we have:
    $$\text{rank}(J(f_1, \ldots, f_k)(p)) = m-n \quad \textcolor{red}{(*)}\iff \quad \dim_{\mathbb{C}} \mathfrak{m}_{p}/\mathfrak{m}_p^2 = n.$$
    Know that $p$ is also a closed point of $\mathbb{A}^{m}_{\mathbb{C}}$ with maximal ideal $\mathfrak{m} = (z_1 - \alpha_1, \cdots, z_m - \alpha_m)$. So $\mathfrak{m}_p = \mathfrak{m}/I$.
    Define
    \begin{align*}
        \mathbb{C}[z_1, \ldots, z_m] & \longrightarrow V = \mathbb{C}^m                                                                          \\
        f                            & \longmapsto \left( \frac{\partial f}{\partial z_1}(p), \cdots, \frac{\partial f}{\partial z_m}(p) \right)
    \end{align*}
    By HW.2. Prob.3., since $\A^n_\C$ is smooth, this matches the following isomorphism:
    \begin{alignat*}{3}
        \varphi: \m/\m^2 & \xrightarrow{\sim}   \Omega_{\A^n_\C,p}=\Omega_{\A^n_\C}\otimes_{\C[z_1, \ldots, z_m]} \C[z_1, \ldots, z_m]/\mathfrak{m} &  & =\bigoplus_{i=1}^m \C\, \overline{\mathrm{d}z_i}                                 \\
        \overline{f}     & \xmapsto{\hspace*{17em}}  \overline{\mathrm{d}f}                                                                         &  & = \sum_{i=1}^m \frac{\partial f}{\partial z_i}(p) \cdot \overline{\mathrm{d}z_i} \\
    \end{alignat*}

    Since $I\subseteq \mathfrak{m}$ is generated by $f_{1},f_{2},\dots,f_{k}$, $\frac{I+\m^2}{\m^2}$ is generated by $\overline{f_{1}},\overline{f_{2}},\dots,\overline{f_{k}}$ as $\C[z_1, \ldots, z_m]/\mathfrak{m} = \C$-vector space. So we have:
    $$\dim(\varphi(I)) = \operatorname{rank} \left( J(f_1, \cdots, f_k)(p) \right).$$

    On the other hand, we have short exact sequence
    $$0\rightarrow \frac{I+\m^2}{\m^2}\rightarrow \frac{\m}{\m^2}\rightarrow \frac{\m}{\m^2 + I}\rightarrow 0.$$

    And we have $\mathfrak{m}_{p}=\mathfrak{m} /I$, $\mathfrak{m}_{p} /\mathfrak{m}_{p}^{2}=\frac{\mathfrak{m}/I}{\mathfrak{m}^{2}+I /I}\simeq \frac{\mathfrak{m}}{\mathfrak{m}^{2}+I}$,  with $\mathrm{dim}_{\mathbb{C}}\, \mathfrak{m} /\mathfrak{m}^{2}=m$, we get $\operatorname{rank} \left( J(f_1, \cdots, f_k)(p) \right)+\mathrm{dim}_{\mathbb{C}}\,\mathfrak{m}_{p} /\mathfrak{m}_{p}^{2}=m$, done.
\end{proof}
\quad

Let $Z$ be a complex manifold (or variety) and $\mathcal{O}_{Z}$ the sheaf of all holomorphic functions on $Z$.

We consider $Z=Y^{\mathrm{an}}$, have $\lambda:Y^{\mathrm{an}}\rightarrow Y$, it is a morphism between locally ringed space, so by definition we are also given a morphism of sheaves of rings $S \longrightarrow \mathcal{O}_{Y^{\mathrm{an}}}$ (In this case is to identify regular algebraic functions as holomorphic functions). If $M$ is an $S$-module, then $$
    M^{\mathrm{an}} \coloneqq M \otimes_S \mathcal{O}_{Y^{\mathrm{an}}}$$

To be precise, $Y^{\mathrm{an}}$ rather than $Y$ is where we stand, so $S = \lambda^{-1}\mathcal{O}_{Y}$, and $M^{\mathrm{an}} := \mathcal{O}_{Y^{\mathrm{an}}} \otimes_{\lambda^{-1}\mathcal{O}_Y} \lambda^{-1}M$.

We also have the canonical morphism between sheaves of holomorphic differential operators $\lambda^{-1}\mathcal{D}_{Y} \to \D_{Y^{\mathrm{an}}}$.

So by associativity of tensor product \cite[Prop.~6.5.12.]{li2019algebraic}, we have:
$$M^{\mathrm{an}} = \mathcal{O}_{Y^{\mathrm{an}}} \otimes_{\lambda^{-1}\mathcal{O}_Y} \lambda^{-1}M = \D_{Y^{\mathrm{an}}} \otimes_{\lambda^{-1}\D_Y} \lambda^{-1}M$$

So $M^{\mathrm{an}}$ got its $\mathcal{D}_{Y^{\mathrm{an}}}$-module structure by multiplication on the left part of tensor product \cite[Chap.~4.7.]{hotta2008d}.

Then we finally define
$$
    \operatorname{DR}(M) = K(M^{\mathrm{an}}; \partial_1, \cdots, \partial_n).
$$

On the other side, also by associativity of tensor product, which is a natural isomorphism, so also gives the definitions of arrows on the second row:
\begin{align*}
    \mathrm{DR}(M) & = \left[0 \to M^{\mathrm{an}} \to M^{\mathrm{an}} \otimes_S \Omega_Y^1 \to \cdots \to M^{\mathrm{an}} \otimes_S \Omega_Y^n \to 0\right]                                                                                              \\
                   & = \left[0 \to M^{\mathrm{an}} \to M^{\mathrm{an}} \otimes_{\mathcal{O}_{Y^{\mathrm{an}}}} \Omega_{Y^{\mathrm{an}}}^1 \to \cdots \to M^{\mathrm{an}} \otimes_{\mathcal{O}_{Y^{\mathrm{an}}}} \Omega_{Y^{\mathrm{an}}}^n \to 0\right].
\end{align*}

One should note that, no matter algebraic case or analytic case, the de Rham complex is always a complex of sheaves of $\mathbb{C}$-vector spaces.

Now, let's calculate $\mathrm{DR}(M_{\mathfrak{\alpha}})$.

\begin{proposition}\label{prop:classification-of-local-system}
    For definition of local system, see Definition~\ref{def:local-system}.

    Assume the underlying Euclidean topology of $Y^{\mathrm{an}}$ is connected, for a fixed base point $y\in Y^{\mathrm{an}}$, we
    we have a categorical equivalence:
    \begin{align*}
        \{\text{local systems on } Y^{\mathrm{an}}\} & \longleftrightarrow \{\text{representations of } \pi_1(Y^{\mathrm{an}},y)\} \\
        \mathcal{L}                                  & \longmapsto \mathcal{L}_y
    \end{align*}

    For reference see an English translation of \cite[Prop.~1.3., Cor.~1.4.]{DeligneLNM163}.
\end{proposition}

\begin{example}\label{ex:regular-holonomic-simple-module-in-dimension-1}
    $X = \mathbb{A}^1_{\mathbb{C}}$,  $M_{\alpha} := \mathbb{C}[x, \frac{1}{x}] \cdot x^{\alpha}$, an algebraic $\D_X$-module.
    \begin{enumerate}
        \item $\alpha \not\in \mathbb{Z}$, $\mathrm{DR}(M_{\alpha}) \cong [M_{\alpha} \xrightarrow{\mathrm{d}_{\alpha}} M_{\alpha}]$.
              \begin{enumerate}
                  \item \textbf{On the punctured disk } $X^\mathrm{an} \setminus \{0\}$:
                        The kernel of the connection describes the local system. Since
                        \[
                            \partial_{x}(x^{-\alpha} \cdot x^{\alpha}) = \partial_{x} \cdot 1 = 0,
                        \]
                        we have $\operatorname{Ker}(\partial_{x})|_{X^\mathrm{an} \setminus \{0\}} = \mathbb{C} \cdot x^{-\alpha} \cdot x^{\alpha}$. This implies that the kernel corresponds to the non-trivial local system $L_{\lambda}$ defined by the multivalued function $x^{-\alpha}$ (1-dimensional $\C$-vector space). The monodromy $T$ (action of the generator of $\pi_1(X^{\mathrm{an}} \setminus \{0\}, x_0) $) acts as follows:
                        \begin{align*}
                            T \cdot x^{-\alpha} & = T \cdot e^{-\alpha \log x}                                                   \\
                                                & := e^{-\alpha(\log x + 2\pi i)}                                                \\
                                                & = e^{-2\pi i \alpha} \cdot e^{-\alpha \log x}                                  \\
                                                & = \lambda \cdot x^{-\alpha}, \quad \text{where } \lambda = e^{-2\pi i \alpha}.
                        \end{align*}
                        Since $\partial_{x}$ is surjective on $X^\mathrm{an} \setminus \{0\}$, we obtain the quasi-isomorphism:
                        \[
                            \mathrm{DR}(M_{\alpha})|_{X^\mathrm{an} \setminus \{0\}} \overset{\text{qiso.}}{\simeq} L_{\lambda}[0].
                        \]

                  \item \textbf{The stalk/fiber At the origin } $\{0\}$:
                        For $\alpha \notin \mathbb{Z}$, the cohomology vanishes locally at the origin:
                        \[
                            H^{0}\mathrm{DR}(M_{\alpha})|_{\{0\}} = H^{1}\mathrm{DR}(M_{\alpha})|_{\{0\}} = 0.
                        \]
              \end{enumerate}


        \item $\alpha \in \mathbb{Z} \Rightarrow \lambda = 1$, $\mathrm{DR}(M_{\alpha}) \mid_{X^\mathrm{an} \setminus \{0\}} = \mathbb{C}_{X^\mathrm{an} \setminus \{0\}}$ (a 1-dimensional local system with trivial monodromy), $H^{0}\mathrm{DR}(M_{\alpha}) \cong H^{1}\mathrm{DR}(M_{\alpha}) \cong \mathbb{C}_{\{0\}}$ (skycraper sheaf).

        \item $N=\mathbb{C}[\delta]\simeq \mathbb{C}[\partial_{x}]$, $\mathrm{DR}(N)\overset{\text{qiso.}}{\simeq} \mathbb{C}_{\{0\}}[-1]$ (quasi-isomorphism).
    \end{enumerate}
\end{example}

\section{Basic version Riemman-Hilbert correspondence}

Let $Y$ be a complex manifold, $\mathcal{O}_Y$: the sheaf of holomorphic functions on $Y$, $\mathcal{L}$: a sheaf of $\mathbb{C}$-vector spaces.

\begin{definition}\label{def:local-system}
    $\mathcal{L}$ is called a $\textbf{local system}$ if $\forall y \in Y$, $\exists$ open neighborhood $U_y$ of $y$ such that $\mathcal{L}|_{U_y} \simeq\underline{\mathbb{C}}^r_{U_y}$, where $\underline{\mathbb{C}}^{r}_{U_y}$ is the constant sheaf on $U_y$.
\end{definition}

\begin{theorem}\label{thm:basic-RH-corr}
    Let $X = \text{Spec } R$ be a smooth affine algebraic variety over $\mathbb{C}$ and $M$ an integrable connection on $X$,
    then $$M^{\mathrm{an}} \simeq \mathcal{L} \otimes_{\underline{\mathbb{C}}_{X^{\mathrm{an}}}} \mathcal{O}_{X^{\mathrm{an}}}$$
    with $\mathcal{L} \subseteq M^{\mathrm{an}}$, the collection of flat sections of $M^{\mathrm{an}}$, which forms a local system.
\end{theorem}
\begin{remark}
    This theorem does NOT mean $M$ is regular holonomic (see Section~\ref{sec:regular-holonomic-d-module}) in algebraic sense, but $M^{\mathrm{an}}$ is indeed regular holonomic in analytic sense.
\end{remark}
\begin{remark}[Remarks on this version of Riemann-Hilbert correspondence]\label{rmk:basic-RH-corr}
    \mbox{}
    \begin{enumerate}
        \item $\mathcal{L}$ is the kernel of the connection $\nabla := \sum_{i} \partial_{x_i}\otimes \mathrm{d}x_i : M^{\mathrm{an}} \to M^{\mathrm{an}} \otimes_{\mathcal{O}_{X^{\mathrm{an}}}} \Omega_{X^{\mathrm{an}}}^1$, i.e. the sheaf of \textbf{flat sections} of $M^{\mathrm{an}}$.
        \item The $\D_{X^{\mathrm{an}}}$-module structure of $\mathcal{L} \otimes_{\underline{\mathbb{C}}_{X^{\mathrm{an}}}} \mathcal{O}_{X^{\mathrm{an}}}$ is to only apply differential operators on the second factor.
        \item With the Corollary~\ref{cor:basic-RH-corr}, we get the complete description Riemann-Hilbert correspondence, which claims that taking de Rham complexes between derived categories gives an categorical equivalence \cite[Thm.~7.2.1]{hotta2008d}.
        \item So the $\mathcal{L} \mapsto \mathcal{L} \otimes_{\underline{\mathbb{C}}_{X^{\mathrm{an}}}} \mathcal{O}_{X^{\mathrm{an}}}$ actually gives a quasi-inverse of de Rham functor in this basic (affine + integrable connection) case.
    \end{enumerate}
\end{remark}
\begin{proof}
    \quad

    Goal: $M$ is locally free, we want to show for small enough neighborhood, we can pick a basis $s_{1},\dots,s_{r}$ makes $\partial_{x_{i}}s_{j}=0$, thus
    We have known $M$ is a locally free $R$-module, $\forall x \in X$, take an open neighborhood $U \subset X$ such that $M|_U$ is free, then $M^{\mathrm{an}}|_{U^{\mathrm{an}}}$ is a free $\mathcal{O}_{U^{\mathrm{an}}}$-module, generated by $\underline{\mathfrak{m}} = (m_1, \cdots, m_r)$ (underline stands for vector presentation).

    Denote $(x_1, \cdots, x_n)$ coordinates of $U^{\mathrm{an}}$, then we have $\partial_{n} \underline{\mathfrak{m}} = X \cdot\underline{ \mathfrak{m}}$,
    where $X$ is a matrix of holomorphic functions on $U^{\mathrm{an}}$. Expand $X$ w.r.t. $x_n$ in a small enough neighborhood of
    $x\in V\subseteq U^{\mathrm{an}}$, we have $$X = \sum_{i\ge0} x_n^i \cdot X_i,$$where $X_i$ is a $r\times r$ matrix of holomorphic functions in $(x_1, \cdots, x_{n-1})$.

    Take $\Psi=\sum\limits_{i\ge 0} \frac{1}{i+1} x_{n}^{i+1}X_{i}$, so $\partial_{n}(\Psi)=X$, then $\Phi=\mathrm{\exp}(-\Psi)$, which is an invertible matrix. Set $\underline{s}=(s_{1},\dots,s_{r})=\Phi(\underline{\mathfrak{m}})$, so $\partial_{n}\underline{s}=\partial_{n}(\Phi)\cdot \underline{\mathfrak{m}}+\Phi \cdot(\partial_{n} \underline{\mathfrak{m}})=(\partial_{n}(\Phi)+\Phi \cdot X)\underline{\mathfrak{m}}=0$.

    Since $\Phi$ is an invertible matrix, we obtain $s_{1},\dots,s_{r}$ also a basis of $U^{\mathrm{an}}|_{V}$, so we have $M^{\mathrm{an}}|_{V}=\bigoplus\limits_{i=1}^{r}\mathcal{O}_{V}s_{i}$ such that $\partial_{n}s_{i}=0$.

    Want to clarify the relation of $s_{i}$ and other $\partial_{x_{j}}\, (j < n)$, then Take $M_1 = \ker[M^{\mathrm{an}}|_V \xrightarrow{\partial_n} M^{\mathrm{an}}|_V]=\left\{ \partial_{n}\left( \sum f_{i}s_{i} \right)=0:f_{i}\in \mathcal{O}_{V} \right\}=\left\{ \sum\limits_{i=1}^{n}\partial_{n}(f_{i})s_{i}=0 \right\}.$ So we get $\partial_n(f_i) = 0$ for all $i=1,\dots,r$, i.e., $f_{i}$ is independent of $x_{n}$.

    So we have $M_{1}=\bigoplus\limits_{i=1}^{r}\mathcal{O}_{V}^{n-1}s_{i}$, $\mathcal{O}_{V}^{n-1}$ are holomorphic functions that are independent of $x_{n}$, which suggests that we do not need the variable $x_{n}$. And do the same thing for $M_{1}$,we get a new basis with $\partial_{n}s_{i}=\partial_{n-1}s_{i}=0$ for all $i=1,\dots,r$, and the change of basis is via an invertible matrix of homomorphic functions in $(x_{1},\dots,x_{n-1})$.

    By induction, we finally obtain:  $$ M_n = \bigoplus_{i=1}^r \mathbb{C} \cdot s_i $$with $\partial_j(s_i) = 0 \, (\forall i, j)$. After a sequence of invertible changes of basis, we know that for a small enough neighborhood $V \subseteq U^{\mathrm{an}}$, we get $M^{\mathrm{an}}|_{V}=\bigoplus\limits_{i=1}^{r}\mathcal{O}_{V}s_{i}$, with $\partial_{j}(s_{i})=0$ .
\end{proof}

\section{Regular holonomic $\mathcal{D}$-module}\label{sec:regular-holonomic-d-module}

Let $X = \text{Spec } R$ be a smooth affine algebraic variety over $\mathbb{C}$ and $M$ a holonomic $\mathcal{D}_X$-mod.
For any point $x \in X^{\mathrm{an}}$, we consider the embedding:
$$
    i: \Delta \hookrightarrow X^{\mathrm{an}}, \quad 0 \mapsto x.
$$
where $\Delta$ is a 1-dimensional small disk in the complex plane.
Then we have a morphism of sheaves of rings:
$$
    \mathcal{O}_{X^{\mathrm{an}}} \to \mathcal{O}_{\Delta}, \quad f \mapsto f|_{\Delta}.
$$
We define $i^*M^{\mathrm{an}} = M^{\mathrm{an}} \otimes_{\mathcal{O}_{X^{\mathrm{an}}}} \mathcal{O}_{\Delta}$.

\begin{definition}[Analytic regular holmonicity]\label{def:analytic_regular_holmonicity}
    \mbox{}

    $M^{\mathrm{an}}$ is called $\textbf{regular holonomic}$, if $i^{\ast}M^{\mathrm{an}}$ is generated by $M_{\alpha}^{\mathrm{an}}|_{\Delta}$ and $N^{\mathrm{an}}|_{\Delta}$ in $\mathbf{K}_{0}(\mathcal{D}_{\Delta})$ for every small enough disk embedding $i:\Delta \hookrightarrow X^{\mathrm{an}}$.
\end{definition}

\bigskip

We justify this definition below:

\begin{remark}[Explanation of this definition]\label{rmk_definition_of_regular_holonomicity_jusitfied}
    \mbox{}

    There are two kinds of regular holonomic $\D$-module, algebraic and analytic. And there are also two kinds of \textbf{holonomic} $\D$-module, algebraic and analytic.

    In this remark, we investigate definition of analytic regular holonomic $\D$-module the first, and algebraic regular holonomic $\D$-module the second.


    \begin{definition}[Good filtration on a complex manifold]\label{def:good_filtration_on_complex_manifold}
        See \cite[Definition~1.4.1.]{Bjork1993AnalyticDModules}.

        On a complex manifold $X$, $\D_X$ is a coherent and noetherian sheaf of rings.
        The coherence gives the abelian category whose objects are coherent left $\D_X$-modules.

        And we have the positive filtration
        $\{\D_X(v)\}$ and begin to discuss filtered $\D_X$-modules, see \cite[Definition~1.2.1]{Bjork1993AnalyticDModules},
        just the same as \cite[Definition~3.1]{chen2024dmodules} or \cite[\href{https://stacks.math.columbia.edu/tag/09CI}{Tag 09CI}]{stacks-project},
        which is \textbf{coordinate-free}.

        A \textbf{good filtration} on a coherent left $\D_X$-module $\mathcal{M}$ consists of
        a sequence of subsheaves $\{\mathcal{M}_v\}$ such that every $x_0 \in X$ has an open neighborhood
        $U$ where
        \[
            \mathcal{M}_v = \D_X(v - k_1) m_1 + \ldots + \D_X(v - k_s) m_s.
        \]
        Here $m_1, \ldots, m_s \subset \mathcal{M}(U)$ and $k_1, \ldots, k_s$ are integers.
        In other words, the filtration is finite \textbf{locally generated by sections}, see \cite[\href{https://stacks.math.columbia.edu/tag/01AL}{Tag 01AL}, \href{https://stacks.math.columbia.edu/tag/01AN}{Tag 01AN}, \href{https://stacks.math.columbia.edu/tag/01B2}{Tag 01B2}]{stacks-project}.
    \end{definition}

    \begin{proposition}[Holonomic $\mathcal{D}$-module have finite length (algebraic and analytic)]
        For algebraic case, see \cite[Prop.~3.1.2.(ii)]{hotta2008d}.
        For analytic case, see \cite[Prop.~3.1.5.]{Bjork1993AnalyticDModules}.
    \end{proposition}
    \begin{proof}[Sketch]
        In both case, this especially due to for an exact sequence of coherent $\mathcal{D}_X$-modules:
        \[
            0 \rightarrow L \rightarrow M \rightarrow N \rightarrow 0
        \]
        we have
        \[
            \CC(M) = \CC(L) + \CC(N).
        \]
        For definition of \textbf{Characteristic variety} for a coherent $\D_X$-module on complex manifold, see \cite[Definition~1.6. or 1.6.16]{Bjork1993AnalyticDModules}.
        And for definition of \textbf{Characteristic cycle}, see \cite[8.2.8 Characteristic cycles.]{Bjork1993AnalyticDModules}.
    \end{proof}

    \begin{definition}[Regularity defined on stalks in dimension 1]
        For definition of a \textbf{Regular holomorphic} $\D_{X,x_0}$-module complex $M^\bullet$ on a \textbf{1-dimensional} complex manifold, see \cite[Prop.~5.2.5.]{Bjork1993AnalyticDModules}, that is the chain map
        \[
            R\Hom_{\D_{X,x_0}}(M^\bullet,\mathcal{O}_{X,x_0})\rightarrow  R\Hom_{\D_{X,x_0}}(M^\bullet,\widehat{\mathcal{O}_{X,x_0}})
        \]
        is an isomorphism.
    \end{definition}

    \begin{remark}[Indecomposable objects in $\mathbf{RH}(\mathcal{D})$]\label{rmk:indecomposable_objects_in_RH_D}
        \cite[5.2.17 Indecomposable objects in $\textbf{RH}(\mathcal{D})$.]{Bjork1993AnalyticDModules}
        A regular holonomic $\mathcal{D}$-module $\mathcal{M}$ is decomposed into a direct sum
        \[
            \mathcal{M} = \bigoplus_{\alpha \in \mathcal{F}} \mathcal{M}_\alpha
        \]
        where $\mathcal{F} \subset \Lambda$ is a finite set and $V_0(\mathcal{M}_\alpha)$ is generated by a subspace annihilated
        by a power of $(\nabla - \alpha)$. If $\alpha \neq 0$ it is easy to check that $\mathcal{M}_\alpha$ is a direct sum
        of indecomposable Deligne modules given by $\mathcal{D}/\mathcal{D}(\nabla - \alpha)^k$ where $k$ is a positive
        integer.

        If $\alpha = 0$ one refers to $\mathcal{M}_0$ as the \textbf{unipotent} part. If $\mathcal{M}_0$ is non-zero it is a direct
        sum of indecomposable modules which occur in Proposition~\ref{prop:classification-unipotent-indecomposable-regular-holonomic} below.
    \end{remark}

    \begin{proposition}[Classification of unipotent indecomposable regular holonomic $\D :=\mathcal{D}_1$-modules]\label{prop:classification-unipotent-indecomposable-regular-holonomic}
        \cite[Prop~5.2.18.]{Bjork1993AnalyticDModules}
        $\D_1$ is a stalk of $\D_X$ for $X$ a 1 dimensional complex manifold.

        Let $\mathcal{M}$ be a unipotent and indecomposable regular holonomic $\mathcal{D}$-module, and $\nabla := x\partial$,
        Then $\mathcal{M}$ is isomorphic to one of the following modules:
        \begin{enumerate}
            \item $\mathcal{D}/\mathcal{D}(\nabla + 1)^k; k \geq 1$
            \item $\mathcal{D}/\mathcal{D}\nabla^k$
            \item $\mathcal{D}/\mathcal{D}x$
            \item $\mathcal{D}/\mathcal{D}\partial$
            \item $\mathcal{D}/\mathcal{D}(\nabla + 1)^k\partial$
            \item $\mathcal{D}/\mathcal{D}x(\nabla + 1)^k$
            \item $\mathcal{D}/\mathcal{D}\nabla^k x$
        \end{enumerate}
        where $k \geq 1$ in (5-7).
    \end{proposition}

    \begin{definition}[Regular holonomic $\mathcal{D}$-module on complex manifold]\label{def:Bjork_regular_holmonicity}
        \cite[Definition~5.3.1.]{Bjork1993AnalyticDModules}
        For every point $x_0$ in $X$, $M_{x_0}$ we have
        \[
            R\Hom_{\D_{X,x_0}}(M^\bullet,\mathcal{O}_{X,x_0})\rightarrow  R\Hom_{\D_{X,x_0}}(M^\bullet,\widehat{\mathcal{O}_{X,x_0}})
        \]
        is an isomorphism, note $X$ can have higher dimension.
    \end{definition}

    \begin{proposition}[Equivalent definition on complex manifold]
        In \cite[Definition~5.2.]{Kashiwara1984RiemannHilbertHolonomic},
        they define regular holomonic $\D_X$-module $M$ as there \textbf{globally} exists a good filtration
        (usually in analytic case a \textbf{good filtration} can only be achieved locally) $\{M_i\}_{i\in\mathbb{Z}}$ such that $f\gr M = 0$ for any $f\in\gr\D_X$
        vanishing on $\Ch(M)$.

        This article cite \cite[Thm.~6.3.1 and Thm.~6.4.1]{KashiwaraKawai1981HolIII} to show the equivalence of two definition of
        regular holmonicity (with Def.~\ref{def:Bjork_regular_holmonicity}). This can be also seen in \cite[Rmk.~7.3.2.]{hotta2008d}.

        Since taking stalks is a local notion, we can also reformulate \cite[Definition~5.2.]{Kashiwara1984RiemannHilbertHolonomic} as
        the there \textbf{locally} exists such a good filtration, see Def.~\ref{def:good_filtration_on_complex_manifold} and \cite[Def.~6.1.8.]{hotta2008d}
    \end{proposition}

    \begin{remark}
        For the definition of annihilating ideal on complex manifold, see \cite[1.6.11~Characteristic~ideals. and 1.6.15~Remark.]{Bjork1993AnalyticDModules}.

        For vanishing ideal of a given \textbf{analytic set}, see \cite[A:V.l.~THE~CATEGORY~OF~O-MODULES]{Bjork1993AnalyticDModules}.

        And one should also see the theorem called \textbf{Analytic Nullstellensatz}.
    \end{remark}

    \begin{definition}[Derived inverse image functor on complex manifold]
        \cite[Def.~2.3.6.]{Bjork1993AnalyticDModules}
        Let $F: Y \to X$ be a holomorphic map. The derived inverse image functor from $D^b(\D_X)$ into $D^b(\D_Y)$ is denoted by $F^+$ where
        \[
            F^+ \mathcal{M} = \D_{Y \to X} \overset{L}{\otimes}_{F^{-1}(\D_X)} F^{-1}(\mathcal{M})
        \]
        for every $\mathcal{M} \in D^b(\D_X)$.
        We have defined \textbf{Transfer bimodules}, see \ref{example:side-changing-transfer-bimodules}.
        We also defined the \textbf{Inverse image functor} in Rmk.~\ref{rem:kashiwara_explicit}.
    \end{definition}

    \begin{definition}[Curve regularity on complex manifold]
        \cite[Def.~5.6.1.]{Bjork1993AnalyticDModules}
        A holonomic complex $\mathcal{M}$ is \textbf{curve regular} if the following hold:
        To every pair $(\alpha, C)$, where $C$ is a 1-dimensional complex manifold and $\alpha: C \to X$ a holomorphic map, it follows that $\alpha^+(\mathcal{M}) \in D^b_{\mathrm{r.h}}(\D_C)$.
        The family of curve regular holonomic complexes on $X$ is denoted by $\mathcal{C}^b_{\mathrm{r.h}}(\D_X)$.
    \end{definition}

    \begin{theorem}[Curve regularity equals regularity on complex manifold]\label{thm:curve_regularity_equals_regularity_on_complex_manifold}
        \cite[Thm.~5.6.2.]{Bjork1993AnalyticDModules}
        For every complex manifold $X$ one has $\mathcal{C}^b_{\mathrm{r.h}}(\D_X) = D^b_{\mathrm{r.h}}(\D_X)$.
    \end{theorem}

    Curve test Thm.~\ref{thm:curve_regularity_equals_regularity_on_complex_manifold},
    definition of regular holonomic on complex manifold Def.~\ref{def:Bjork_regular_holmonicity},
    the decomposition Rmk.~\ref{rmk:indecomposable_objects_in_RH_D},
    and the list Prop.~\ref{prop:classification-unipotent-indecomposable-regular-holonomic}.
    Combined together, they justified the definition given in Def.~\ref{def:analytic_regular_holmonicity}.

    \bigskip

    This is the second part, the regular holonomicity of $\D$-modules on algebraic varieties.

    \begin{definition}[Regular holonomic $\mathcal{D}_X$-module on algebraic variety]
        \cite[Def.~6.1.1.]{hotta2008d}
        Let $X$ be a smooth algebraic variety.
        We say that a holonomic $\mathcal{D}_X$-module $M$ is \textbf{regular}
        if any composition factor of $M$ is isomorphic to the minimal extension $L(Y, N)$ of some regular integrable connection $N$
        on a locally closed smooth subvariety $Y$ of $X$ such that the inclusion $Y \to X$ is affine (see \cite[Theorem~3.4.2]{hotta2008d}).
    \end{definition}

    \begin{theorem}[\protect{\cite[Thm.~6.1.12.]{hotta2008d}}]\label{thm:equivalence_regular_holonomic_on_algebraic_with_analytic}
        One can show the following using a result in Kashiwara--Kawai \cite{KashiwaraKawai1981HolIII}.

        Let $X$ be a smooth algebraic variety,
        and let $M$ be a holonomic $\D_X$-module.
        Take a divisor completion $j: X \hookrightarrow \overline{X}$ of $X$.
        Then $M$ is regular if and only if $(j_\ast M)^{\mathrm{an}}$ is regular.
    \end{theorem}

    \begin{remark}
        For an open embedding the $\D$-module theoretic direct image reduce to a sheaf theoretic one, but still remains in derived category,
        see \cite[Example~1.5.22.]{hotta2008d}.
    \end{remark}

    \begin{remark}[Equivalent definition of regular holonomic $\mathcal{D}_X$-module on algebraic variety]\label{rmk:algebraic_KK_regularity_holomonic}
        In the algebraic one we also encounter the KK(Kashiwara and Kawai)-style definition of regular holonomicity.

        \begin{enumerate}
            \item \cite[Def.~21.2.]{schnell_dmodules_notes}
                  Let $X$ be a nonsingular algebraic variety that is \textbf{proper} over $\Spec k$.
                  A holonomic left $\mathcal{D}_X$-module $\mathcal{M}$ is called \textbf{regular} (in the sense of Kashiwara and Kawai)
                  if it admits a good filtration $F_\bullet \mathcal{M}$ such that $\mathcal{I}_{\Ch(\mathcal{M})} \cdot \gr^F \mathcal{M} = 0$.

                  Here $\mathcal{I}_{\Ch(\mathcal{M})}\subseteq\mathcal{O}_{T^\ast(X)}$ is defined as:
                  \[
                      \mathcal{I}_{\Ch(\mathcal{M})}:= \sqrt{\vphantom{\big|}\mathrm{Ann}_{\gr \D_X}\gr \mathcal{M}}
                  \]

            \item \cite[Def.~21.5.]{schnell_dmodules_notes}
                  Let $X$ be a nonsingular algebraic variety.
                  A holonomic left $\mathcal{D}_X$-module $\mathcal{M}$ is called \textbf{regular} (in the sense of Kashiwara and Kawai)
                  if, for any affine open embedding $j: X \hookrightarrow \overline{X}$ into a nonsingular algebraic variety $\overline{X}$
                  that is proper over $\Spec k$, the direct image $j_+ \mathcal{M}$ is regular on $\overline{X}$.

                  We will define the $\D$-module theoretic \textbf{direct image} in Rmk.~\ref{rem:kashiwara_explicit}.
        \end{enumerate}

        We here prove the equivalence of this with Def.~\cite[Def.~6.1.1.]{hotta2008d}.
    \end{remark}
    \begin{proof}
        \mbox{}

        \begin{enumerate}
            \item KK-regular $\implies$ Hotta-regular:

                  By Def.~\cite[Def.~21.5.]{schnell_dmodules_notes}, there exists a good filtration $F$ on $j_\ast\mathcal{M}$ (which is a coherent $\D_{\overline{X}}$-module since $\overline{X}$ is proper) such that $\mathcal{I}_{\Ch(j_\ast\mathcal{M})} \cdot \gr^F (j_\ast\mathcal{M}) = 0$.
                  We claim this property remains true after analytification.

                  One can verify for any point $p \in \overline{X}$, the natural map on stalks $\mathcal{O}_{\overline{X},p} \to \mathcal{O}_{\overline{X}^\mathrm{an},p}$ is faithfully flat.
                  Since the formation of annihilators commutes with flat base change (see \cite[\href{https://stacks.math.columbia.edu/tag/0250}{Tag 0250}]{stacks-project} and \cite[\href{https://stacks.math.columbia.edu/tag/07T8}{Tag 07T8}]{stacks-project}), and the characteristic variety is compatible with analytification, we have locally on $\overline{X}^\mathrm{an}$:
                  \[
                      \mathcal{I}_{\Ch((j_\ast\mathcal{M})^\mathrm{an})} \cdot \gr^{F^\mathrm{an}} (j_\ast\mathcal{M})^\mathrm{an} = 0.
                  \]
                  This implies that $(j_\ast \mathcal{M})^\mathrm{an}$ is regular holonomic on the complex manifold $\overline{X}^\mathrm{an}$ (in the sense of Kashiwara--Kawai).
                  Finally, by Theorem~\ref{thm:equivalence_regular_holonomic_on_algebraic_with_analytic}, we conclude that $\mathcal{M}$ is regular holonomic in the algebraic sense (Def.~\cite[Def.~6.1.1.]{hotta2008d}).

            \item Hotta-regular $\implies$ KK-regular:

                  That is equivalent to a proposition we will prove below:
                  \begin{proposition}
                      Let $X$ be a smooth $\C$-variety, and let $i:Y\hookrightarrow X$ be an \textbf{affine} \cite[\href{https://stacks.math.columbia.edu/tag/01S6}{Tag 01S6}]{stacks-project}
                      locally closed immersion with $Y$ smooth and connected.
                      Let $N$ be a simple \textbf{regular integrable connection}
                      on $Y$ \cite[Thm.~5.2.17, Thm.~5.2.20, Lem.~5.3.1, and Prop.~5.3.8.]{hotta2008d}.
                      Then the minimal extension $L_X(Y,N)$ is a KK-regular
                      holonomic $\mathcal D_X$-module.
                  \end{proposition}

                  \begin{proof}
                      \textbf{Notation.}

                      If $j:U\hookrightarrow V$ is an \textbf{affine} open immersion, write $j_*$ and $j_!$ for the
                      direct images of holonomic $\mathcal D$-modules, and $j_{!*}:=\mathrm{Im}(j_!\to j_*)$
                      (affine-ness guarantee higher cohomology vanishes, see \cite[Sect.~3.4]{hotta2008d})
                      for the \textbf{intermediate extension/minimal extension} \cite[Thm.~3.2.16]{hotta2008d}.
                      We don't introduce the definition of $j_!$, one can see \cite[Def.~3.2.13]{hotta2008d}.

                      If $f:V\to W$ is a morphism between smooth varieties, write $f_+$ for the \textbf{derived
                          direct image} of $\mathcal D$-modules, and $f^\dagger$ for (derived) \textbf{shifted inverse image},
                      see Rmk.~\ref{rem:kashiwara_explicit}

                      \medskip

                      \textbf{Step 1: SNC compactification of $Y$.}

                      By Nagata compactification \cite[\href{https://stacks.math.columbia.edu/tag/0ATT}{Tag 0ATT}, \href{https://stacks.math.columbia.edu/tag/0F41}{Tag 0F41}]{stacks-project},
                      there exists an open immersion $j_0:Y\hookrightarrow \overline Y_0$
                      with $\overline Y_0$ proper over $k$.

                      We perform \emph{``Resolution of singularities, III''} functor $\mathcal{BR}$ in \cite[Thm.~36]{kollar-seattle} on $\overline Y_0$ to obtain
                      a resolution of singularities
                      \[
                          \Pi_1: \overline Y_1 \to \overline Y_0
                      \]
                      such that
                      \begin{enumerate}
                          \item $\Pi_1$ is an isomorphism over the smooth locus of $\overline Y_0$,which means, as a morphism, it has a \textbf{rational map} inverse.
                          \item $\Pi_1$ is a composition of blow-ups.
                      \end{enumerate}

                      Since $\Pi_1$ is a compose of blow-ups, it is a \textbf{proper} morphism \cite[\href{https://stacks.math.columbia.edu/tag/02NS}{Tag 02NS}, \href{https://stacks.math.columbia.edu/tag/01WC}{Tag 01WC}]{stacks-project}.
                      Since peoperness is preserved by composition, we have $\overline Y_1$ is proper over $\Spec \C$.

                      Composing $Y\to\overline Y_0$ with the rational map inverse of $\Pi$, we get a morphism
                      \[
                          j_1: Y \to \overline Y_1
                      \]
                      such that $\Pi_1 \circ j_1 = j_0$.

                      Then we perform \emph{``Principalization, IV''} functor $\mathcal{BP}$ in \cite[Thm.~35]{kollar-seattle} on $\overline Y_1$ and the ideal sheaf
                      of $\overline Y_1\setminus Y$ to obtain
                      a resolution of singularities
                      \[
                          \Pi_2: \overline Y \to \overline Y_1
                      \]
                      such that
                      \begin{enumerate}
                          \item The pull-back $\Pi_2^*(I) \subset \mathcal{O}_{\overline Y}$ is the ideal sheaf of a simple normal crossing divisor.
                          \item $\Pi_2 : \overline Y \to \overline Y_1$ is an isomorphism over $\overline Y_1 \setminus D_1$.
                      \end{enumerate}

                      Due to the same reason as before, $\Pi_2$ is a proper morphism, and $\overline Y$ is proper over $\Spec \C$.

                      We compose $j_1: Y\to \overline Y_1$ with the rational map inverse of $\Pi_2$ to obtain a morphism
                      \[
                          j: Y \to \overline Y
                      \]
                      such that $\Pi_1 \circ \Pi_2 \circ j = j_0$.

                      Then we finally get $D := \overline Y\setminus Y$ is a simple normal crossing divisor.
                      In particular, $D$ is an \textbf{effective Cartier divisor} on the smooth variety $\overline Y$.
                      Therefore, the open immersion $j: Y \to \overline Y$ is an \textbf{affine morphism} \cite[\href{https://stacks.math.columbia.edu/tag/01WR}{Tag 01WR}, \href{https://stacks.math.columbia.edu/tag/01WS}{Tag 01WS}]{stacks-project}.
                      So $\D$-module theoretic derived direct image $j_*$ and $j_!$ reduced to a non-derived one.

                      \medskip

                      \textbf{Step 2: KK-regularity on the proper SNC compactification.}

                      Consider the $\mathcal D_{\overline Y}$-module $j_*N$ on $\overline Y$.
                      We get $N$ is regular along $D$ \cite[Thm~5.3.8. and Thm~5.2.20.]{hotta2008d},
                      which is compatible with the definition given in \cite{cailotto_fiorot_2015_algebraic_connections_vs_dmodules},
                      since both of them cited \cite{DeligneLNM163} for definition of a \textbf{regular integrable connection}.

                      Then \cite[Thm.~3.1.]{cailotto_fiorot_2015_algebraic_connections_vs_dmodules} gives that
                      $j_*N$ is regular holonomic in the Kashiwara--Kawai sense,
                      i.e. we have a good filtration on $j_*N$ such that
                      \[
                          I_{\mathrm{Ch}(j_*N)}\cdot \mathrm{gr}(j_*N)=0.
                      \]

                      Let
                      \[
                          M_{\overline Y} := j_{!*}N \;=\; \mathrm{Im}\bigl(j_!N \to j_*N \bigr).
                      \]
                      Since the subcategory of regular holonomic $\mathcal D$-modules is Serre (thick),
                      it is stable under subquotients; therefore $M_{\overline Y}$ is KK-regular as well.
                      (In the literature, thickness of KK-regularity is mentioned on a complex manifold \cite[Discussion~below~Def.~4.1.1.]{KashiwaraSchapira_RegIrrDmod_2015}.)

                      \medskip

                      \textbf{Step 3: producing a proper map to $X$ extending $i$.}
                      We now relate the open embedding $j:Y\hookrightarrow \overline Y$ to the locally
                      closed immersion $i:Y\hookrightarrow X$ (these are different maps).

                      Since we already have a (inclusion) morphism (in the definiton of \textbf{minimal extension})
                      $i: Y\hookrightarrow X$. We have a graph morphism $\Gamma_i = (i, \mathrm{id}_Y)_{\Spec \C}$, which is induced by the universal property of the fiber product $Y \times X$:
                      \[
                          \begin{tikzcd}
                              Y \arrow[drr, "i", bend left=20]
                              \arrow[ddr, "\mathrm{id}_Y"', bend right=20]
                              \arrow[dr, "\exists! \Gamma_i", dashed] & & \\
                              & Y \times_{\Spec \C} X \arrow[r, "p_2"] \arrow[d, "p_1"] \arrow[dr, phantom, "\lrcorner", very near start] & X \arrow[d] \\
                              & Y \arrow[r] & \Spec \C
                          \end{tikzcd}
                      \]
                      Since $i$ is an \textbf{immersion} \cite[\href{https://stacks.math.columbia.edu/tag/01IO}{Tag 01IO}]{stacks-project}, it is separated; hence its graph $\Gamma_i$ is also locally closed, i.e., an immersion.

                      We can also induce a morphism $(j\circ p_1, \mathrm{id}_X\circ p_2)_{\Spec \C}: Y\times_{\Spec \C}X \to \overline Y \times_{\Spec \C} X$.

                      By composition, we get $(j\circ p_1, \mathrm{id}_X\circ p_2)_{\Spec \C}\circ \Gamma_i: Y\to \overline Y \times_{\Spec \C} X$, which is also locally closed.
                      We denote $(j\circ p_1, \mathrm{id}_X\circ p_2)_{\Spec \C}\circ \Gamma_i$ by $\overline{\Gamma}_i$.

                      \begin{lemma}[Graph-closure resolution produces $Y\hookrightarrow\widetilde Y\to X$]
                          There exist a smooth $k$-variety $\widetilde Y$, an open immersion
                          \[
                              \widetilde j:Y\hookrightarrow \widetilde Y
                          \]
                          and morphisms
                          \[
                              f:\widetilde Y\to \overline Y,\qquad \pi:\widetilde Y\to X
                          \]
                          such that:
                          \begin{enumerate}
                              \item $\pi$ is proper;
                              \item $f\circ \widetilde j = j$ and $\pi\circ \widetilde j = i$.
                          \end{enumerate}
                      \end{lemma}

                      \begin{proof}[Proof of the lemma]

                          $\Gamma_i:Y\to Y\times X$ is a immersion (we omit the $\Spec \C$ subscript).
                          We view $\overline{\Gamma}_i(Y)$ as a locally closed subscheme of $\overline Y\times X$,
                          and let $\overline{\Gamma}$ be its \textbf{schematic closure} \cite[\href{https://stacks.math.columbia.edu/tag/01R8}{Tag 01R8}]{stacks-project} in $\overline Y\times X$.

                          Because $\overline Y$ is proper over $\C$, the projection $p_2:\overline Y\times X\to X$ is proper \cite[\href{https://stacks.math.columbia.edu/tag/01W5}{Tag 01W5}, \href{https://stacks.math.columbia.edu/tag/01W6}{Tag 01W6}]{stacks-project}.
                          Since $\overline{\Gamma}$ is a closed subscheme of $\overline Y\times X$, the inclusion $i:\overline{\Gamma} \to \overline Y \times X$ is a closed immersion, which is also proper \cite[\href{https://stacks.math.columbia.edu/tag/01W5}{Tag 01W5}]{stacks-project}.
                          Hence the induced morphism $\overline{\Gamma}\to X$ is proper as a composition of proper morphisms $p_2\circ i$.

                          Take a resolution of singularities ($\mathcal{BR}$ in \cite[Thm.~36]{kollar-seattle}) $\mu:\widetilde Y\to \overline{\Gamma}$ which is an isomorphism
                          over the smooth open subset $\Gamma_i(Y)\simeq Y$.
                          Let $\widetilde j:Y\hookrightarrow \widetilde Y$ be the (rational map) inverse of this isomorphism onto its image.
                          Define
                          \[
                              \pi := p_2\circ \iota\circ \mu:\widetilde Y\to X,\qquad
                              f := p_1\circ \iota\circ \mu:\widetilde Y\to \overline Y,
                          \]
                          where $\iota:\overline{\Gamma}\hookrightarrow \overline Y\times X$ is the closed immersion and
                          $p_1$ is the projection to $\overline Y$.
                          Then $\pi$ is proper and $f\circ \widetilde j = j$ and $\pi\circ \widetilde j = i$ as required.
                      \end{proof}

                      \medskip

                      \textbf{Step 4: pullback and proper pushforward preserve KK-regularity.}
                      Set
                      \[
                          M^\bullet_{\widetilde Y} := f^\dagger(M_{\overline Y}) \in D^b_{\mathrm{hol}}(\mathcal D_{\widetilde Y}).
                      \]
                      Derived inverse image preserves KK-regularity, by definition, each cohomology of the complex $M^\bullet_{\widetilde Y}$ is KK-regular.
                      Moreover, since $f\circ \widetilde j=j$, we have
                      \[
                          \widetilde j^\dagger M^\bullet_{\widetilde Y} = \widetilde j^\dagger f^\dagger M_{\overline Y} \simeq j^\dagger M_{\overline Y} \simeq N.
                      \]
                      The last isomorphism holds because for an \textbf{open immersion} $j: Y \hookrightarrow \overline Y$,
                      the \textbf{shifted derived inverse image} $j^\dagger$ (which includes a shift by $\dim Y - \dim \overline Y = 0$)
                      simply coincides with the usual sheaf restriction $j^{-1}$ (or $|_Y$) \cite[Example~1.5.12]{hotta2008d}.
                      Since $M_{\overline Y} = j_{!*}N = \mathrm{Im}(j_!N \to j_*N)$ is the minimal extension (intermediate extension) of $N$ from $Y$.
                      Because the restriction functor to an open subset $j^\dagger = j^{-1}$ is an \textbf{exact functor} \cite[\href{https://stacks.math.columbia.edu/tag/01AJ}{Tag 01AJ}]{stacks-project} on the category of $\D$-modules, it commutes with taking images.
                      Therefore, we have:
                      \[
                          j^\dagger M_{\overline Y} = j^{-1} \mathrm{Im}(j_!N \to j_*N) \simeq \mathrm{Im}(j^{-1}j_!N \to j^{-1}j_*N).
                      \]
                      At this point we have to point out the definition of $j_!$:
                      \[
                          j_! := \int_{j_!} := \mathbb{D}_{\overline Y}\circ \int_{j} \circ \mathbb{D}_{Y}
                      \]
                      where the duality functor $\mathbb{D}_X : D^-(\mathcal D_X) \to D^+(\mathcal D_X)^{\mathrm{op}}$ is defined as \cite[Def.~2.6.1]{hotta2008d}:
                      \begin{align*}
                          \mathbb{D}_X(M^\bullet) & := R\mathcal{H}om_{\mathcal D_X}(M^\bullet, \mathcal D_X) \otimes_{\mathcal O_X} \Omega_X^{\otimes -1}[\dim X]      \\
                                                  & \simeq R\mathcal{H}om_{\mathcal D_X}(M^\bullet, \mathcal D_X \otimes_{\mathcal O_X} \Omega_X^{\otimes -1}[\dim X]).
                      \end{align*}
                      Firstly, $j^{-1}j_* = \mathrm{id}$ for a $\mathcal{D}_Y$-module $N$.

                      Secondly, we verify $j^{-1}j_!N \simeq N$ by explicitly expanding the definition of the duality functor $\mathbb{D}$:
                      \begin{align*}
                           & j^{-1} \mathbb{D}_{\overline Y}(M^\bullet)                                                                                                                                                                                                          \\
                           & = j^{-1} \left( R\mathcal{H}om_{\mathcal D_{\overline Y}}(M^\bullet, \mathcal D_{\overline Y}) \otimes_{\mathcal O_{\overline Y}} \Omega_{\overline Y}^{\otimes -1}[\dim \overline Y] \right) &  & (\text{definition of } \mathbb{D}_{\overline Y}) \\
                           & \simeq j^{-1}R\mathcal{H}om_{\mathcal D_{\overline Y}}(M^\bullet, \mathcal D_{\overline Y}) \otimes_{\mathcal O_Y} j^{-1}\Omega_{\overline Y}^{\otimes -1}[\dim Y]                            &  & (\text{exact $j^{-1}$ commutes with } \otimes)   \\
                           & \simeq R\mathcal{H}om_{\mathcal D_Y}(j^{-1}M^\bullet, j^{-1}\mathcal D_{\overline Y}) \otimes_{\mathcal O_Y} j^{-1}\Omega_{\overline Y}^{\otimes -1}[\dim Y]                                                                                        \\
                           & (\text{$M^{\bullet}$ may not be locally free, but $R\mathcal{H}om$ only consider cohomology})                                                                                                                                                       \\
                           & \simeq R\mathcal{H}om_{\mathcal D_Y}(j^{-1}M^\bullet, \mathcal D_Y) \otimes_{\mathcal O_Y} \Omega_Y^{\otimes -1}[\dim Y]                                                                      &  & (\text{canonical bundle is locally free})        \\
                           & \simeq \mathbb{D}_Y(j^{-1}M^\bullet).                                                                                                                                                         &  & (\text{definition of } \mathbb{D}_Y)
                      \end{align*}
                      This explicitly proves that $j^{-1}$ commutes with duality. Now, applying this to the definition of $j_! N := \mathbb{D}_{\overline Y}(j_*(\mathbb{D}_Y(N)))$:
                      \begin{align*}
                          j^{-1}(j_! N) & = j^{-1} \bigl( \mathbb{D}_{\overline Y} ( j_*( \mathbb{D}_Y(N) ) ) \bigr) \\
                                        & \simeq \mathbb{D}_Y \bigl( j^{-1} j_*( \mathbb{D}_Y(N) ) \bigr).
                      \end{align*}
                      Since we just established $j^{-1} j_* = \mathrm{id}$, the inner term $j^{-1} j_*(\mathbb{D}_Y(N))$
                      simplifies to exactly $\mathbb{D}_Y(N)$. This leaves, and by \cite[Prop.~2.6.5]{hotta2008d}:
                      \[
                          j^{-1}(j_! N) \simeq \mathbb{D}_Y \bigl( \mathbb{D}_Y(N) \bigr) \simeq N.
                      \]
                      Hence, the map becomes $\mathrm{Im}(N \xrightarrow{\mathrm{id}} N)$, which is clearly just $N$.
                      So the complex restricts to a single module $N$ on $Y$ (concentrated in degree 0).

                      Consider the derived direct image
                      \[
                          K^\bullet := \pi_+(M^\bullet_{\widetilde Y}) \in D^b_{\mathrm{hol}}(\mathcal D_X).
                      \]
                      Proper direct image preserves KK-regularity; hence each cohomology sheaf
                      $\mathcal H^m(K^\bullet)$ is a KK-regular holonomic $\mathcal D_X$-module.
                      Let
                      \[
                          K := \mathcal H^0(K^\bullet).
                      \]

                      \medskip
                      \textbf{Step 5: extract the composition factor $L_X(Y,N)$.}
                      Let $Z:=\overline{i(Y)}\subseteq X$ and let $\eta$ be the generic point of $Z$.
                      Since $\widetilde Y\setminus Y$ has codimension $\ge 1$ in $\widetilde Y$,
                      its image $\pi(\widetilde Y\setminus Y)$ cannot contain $\eta$.
                      As $\pi$ is proper, $\pi(\widetilde Y\setminus Y)$ is closed. Moreover, since $i(Y)$ is locally closed in $X$, it is an open subset of its closure $Z$. Hence there exists
                      an \textbf{open} neighborhood $U\subseteq X$ of $\eta$ such that
                      \[
                          U\cap Z\subseteq i(Y)\quad\text{and}\quad U\cap \pi(\widetilde Y\setminus Y)=\varnothing.
                      \]
                      Then $\pi^{-1}(U)\subseteq Y$, and on $\pi^{-1}(U)$ we have $\pi=i$. Since $i$ is an immersion and $U\cap Z \subseteq i(Y)$, the restriction of $i$ to $\pi^{-1}(U)$ factors as a \textbf{closed immersion} into $U$.
                      Let us denote this closed immersion by $i_U : \pi^{-1}(U) \hookrightarrow U$.

                      In Step 4 we saw that the restriction of $M^\bullet_{\widetilde Y}$ to $Y$ (which completely contains $\pi^{-1}(U)$) is concentrated in degree $0$ and isomorphic to $N$.
                      Therefore, on the open set $U$, the derived pushforward $K^\bullet = \pi_+(M^\bullet_{\widetilde Y})$ restricts to the pushforward of $M^\bullet_{\widetilde Y}|_{\pi^{-1}(U)}$ along the closed immersion $i_U$. Hence $K^\bullet$ is concentrated entirely in degree $0$ on $U$, and we obtain:
                      \[
                          K|_U = \mathcal H^0(K^\bullet)|_U \;\simeq\; (i_U)_+(N|_{\pi^{-1}(U)}) .
                      \]

                      Now $K$ is holonomic, hence of finite length \cite[Prop.~3.1.2]{hotta2008d}.
                      This guarantees the existence of a finite Jordan--H\"older composition series:
                      \[
                          0 = K_0 \subsetneq K_1 \subsetneq \dots \subsetneq K_r = K
                      \]
                      where each quotient $S_m := K_m / K_{m-1}$ is a simple holonomic $\mathcal D_X$-module.
                      We know that restriction to an open subset is an exact functor. Therefore, restricting this entire series to $U$ yields a filtration of the restricted module:
                      \[
                          0 = K_0|_U \subseteq K_1|_U \subseteq \dots \subseteq K_r|_U = K|_U \simeq (i_U)_+(N|_{\pi^{-1}(U)}).
                      \]
                      Recall that $N$ is a simple $\mathcal D_Y$-module. Thus, $N|_{\pi^{-1}(U)}$ is a simple $\mathcal D_{\pi^{-1}(U)}$-module.
                      Since $(i_U)_+$, as a pushforward along a closed immersion, preserves simplicity (by Kashiwara's equivalence \ref{thm:kashiwara_equiv}), $(i_U)_+(N|_{\pi^{-1}(U)})$ is a simple $\mathcal D_U$-module.

                      In \textbf{any} filtration of a simple module (like $K|_U$), the successive quotients $(S_m)|_U \simeq K_m|_U / K_{m-1}|_U$ must be either $0$ or the simple module $K|_U$ itself.
                      Therefore, there exists a \textbf{unique} composition factor $S$ (one of the $S_m$) of $K$ such that its restriction $S|_U \simeq (i_U)_+(N|_{\pi^{-1}(U)})$. All other composition factors of $K$ are supported strictly outside of $U$.

                      By \cite[Thm.~3.4.2]{hotta2008d}, the minimal extension $L_X(Y,N)$ is the unique simple submodule of $i_+ N$.
                      Two simple modules $S$ and $L_X(Y,N)$ agree on an open subset $U\subseteq X $, so $S$ must be isomorphic to the minimal extension $L_X(Y,N)$, by \cite[Prop.~3.1.7]{hotta2008d}.
                      Finally, the subquotient $S\simeq L_X(Y,N)$ of $K$ is also KK-regular.

                  \end{proof}

        \end{enumerate}

    \end{proof}

    This is the end of Rmk.~\ref{rmk_definition_of_regular_holonomicity_jusitfied}.

\end{remark}

\begin{example}\label{ex:first-example-non-regular-d-module}
    $M=\mathbb{C}\left[ x, \frac{1}{x} \right]\cdot e^{1/x}$ is not regular because $M$ is simple and $M\simeq \mathcal{D}_{X} /\mathcal{D}_{X}(x^{2}\partial_{x}+1)$.
\end{example}

\begin{proof}
    This is a 1-dimensional case,
    so after analytification $M^{\mathrm{an}} = \D^{\mathrm{an}}_X / \D^{\mathrm{an}}_X (x^2\partial_x+1)$.

    We consider the stalk at $x=0$, so we can use Prop.~\ref{prop:classification-unipotent-indecomposable-regular-holonomic}.
    As a cyclic $\mathcal{D}_X$-module, $M^{\mathrm{an}}(0)$ is simple,
    this is essentially due to $\partial \in \D_1 / \D_1 (x^2\partial_x+1)$ have both a left and a right inverse.

    But $\partial \in \D_1 / \D_1 (x^2\partial_x+1)$ is not in the list of Prop.~\ref{prop:classification-unipotent-indecomposable-regular-holonomic}.
    So it is not regular holonomic. Or by compare two indices in \cite[Def.~5.1.10]{Bjork1993AnalyticDModules}.

    So the whole analytic $\D$-module $M^{\mathrm{an}}$ is not regular holonomic.

    By \cite[Prop.~6.1.12]{hotta2008d}, $M$, as an algebraic $\D_{\mathbb{A}_1}$-module, is not regular holonomic.
\end{proof}



\section{Kashiwara's equivalence}
\begin{remark}[Recall]
    The $\D$-module we talk about are all quasi-coherent, so $\mathrm{Mod}(\mathcal{D}_{X})$ and $\mathrm{Mod}^{X}(\mathcal{D}_{Y})$ exactly mean the categories of quasi-coherent $\D_X$-modules and quasi-coherent $\D_Y$-modules supported on $X$ respectively.
\end{remark}

\begin{theorem}[Kashiwara's equivalence]\label{thm:kashiwara_equiv}
    Let $Y$ be a smooth affine algebraic variety over $\mathbb{C}$  with coordinate $(x_{1},\dots,x_{n},y_{1},\dots,y_{\ell})$,
    and $X$ a subvariety defined by $(y_{1}=y_{2}=\cdots=y_{\ell}=0)$,
    so $X$ is smooth with coordinate $(x_{1},\dots,x_{n})$.
    Let $i: X \hookrightarrow Y$ be the closed embedding, $\mathrm{Mod}(\mathcal{D}_{X})$
    the category of $\mathcal{D}_X$-mod and $\mathrm{Mod}^{X}(\mathcal{D}_{Y})$
    the category of $\mathcal{D}_Y$-mod supported on $X$, then we have categorical equivalence:
    \begin{align*}
        \mathrm{Mod}(\mathcal{D}_{X})                        & \xrightarrow{\sim} \mathrm{Mod}^{X}(\mathcal{D}_{Y})               \\
        M                                                    & \longmapsto i_{+}M = M[\partial_{y_{1}},\dots,\partial_{y_{\ell}}] \\
        H^0(i^\dagger N) = H^{0}(K(N; y_{1}, \ldots, y_{l})) & \longmapsfrom N
    \end{align*}
\end{theorem}

\begin{remark}[Kashivara's equivalence of affine closed embeddings]\label{rem:kashiwara_explicit}
    \mbox{}

    Assume we have following two cases:

    \begin{enumerate}
        \item (General) $f: X\to Y$ morphism between smooth algebraic varieties. Recall $X$ and $Y$ are locally ringed spaces, so we have the morphism of sheaves of rings $f^{-1}\mathcal{O}_Y \to \mathcal{O}_X$ and $f^{-1}\D_Y \to \D_X$ respectively.
        \item (Special) $i: X\hookrightarrow Y$ a closed embedding between affine smooth varieties, defined by the vanishing of coordinates $(y_{r+1}, \dots, y_{n})$. Let $(x_1, \dots, x_r)$ be coordinates on $X$, so $Y$ has coordinate $(x_1, \dots, x_r, y_{r+1}, \dots, y_{n})$.
    \end{enumerate}

    Then we define following terms:

    \begin{enumerate}
        % ------------------------------------------------
        % 1. Transfer Bimodule
        % ------------------------------------------------
        \item The $(\D_X, f^{-1}\D_Y)$-bimodule $\mathcal{O}_X \otimes_{f^{-1}\mathcal{O}_Y} f^{-1}\D_Y$ is denoted by $\D_{X \to Y}$. And the $(f^{-1}\D_Y, \D_X)$-bimodule $\D_{Y \leftarrow X}$ by $\D_{Y \leftarrow X} := \Omega_X \otimes_{\mathcal{O}_X} D_{X \rightarrow Y} \otimes_{f^{-1}\mathcal{O}_Y} f^{-1}\Omega_Y^{\otimes -1}$. These two are \textbf{The Transfer Bimodules} \cite[Def~1.3.1., Def~1.3.3.]{hotta2008d}.
        \item For the special case, $\D_{X \to Y} \simeq \D_X \otimes_{\mathbb{C}} \mathbb{C}[\partial_{y_{r+1}}, \dots, \partial_{y_n}]$ as a left $\D_X$-module. In particular, $\D_{X \to Y}$ is a locally free $\D_X$-module of infinite rank (unless $r = n$) \cite[Ex.~1.3.2.]{hotta2008d}.
        \item \cite[Ex.~1.3.5.]{hotta2008d} For the special case, Set $\D' = \bigoplus_{m_1,\dots,m_r} \partial_{y_1}^{m_1} \dots \partial_{y_r}^{m_r} \mathcal{O}_Y \subset \D_Y$. It is a subring of $\D_Y$ and we have $\D_Y \simeq \mathbb{C}[\partial_{y_{r+1}}, \dots, \partial_{y_n}] \otimes_{\mathbb{C}} \D'$ as a right $\D'$-module. Hence we have
              $$\D_{Y \leftarrow X} \simeq \mathbb{C}[\partial_{y_{r+1}}, \dots, \partial_{y_n}] \otimes_{\mathbb{C}} (i^{-1}\D' \otimes_{i^{-1}\mathcal{O}_Y} \mathcal{O}_X). $$
              \begin{enumerate}
                  \item $i^{-1}\D' \otimes_{i^{-1}\mathcal{O}_Y} \mathcal{O}_X = \D_X$ by identifying $f = f(x,y) \in \mathcal{O}_Y$ with $f|_X := f(x,0) \in \mathcal{O}_X$ (i.e. $f^{-1}\mathcal{O}_Y\to \mathcal{O}_X$ mentioned above), so
                        $$\D_{Y \leftarrow X} \simeq \mathbb{C}[\partial_{y_{r+1}}, \dots, \partial_{y_n}] \otimes_{\mathbb{C}} \D_X.$$
                  \item So the left $i^{-1}\D_Y$-module structure can be described by the decomposition $\D_Y \simeq \mathbb{C}[\partial_{y_{r+1}}, \dots, \partial_{y_n}] \otimes_{\mathbb{C}} \D'$.
              \end{enumerate}

        \item We can define a \textbf{inverse image} functor
              $$
                  Lf^* : D^b(\D_Y) \to D^b(\D_X) \quad (M^* \longmapsto \D_{X \to Y} \otimes_{f^{-1}\D_Y}^L f^{-1}M^*)
              $$
              by using a flat resolution of $M^*$.
        \item We will also use the \textbf{shifted inverse image} functor
              $$
                  f^\dagger := Lf^*[\dim X - \dim Y] : D^b(\D_Y) \to D^b(\D_X).
              $$
        \item We can define a \textbf{direct image} ($f_+$ or $\int_f$) functor
              $$
                  \int_f M^* := Rf_*(\D_{Y \leftarrow X} \otimes_{\D_X}^L M^*) \quad (M^* \in D^b(\D_X)),\quad \int_f^k M^* := H^k \left( \int_f M^* \right).
              $$
        \item \cite[Ex.~1.5.23.]{hotta2008d} For the special case, we have:
              $$
                  \int_i^k M = 0 \; (k \ne 0), \qquad \int_i^0 M \simeq \mathbb{C}[\partial_{y_{r+1}}, \dots, \partial_{y_n}] \otimes_{\mathbb{C}} i_* M.
              $$
              The action of $\partial_{y_k}$ for $k > r$ is given by the multiplication on the first factor $\mathbb{C}[\partial_{y_{r+1}}, \dots, \partial_{y_n}]$.
              The action of $\varphi \in \mathcal{O}_Y$ and $\partial_{y_k}$ for $k \le r$ is given by:
              \begin{align*}
                  \varphi(1 \otimes m)        & = 1 \otimes (\varphi|_X)m \qquad (\varphi \in \mathcal{O}_Y), \\
                  \partial_{y_k}(1 \otimes m) & = 1 \otimes \partial_{x_k}m \qquad (1 \le k \le r).
              \end{align*}

        \item For the spectcial case, in order to calculate $i^\dagger N$:
              \begin{enumerate}
                  \item \cite[Equation~1.5.1.]{hotta2008d} The Koszul complex is a free (hence flat) $i^{-1}\mathcal{O}_Y$-module resolution of $\mathcal{O}_X$.
                        But the Koszul complex is a different definition \cite[\href{https://stacks.math.columbia.edu/tag/0623}{Tag 0623}]{stacks-project}, which can also be constructed as mapping cone \cite[\href{https://stacks.math.columbia.edu/tag/0629}{Tag 0629}]{stacks-project}:
                        \begin{align*}
                            \left[\, 0 \to K^\bullet(i^{-1}\mathcal{O}_Y; y_{r+1}, \dots, y_{n}) \to \mathcal{O}_X \to 0\, \right] := \left[\, 0 \to K_{n-r} \to \dots \to K_1 \to K_0 \to \mathcal{O}_X \to 0\, \right]
                        \end{align*}
                        where the terms are defined as:
                        $$
                            K_j = \bigwedge^j \left( \bigoplus_{k=r+1}^n i^{-1}\mathcal{O}_Y\, \mathrm{d}y_k \right).
                        $$
                        The differential $K_j \to K_{j-1}$ is given by:
                        $$
                            f dy_{k_1} \wedge \dots \wedge dy_{k_j} \longmapsto \sum_{p=1}^j (-1)^{p+1} y_{k_p} f dy_{k_1} \wedge \dots \wedge \widehat{dy_{k_p}} \wedge \dots \wedge dy_{k_j}.
                        $$
                  \item \cite[Prop.~1.5.8.]{hotta2008d} We have isomorphic complexes $\D_{X \to Y} \otimes_{f^{-1}\D_Y}^L f^{-1}M^* \cong \mathcal{O}_X \otimes_{f^{-1}\mathcal{O}_Y}^L f^{-1}M^*$, so we use resolution of $\mathcal{O}_X$ to calculate $i^\dagger N$. Which is just:
                        $$i^\dagger N = K^\bullet(i^{-1}\mathcal{O}_Y; y_{r+1}, \dots, y_{n}) \otimes_{i^{-1}\mathcal{O}_Y} i^{-1}N [r-n].$$
              \end{enumerate}
    \end{enumerate}

\end{remark}

\begin{proof}
    By induction on co-dimension, it is enough to assume
    \[
        \begin{aligned}
            i: X                & \longrightarrow Y                  \\
            (x_{1},\dots,x_{n}) & \longmapsto (x_{1},\cdots,x_{n},0)
        \end{aligned}
    \]
    i.e. $X$ is defined by $\langle y = 0 \rangle$ in $Y$.

    Here we have $i_{+}: M\mapsto M[\partial_{y}]$, and we want to define a quasi-inverse of $i_{+}$. To define $i_{+}^{\#}$, we shall study how $\partial_{y}$ acts on $N$.

    For $N \in \mathrm{Mod}^{X}(\mathcal{D}_{Y})$, consider the eigenvectors $N^{j} = \{ s \in N \mid y\partial_{y} \cdot s = j s \}$. We want to show that $N=\bigoplus\limits_{i=1}^{\infty}N^{-i}$.

    Since we have the relation $[\partial_{y}, y] = \partial_{y}y -  y\partial_{y} = 1$:
    \begin{itemize}
        \item We can check that $y N^{j} \subseteq N^{j+1}$ and $\partial_{y} N^{j} \subseteq N^{j-1}$.
        \item We have $y\partial_{y}: N^{j}\rightarrow N^{j}$ is an isomorphism for all $j \neq 0$.
        \item $\partial_{y}y = y\partial_{y} + 1: N^{j}\rightarrow N^{j}$, $s\mapsto (j+1)s$ is an isomorphism for all $j \neq -1$.
        \item So for $j\neq -1$ and $0$, especially for $j<-1$, we have $\partial_{y}:N^{j+1}\rightarrow N^{j}$ and $y:N^{j}\rightarrow N^{j+1}$ are both isomorphisms.
    \end{itemize}

    Now we show $N=\bigoplus\limits_{i=1}^{\infty}N^{-i}$.
    Since $N$ is supported on $X$, by Nullstellensatz, $\forall s\in N$, $y^{k}\cdot s = 0$ for some $k\gg0$, i.e., $N = \bigcup\limits_{k\geq1} \ker(N\xrightarrow{y^{k}} N)$.

    Since $N^{-i}$ are eigenspaces, $\bigoplus\limits_{i=1}^{\infty }N^{-i}\subseteq N$, it suffices to prove that $N\subseteq \bigoplus\limits_{i=1}^{\infty }N^{-i}$. To show this, it's enough to prove that for all $k\ge 1$, we have $\ker(N\xrightarrow{y^{k}} N)\subseteq\bigoplus\limits_{i=1}^{k}N^{-i}$, and we do induction on $k$.

    \begin{description}
        \item[Base case ($k=1$):] Let $s\in \mathrm{ker}(N\xrightarrow{y}N)$, so we have $ys=0$. Then
            \[
                \partial_{y}ys=(y\partial_{y}+1)s=0 \implies y\partial_{y}=-1\cdot s \implies s\in N^{-1}.
            \]

        \item[Inductive step ($k>1$):] Now take $s\in \ker(N \xrightarrow{y^{k}}N)$, so we have $y^{k}s=y^{k-1} \cdot(ys)=0$. Then $ys \in \ker(N \xrightarrow{y^{k-1}}N)$, which suggests $ys \in \bigoplus\limits_{i=1}^{k-1} N^{-i}$ by inductive hypothesis.
            Since we have $\partial_{y}: N^{j+1}\rightarrow N^{j}$, we obtain $\partial_{y}ys \in \bigoplus\limits_{i=2}^{k} N^{-i}$, and $\partial_{y}ys=y\partial_{y}s+s$.

            On the other hand, we obtain
            \[
                \partial_{y}y^{k}s=y^{k}\partial_{y}s+ky^{k-1}s=y^{k-1}(y\partial_{y}s+ks)=0,
            \]
            so $y\partial_{y}s+ks\in \bigoplus\limits_{i=1}^{k-1} N^{-i}$ by inductive hypothesis.

            Thus, $(k-1)s=(y\partial_{y}s+ks)-(y\partial_{y}s+s)\in \bigoplus_{j=1}^{k} N^{-i}$ implies that $s\in \bigoplus_{j=1}^{k} N^{-j}$.
    \end{description}

    With above, we now obtain $N=\bigoplus\limits_{i=1}^{\infty}N^{-i}$ with $N^{-1}=\mathrm{ker}(N\xrightarrow{y}N)$, so we have $N\simeq N^{-1}[\partial_{y}]$. Then we define $i^{\#}N = \ker(N \xrightarrow{y} N)$ (the standard notation is $i^{\natural}$ \cite[Thm.~1.6.1.]{hotta2008d}), $i^{\#}$ is the quasi-inverse of $i_{+}$.

    In general, $i^{\#}N = H^{0}(K(N; y_{1}, \ldots, y_{k}))$.
\end{proof}

\begin{corollary}\label{cor:kashiwara_equiv_conormal}
    For $N \in \text{Mod}^{X}(\mathcal{D}_{Y})$ finitely-generated, we have $\text{Ch}(N) = T_{X}^{*}Y$ if and only if $i^{\#}N \in \text{Mod}(\D_X)$ is an integrable connection.
\end{corollary}
\begin{proof}
    We prove a lemma first.

    \begin{lemma}[Characteristic variety of a closed embedding pushforward]\label{prop:charvar-closed-emb}
        Let $i:X\hookrightarrow Y$ be a closed embedding of smooth $k$-varieties, such that its image is also smooth (char$(k)=0$).
        Write
        \[
            T^*Y|_X:=X\times_Y T^*Y,
        \]
        and denote by $\widetilde i:T^*Y|_X\to T^*Y$ the second projection (the natural inclusion of the
        restriction, or \textbf{scheme-theoretic inverse image}
        \cite[\href{https://stacks.math.columbia.edu/tag/01JV}{Tag 01JV}, \href{https://stacks.math.columbia.edu/tag/01HO}{Tag 01HO}]{stacks-project}),
        and by $\rho_i:T^*Y|_X\to T^*X$ the map induced by $(\mathrm{d}i)^*:i^*\Omega_Y^1\to \Omega_X^1$
        \cite[\href{https://stacks.math.columbia.edu/tag/01LQ}{Tag 01LQ}, \href{https://stacks.math.columbia.edu/tag/01M1}{Tag 01M1}]{stacks-project}.
        Let $M$ be a coherent left $\mathcal{D}_X$-module, and set $N:=i_+M$ (the direct image for a closed embedding).
        Then
        \[
            \mathrm{Ch}(N)=\widetilde i\bigl(\rho_i^{-1}(\mathrm{Ch}(M))\bigr)\subset T^*Y.
        \]
    \end{lemma}

    \begin{proof}
        The statement is local on $Y$, so we work on an affine open subset with adapted coordinates.
        Assume $Y=\mathrm{Spec}(A)$ with \textbf{étale coordinates} (that is what exactly algebraic coordinate meaning)
        \[
            (x_1,\dots,x_m,y_1,\dots,y_n),\qquad X=\{y_1=\cdots=y_n=0\},
        \]
        The existence of such coordinates system is guaranteed by the image of $i$ is smooth, then we use Prop.~\ref{prop:smooth-subvariety}.

        And let $I=(y_1,\dots,y_n)\subset A$, so $X=\mathrm{Spec}(B)$ with $B=A/I$.
        Equip $\mathcal{D}_Y$ and $\mathcal{D}_X$ with the order filtration $F_\bullet$. Then
        \[
            \mathrm{gr}\,\mathcal{D}_Y \simeq A[\xi_{x_1},\dots,\xi_{x_m},\xi_{y_1},\dots,\xi_{y_n}],
            \qquad
            \mathrm{gr}\,\mathcal{D}_X \simeq B[\xi_{x_1},\dots,\xi_{x_m}].
        \]
        Choose a good filtration $F_\bullet M$; then $\mathrm{gr}\,M$ is a finitely generated graded module over
        \[
            R:=B[\xi_{x_1},\dots,\xi_{x_m}].
        \]
        Set
        \[
            S:=B[\xi_{x_1},\dots,\xi_{x_m},\xi_{y_1},\dots,\xi_{y_n}]=R[\xi_y].
        \]

        For a closed embedding, one constructs an induced filtration on $N=i_+M$ such that the associated graded
        module satisfies the explicit formula
        \[
            \mathrm{gr}\,N \;\cong\; \mathrm{gr}\,M \otimes_R S,
            \qquad\text{and } I \text{ acts trivially on }\mathrm{gr}\,N.
        \]

        To see this, We define a filtration on $N \cong M [\partial_{y_1}, \dots, \partial_{y_\ell}]$ by:
        \[
            F_k N = \sum_{p + |q| \le k} F_p M \otimes \partial_y^q,
        \]
        where $q = (q_1, \dots, q_\ell)$ is a multi-index and $\partial_y^q = \partial_{y_1}^{q_1} \dots \partial_{y_\ell}^{q_\ell}$.
        This is a good filtration because if $u_1, \dots, u_m$ generate $\mathcal{R}^{F}_{\bullet}(M)$ over $\mathcal{R}^{\Gamma}_{\bullet}(\mathcal{D}_X)$, they also generate $\mathcal{R}^{F}_{\bullet}(N)$ over $\mathcal{R}^{\Gamma}_{\bullet}(\mathcal{D}_Y)$.
        Then the associated graded module is:
        \begin{align*}
            \mathrm{gr}(N) & \cong \mathrm{gr}(M) \otimes_{\mathbb{C}} \mathbb{C}[\xi_{y_1}, \dots, \xi_{y_n}]                                                                                                                                      \\
                           & \cong \mathrm{gr}(M) \otimes_{B[\xi_{x_1}, \dots, \xi_{x_m}]} B[\xi_{x_1}, \dots, \xi_{x_m}, \xi_{y_1}, \dots, \xi_{y_n}] &  & \text{($\gr M$ is also a $\gr \mathcal{D}_X = B[\xi_{x_1}, \dots, \xi_{x_m}]$-module)},
        \end{align*}
        where $\xi_{y_j} := \sigma(\partial_{y_j})$ for the principal symbol map $\sigma$.
        To see this, we look at the degree $k$ part:
        \[
            \mathrm{gr}_k(N) = F_k N / F_{k-1} N \cong \bigoplus_{j=0}^k \left( F_{k-j} M / F_{k-j-1} M \right) \otimes \left( \text{Span}\{\partial_y^q : |q| = j\} \right).
        \]
        The first factor is $\mathrm{gr}_{k-j}(M)$, and the second factor corresponds to homogeneous polynomials in $\xi_y$ of degree $j$.
        Summing over all $k$, we get the tensor product decomposition of $\mathrm{gr}(N)$.

        The action of $\mathcal{O}_Y$ on $N$ is determined by the module structure of $i_+M$:
        \begin{align*}
            y_j \cdot (m \otimes \partial_y^q)
             & = m \otimes (y_j \partial_y^q)                                       &  & \text{(Definition of $\mathcal{D}_Y$-action \ref{rem:kashiwara_explicit})} \\
             & = m \otimes \left( \partial_y^q y_j + [y_j, \partial_y^q] \right)                                                                                    \\
             & = m \otimes (\partial_y^q y_j) + m \otimes [y_j, \partial_y^q]                                                                                       \\
             & = (y_j \cdot m) \otimes \partial_y^q + m \otimes [y_j, \partial_y^q]                                                                                 \\
             & = 0 \otimes \partial_y^q + m \otimes [y_j, \partial_y^q]             &  & \text{(Since $y_j = 0$ in $\mathcal{O}_X$)}                                \\
             & = m \otimes \left( -q_j \partial_y^{q-e_j} \right)                                                                                                   \\
             & = -q_j (m \otimes \partial_y^{q-e_j})                                &  & \text{(Extracting scalar coefficient)}
        \end{align*}
        Notice that the commutator of $y_j$ lowers the order of differential operators, so in $\mathrm{gr}(N)$, the symbol $y_j$ acts as $0$. Thus the support $\mathrm{Supp}(\mathrm{gr}(N))$ is contained in the locus $y_1 = \dots = y_n = 0$.

        Let
        \[
            J:=\mathrm{Ann}_R(\mathrm{gr}\,M)\subset R.
        \]
        Since $S=R[\xi_t]$ is a free (hence flat) $R$-algebra,
        \cite[\href{https://stacks.math.columbia.edu/tag/0250}{Tag 0250}]{stacks-project},
        and \cite[\href{https://stacks.math.columbia.edu/tag/07T8}{Tag 07T8}]{stacks-project} gives
        \[
            \mathrm{Ann}_S(\mathrm{gr}\,N)
            =\mathrm{Ann}_S(\mathrm{gr}\,M\otimes_RS)
            =\mathrm{Ann}_R(\mathrm{gr}\,M)\cdot S
            =JS.
        \]
        Because $\mathrm{gr}\,N$ is a finite $S$-module, its support is the closed subset cut out by its annihilator:
        \[
            \mathrm{Supp}(\mathrm{gr}\,N)=V(\mathrm{Ann}_S(\mathrm{gr}\,N))=V(JS)\subset \mathrm{Spec}(S),
        \]
        by
        \cite[\href{https://stacks.math.columbia.edu/tag/00L2}{Tag 00L2}]{stacks-project}.

        Now note that
        \[
            T^*X=\mathrm{Spec}(R),\qquad T^*Y|_X=\mathrm{Spec}(S),
        \]
        and that the morphism $\rho_i:T^*Y|_X\to T^*X$ corresponds to the inclusion $R\hookrightarrow S$
        (adjoining the $\xi_t$ variables), while $\widetilde i:T^*Y|_X\hookrightarrow T^*Y$ corresponds to the quotient map
        $A[\xi_x,\xi_y]\twoheadrightarrow (A/I)[\xi_x,\xi_y]=S$.

        Since $I$ acts trivially on $\gr N$, $\Supp(\gr N)\subset T^*Y$ is contained in $T^*Y|_X$ (viewed as a closed subscheme of $T^*Y$ \cite[\href{https://stacks.math.columbia.edu/tag/02RA}{Tag 02RA}]{stacks-project}), so
        \[
            \mathrm{Ch}(N)
            =\widetilde i\bigl(\mathrm{Supp}(\mathrm{gr}\,N)\bigr)
            \subset \widetilde i(T^*Y|_X).
        \]

        For summary,
        \[
            \mathrm{Ch}(N)
            =\widetilde i\bigl(\mathrm{Supp}(\mathrm{gr}\,N)\bigr)
            =\widetilde i\bigl(V(JS)\bigr)
            =\widetilde i\bigl(\rho_i^{-1}(V(J))\bigr)
            =\widetilde i\bigl(\rho_i^{-1}(\mathrm{Ch}(M))\bigr).
        \]

        For one wondering why $V(JS) = \rho_i^{-1}(V(J))$, see \cite[\href{https://stacks.math.columbia.edu/tag/05CK}{Tag 05CK}]{stacks-project} and \cite[Example~1.13]{cring-ch15-flatness}.

    \end{proof}

    It suffices to show $\mathrm{Ch}(M)=T_X^*X$ if and only if $\mathrm{Ch}(N)=T_X^*Y$.

    If $\mathrm{Ch}(N)=T_X^*Y$, the closed subset determined by $(y_1=\dots=y_n=\xi_{x_1}=\dots=\xi_{x_m}=0)$.

    Then we apply Proposition~\ref{prop:charvar-closed-emb} and keep notation in it. First we calculate $\widetilde i^{-1}(\mathrm{Ch}(N))$,
    which is in $T^*_X Y = (y_1=\dots=y_n=0)$,
    and $\Supp \gr N = (\xi_{x_1}=\dots=\xi_{x_d}=0)$.

    To calculate $\rho_i(\Supp \gr N)$, we need to pull the ideal $\langle \xi_{x_1},...,\xi_{x_m} \rangle \subseteq (A/I)[\xi_x,\xi_y]$ back to $(A/I)[\xi_x]$,
    by  \cite[\href{https://stacks.math.columbia.edu/tag/05CK}{Tag 05CK}]{stacks-project} and \cite[Example~1.13]{cring-ch15-flatness},
    it is also $\langle \xi_{x_1},...,\xi_{x_m} \rangle \subseteq (A/I)[\xi_x]$ which is just $T^*_X X$.

\end{proof}
