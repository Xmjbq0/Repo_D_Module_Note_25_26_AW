\section{Bj\"ork's approach to existence of b-functions}

\begin{theorem}[Bj\"ork]\label{Theorem 8.4}
    Let $M$ be a holonomic $\mathcal{D}_X$-module with a good filtration $\Omega_{\bullet} M$ over $V_{\bullet} \mathcal{D}_X$. There exists a non zero $b(s) \in \mathbb{C}[s]$ such that
    \[
        b(t\partial_t + k) \text{ kills } \frac{\Omega_k M}{\Omega_{k-1} M}, \quad \forall k \in \mathbb{Z}.
    \]
    The monic minimal degree $b(s)$ is called the $\textbf{$b$-function}$ of $M$ along $Y$.
\end{theorem}
\begin{remark}
    In the handwritten notes provided by the lecturer, says $b(t\partial_t \textcolor{red}{ - } k) \text{ kills } \frac{\Omega_k M}{\Omega_{k-1} M}$,
    but this is not compatible with the increasing filtration setting in Rmk.~\ref{rmk:increasing_V_filtration}.
\end{remark}


\begin{lemma}\label{lemma 8.5}
    Let $Y$ be a smooth variety over $\mathbb{C}$, $N$ a holonomic $\mathcal{D}_Y$-module, and $\theta\in \mathrm{End}_{\mathcal{D}_Y}(N)$. Then there exists $b(s)\in\mathbb{C}[s]$ such that $b(\theta)=0$.
\end{lemma}
\begin{proof}
    Since $N$ is holonomic, we obtain that
    \[
        \mathrm{Ch}(N) = \bigcup_{i} T^*_{Y_i} Y,
    \]
    with $Y_{i}$ locally closed.

    Take a $Y_i$ of maximal dimension and take an (affine) open subset $U \subset Y$
    such that $Y_{i}^{\circ} \hookrightarrow U$ is a closed embedding
    with $Y_{i}^{\circ}$ smooth, $\overline{Y_i^{\circ}} = Y_{i}$, $U \cap Y_i = Y_i^\circ$
    and $U \cap (\bigcup_{j\neq i}Y_j) = \varnothing$. Then
    \[
        \mathrm{Ch}(N|_U) = T^*_{Y_i^\circ} U.
    \]

    Since $Y^\circ$ is smooth, locally we can split algebraic coordinates to use Kashiwara's equivalence \ref{thm:kashiwara_equiv}, by shrinking $U$:
    \[
        N|_U = i_{+}M = M[\partial_1, \ldots, \partial_k].
    \]

    Consider $U^{\mathrm{an}} \to U$, the analytification functor, since $Y^\circ \hookrightarrow U$ is a closed embedding,
    by \cite[Proposition~4.7.2.(ii)]{hotta2008d} we have $i_{+}(M)^{\mathrm{an}} \to i^{\mathrm{an}}_{+}(M^{\mathrm{an}})$ is an isomorphism.

    We use Riemann-Hilbert correspondence on $(N|_U)^{\mathrm{an}}$:
    \begin{align*}
        (N|_U)^{\mathrm{an}} & \cong (i_{+}M)^{\mathrm{an}}                                                                                                       &  & \text{(By \ref{thm:kashiwara_equiv})}                                                                                   \\
                             & \cong i^{\mathrm{an}}_{+}(M^{\mathrm{an}})                                                                                                                                                                                                                      \\
                             & \cong M^{\mathrm{an}}[\partial_1, \ldots, \partial_k]                                                                              &  & \text{\begin{tabular}[t]{@{}l@{}}($i_+$ of a closed embedding behaves\\ the same on variety and manifold)\end{tabular}} \\
                             & \cong \mathcal{O}^{\mathrm{an}}_{Y_i^\circ} \otimes_{\mathbb{C}} L \otimes_{\mathbb{C}} \mathbb{C}[\partial_1, \ldots, \partial_k] &  & \text{(By \ref{cor:kashiwara_equiv_conormal}, $M$ is an integrable connection, use \ref{thm:basic-RH-corr})}
    \end{align*}
    Then we get
    \[
        (N|_U)^{\mathrm{an}} = \mathcal{O}^{\mathrm{an}}_{Y_i^\circ} \otimes L[\partial_1, \ldots, \partial_k]
    \]
    with $L$ a local system on $Y_i^\circ$.

    For the name \emph{Riemann-Hilbert correspondence}, one may think it is a \textbf{categorical equivalence}, that means there is not only a correspondence of objects but also a correspondence of \textbf{morphisms}.

    That is \cite[Theorem~4.2.4.]{hotta2008d}'s $H^{-\mathrm{d}_X}(\mathrm{DR}(-))$ ($X$ a complex manifold),
    which means for a morphism between two holomorphic $\mathcal{D}_X$-modules $f: M_1 \to M_2$ locally free over $\mathcal{O}_X$ (holomorphic),
    with respective flat sections
    \[
        L_1 := \Ker(\nabla_{M_1}) \quad \text{and}\quad L_2 := \Ker(\nabla_{M_2}) \quad \text{(see \ref{rmk:basic-RH-corr})},
    \]
    we have a morphism between their corresponding local systems
    \[
        H^{-\mathrm{d}_X}(\mathrm{DR}(f)): L_1 \to L_2
    \]
    which is just a restriction of $f$.

    Recall \ref{prop:classification-of-local-system} says if $X$ is connected, we have an another categorical equivalence
    to identify local system and finite diemsional representation of $\pi_1(X,x)$ (take an arbitrary base point $x$), morphisms of representations
    is called \textbf{intertwining operator} with respect to the action of $\pi_1(X,x)$.

    They compose a third categorical equivalence, since categorical equivalence is fully faithful, we have
    \begin{align*}
        \mathrm{End}_{\mathcal{D}_U}(N|_U) & = \mathrm{End}_{\mathcal{D}_{Y^\circ}}(M)                                     &  & \text{(Kashiwara equivalence is also a categorical equivalence)}          \\
                                           & \subseteq \mathrm{End}_{\mathcal{D}^{\mathrm{an}}_{Y^\circ}}(M^{\mathrm{an}})                                                                                \\
                                           & = \mathrm{End}_{\mathrm{LocSys}_{X}}(L)                                       &  & \text{(By \cite[Theorem~4.2.4.]{hotta2008d})}                             \\
                                           & = \mathrm{End}_{\pi_1(X,x)}(L_x)                                              &  & \text{(By \ref{prop:classification-of-local-system}, and $L_x$ is fiber)} \\
                                           & \subseteq \mathrm{End}_{\C}(L_x)                                              &  & \text{(Definition of intertwining operator)}
    \end{align*}
    Therefore, by Cayley-Hamilton theorem, there exists $b_i(s) \in \mathbb{C}[s]$ such that $b_i(\theta|_{U}) = 0$.

    Before we do induction, recall a fact that for a smooth variety $Y$, a irreducible closed subset $X$, we have
    \[
        T^*_X Y = \overline{T^*_{X\cap U} U} \subseteq T^*Y \quad \text{(for affine open $U \subset Y$ such that $X\cap U \subseteq X^{\mathrm{sm}}$ dense in $X$)}
    \]
    is also irreducible, by the fact that
    \[
        T^*_{X \cap U} U \subseteq T^*U.
    \]
    is closed irreducible in $T^*U$.

    Now, take  $N_1 = b_1(\theta)N$.
    Since $b_1(\theta|_U) = 0$, we have $N_1|_U = 0$, which implies $\mathrm{Ch}(N_1) \cap T^*U = \varnothing$.

    By Bernstein's inequality \ref{thm:Bernstein-inequality},
    $N_1$, as a submodule of a holomonic $\D_Y$-module is also a holomonic $\D_Y$-module.
    So irreducible component of $\mathrm{Ch}(N_1)$ has dimension $n = \dim Y$,
    and is contained in some irreducible component of $\mathrm{Ch}(N) = \bigcup_k T^*_{Y_k} Y$.

    Since all irreducible component of a holomonic $\D_Y$-module has dimension $n = \dim Y$ (see Cor.~\ref{cor:holonomic-char-cycle}),
    and we have a fact that a proper closed subset of an irreducible variety has strictly smaller dimension,
    it follows that each irreducible component of $\mathrm{Ch}(N_1)$ must be equal to some $T^*_{Y_k} Y$.

    But it can't be $T^*_{Y_i} Y$, because $\mathrm{Ch}(N_1) \cap T^*U = \varnothing$.
    So we strictly reduce the number of components in the characteristic variety,
    so we are done by induction.


\end{proof}




\begin{theorem}\label{Theorem 8.6}
    Let $M$ be a finitely generated $\mathcal{D}_{X}$-module, and $\Omega_{\bullet}M$ a good filtration over $V_{\bullet}\mathcal{D}_{X}$. Then $j(\mathrm{gr}_{\bullet}^\Omega M)\ge j(M)$.
\end{theorem}
\begin{remark}
    In Thm.~\ref{theorem 7.12} we also proved $j(\gr_\bullet M)\ge j(M)$, but this time we have a different filtration, the $V$-filtration.

    We need the fact that $\gr_\bullet^{V}\D_X$ is Auslander regular, see Cor.~\ref{cor:Rees-ring-of-V-filtration-D-module}. So it can't replace the first proof in Thm.~\ref{theorem 7.12}.
\end{remark}
\begin{proof}
    If $\mathrm{gr}^\Omega_{\bullet}M=0$, there is nothing to prove, so we assume $\mathrm{gr}^\Omega_{\bullet}M\neq 0$ and $\ell=j(\mathrm{gr}_{\bullet}^{\Omega}M)<j(M)$.

    Then the $E_1$-page of the spectral sequence of $\check{F}^\bullet$
    introduced in Section~\ref{sect:more_on_filtered_rings},
    that converges to $\mathrm{Ext}_{\mathcal{D}_{X}}^{j}(M,\mathcal{D}_{X})$,
    satisfies:
    \[
        0 \rightarrow \operatorname{Ext}^{\ell}(\operatorname{gr}_{\bullet}^{\Omega} M, \operatorname{gr}^{V}_{\bullet}\mathcal{D}_X)
        \xrightarrow{\mathrm{d}_1}
        \operatorname{Ext}^{\ell+1}(\operatorname{gr}_{\bullet}^{\Omega} M, \operatorname{gr}^{V}_{\bullet}\mathcal{D}_X).
    \]
    Where we identify $\gr_\bullet \Hom_{\D_X}(F^\bullet, \D_X)$ with $\Hom_{\gr_\bullet^{V} \D_X}(\gr_\bullet F^\bullet, \gr_\bullet^{V} \D_X)$
    as in the proof of Cor.~\ref{corollary 7.11}.

    Since $\ell = j(\gr_\bullet^\Omega M)$, so we obtain that $E^{\ell}_{\bullet}(2)=\ker \mathrm{d}_1\subseteq\operatorname{Ext}^{\ell}(\operatorname{gr}_{\bullet}^{\Omega} M, \operatorname{gr}^{V}_{\bullet}\mathcal{D}_X)$ since no images map to $E^{\ell}_{\bullet}(1)$.

    Then we have the following exact sequence with the quotient $N=E^{\ell}_{\bullet}(1) /\ker \mathrm{d}_{1}\simeq \mathrm{d}_{1}(E_{\bullet}^{\ell}(1))\subseteq E^{\ell+1}_{\bullet}(1)$:
    \[
        0 \rightarrow E_{\bullet}^{\ell}(2) \rightarrow \operatorname{Ext}^{\ell}(\operatorname{gr}_{\bullet}^{\Omega} M, \operatorname{gr}^{V}_{\bullet}\mathcal{D}_X)
        \rightarrow N \rightarrow 0.
    \]
    In general, by induction, $\forall k\ge 1$, we also have such exact sequence:
    \[
        0 \rightarrow E_{\bullet}^{\ell}(k+1) \rightarrow E_{\bullet}^{\ell}(k) \rightarrow N_k \rightarrow 0.
    \]

    Since $j(M)>\ell$, and $\check{F}^\bullet$ is convergent (see Definition~\ref{def:convergent_spectral_sequence}),
    we have $E_{\bullet}^{\ell}(r) = \operatorname{gr}_{\bullet} \operatorname{Ext}^{\ell}(M, \mathcal{D}_X) = 0$ for some $r\gg 0$,
    and $E^{\ell}_{\bullet}(r)=\Ker \mathrm{d}_{r}$.

    By triangle inequality of grade number (one can get this inequality by a long exact sequence corresponds to the short exact sequence), we have
    \[
        j(E_{\bullet}^{\ell}(k)) \ge \min(j(E_{\bullet}^{\ell}(k+1)), j(N_k)).
    \]
    Note that $N_k$ is a subquotient of $E_{\bullet}^{\ell+1}(1) = \operatorname{Ext}^{\ell+1}(\operatorname{gr}_{\bullet}^{\Omega} M, \operatorname{gr}^{V}_{\bullet}\mathcal{D}_X)$.
    Since $\gr_\bullet^{V}\D_X$ is Auslander regular, we have $j(N_k) \ge \ell+1$.

    Since $E_{\bullet}^{\ell}(r) = 0$ for $r \gg 0$, we have $j(E_{\bullet}^{\ell}(r)) = \infty$.
    By induction (downwards on $k$), we obtain
    \[
        j(E_{\bullet}^{\ell}(1)) \ge \ell+1.
    \]
    However, $E_{\bullet}^{\ell}(1) = \operatorname{Ext}^{\ell}(\operatorname{gr}_{\bullet}^{\Omega} M, \operatorname{gr}^{V}_{\bullet}\mathcal{D}_X)$.
    Since $\ell = j(\operatorname{gr}_{\bullet}^{\Omega} M)$, we have $j(E_{\bullet}^{\ell}(1)) = \ell$ by Theorem~\ref{theorem 7.2}'s (2) and (4).
    This contradicts $j(E_{\bullet}^{\ell}(1)) \ge \ell+1$.
\end{proof}

\begin{proof}[proof of (\ref{Theorem 8.4})]
    \quad

    By Theorem~\ref{Theorem 8.6},
    $j(\mathrm{gr}_{\bullet}^\Omega M)\ge j(M)=n$.
    If $\mathrm{gr}_{\bullet}^\Omega M=0$, take $b(s)=1$.

    Since by Cor.~\ref{cor:Rees-ring-of-V-filtration-D-module} and filtration is good, we view $\mathrm{gr}_{\bullet}^\Omega M$ as a finitely generated$\gr_\bullet^{V}\D_X = \D_{T_Y X}$-module.
    And in the proof of Cor.~\ref{cor:Rees-ring-of-V-filtration-D-module}, we verified that $T_{Y}X$ is a smooth variety of dimension $n$ over $\mathbb{C}$.
    So if $\gr_\bullet^{\Omega}M \neq 0$, then $\mathrm{dim}(\mathrm{Ch}(\mathrm{gr}^\Omega_{\bullet}M))\ge n$ by Thm.~\ref{thm:Bernstein-inequality}.

    Since $\mathrm{gr}_{\bullet}^V\mathcal{D}_{X}$ is a Auslander regular ring,
    by Thm.~\ref{theorem 7.12},
    we obtain that $j(\mathrm{gr}_{\bullet}^\Omega M)+\mathrm{dim}(\mathrm{Ch}(\mathrm{gr}_{\bullet}^\Omega M))=2n=2\,\mathrm{dim}T_{Y}X$, then $\mathrm{gr}^\Omega_{\bullet} M$ is holonomic over $T_Y X$.

    Now, take
    \[\theta=\bigoplus \limits_{i}(t\partial_{t}+i) : \gr_\bullet^\Omega M \to \gr_\bullet^\Omega M\quad \text{ for } t\partial_t + i: \gr_i^\Omega M \to \gr_i^\Omega M\]
    since one can verify that for $P\in\D_X$
    \[P \in V_{i}\D_X \Leftrightarrow P = p(x,\partial_x) t^{\alpha}\partial_t^\beta\quad \text{ for } p(-,-) \text{ a polynomial, and }\beta - \alpha \le i\]
    which implies
    \[[t\partial_t, P] = -i P \text{ for } P\in\gr_i^{V}\mathcal{D}_X,\]
    so we have $\theta\in\mathrm{End}_{\mathcal{D}_{T_{Y}X}}(\mathrm{gr}_{\bullet}^\Omega M)$.
    Indeed, for $u \in \gr_j^\Omega M$ and $P \in \gr_k^{V} \D_X$, we calculate
    \[
        \theta(Pu) = (t\partial_t + j + k) Pu = (P t\partial_t - k P + j + k) u = P(t\partial_t + j) u = P \theta(u).
    \]

    By Lemma \ref{lemma 8.5}, there exists $b(s)\in\mathbb{C}[s]$ such that $b(\theta)=0$,
    In particular, $b(t\partial_{t}+k)$ kills $\frac{\Omega_{k}M}{\Omega_{k-1}M}$ for all $k$.
\end{proof}

\begin{remark}\label{rmk:b_function_of_C[x,1/x]e^1/x}
    It is possible that $\mathrm{gr}_{\bullet}^\Omega M=0$. In fact, take
    $M := \mathcal{D}_{\Spec{\C[t]}} \big/ \mathcal{D}_{\Spec{\C[t]}}\langle t^2\partial_t + 1 \rangle = \mathcal{D}_{X}\cdot e^{\frac{1}{t}}$,
    for $e^{\frac{1}{t}}$ represents the canonical section $[1]\in\mathcal{D}_{\Spec{\C[t]}} \big/ \langle t^2\partial_t + 1 \rangle$.

    We note that $t$ also acts bijectively on $M$ (just as Example~\ref{ex:first-examples-of-d-modules}),
    so $M=M[t^{-1}]  = \mathbb{C}[t, \frac{1}{t}]e^{\frac{1}{t}}$,
    take $\Omega_{k}M=V_{k}\mathcal{D}_{X}\cdot e^{\frac{1}{t}}$,
    which is a good filtration.
    Since $t\partial_{t}e^{\frac{1}{t}}= -\frac{1}{t} e^{\frac{1}{t}}$, we have $\Omega_{k}M=M$ and $b(s)=1$.
\end{remark}

\begin{theorem}[Kashiwara-Malgrange]\label{Theorem 8.7}
    Let $M$ be a holonomic $\mathcal{D}_{X}$-module, there exists unique good filtration $V_{\bullet}M$ and $b(s)\in\mathbb{C}[s]$ s.t.
    \begin{enumerate}
        \item $\mathrm{Re}(\text{roots of }b(s))\subseteq[0,1)$;
        \item $b(t\partial_{t}+k)$ kills $\frac{V_{k}M}{V_{k-1}M}$.
    \end{enumerate}
    And the filtration is called $V$-filtration.
\end{theorem}
\begin{remark}
    We use the definition of $V$-filtration in \cite[2.10.21~The Kashiwara-Malgrange filtration. and 2.10.24~The existence]{Bjork1993AnalyticDModules} here.
\end{remark}
\begin{proof}
    This is Kashiwara's trick:

    First, set $\Gamma_{n}=\Omega_{n+k}$. Note that
    \[
        b(t\partial_{t}+n+k)\text{ kills }\Gamma_{n} /\Gamma_{n-1}\text{ for all }n,
    \]
    which means the roots of the $b$-function are shifted by $k$.
    Thus by taking $k\gg 0$, after replacing $b(s)$ by $s\mapsto b(s+k)$, we can assume $\mathrm{Re}(\lambda)\le 0$ for all roots $\lambda$ of $b(s)$.

    Next, suppose that $b(s)$ has a root $\alpha$ whose real part is $< 0$.
    Set
    \[
        \Gamma_{n}'=\Gamma_{n-1}+(t\partial_{t}+n-\alpha)\Gamma_{n}.
    \]
    If $\beta(s) = b(s)/(s-\alpha)$, then we verify that
    \[
        \beta(s+n)(s+n-\alpha-1)\text{ annihilates }\Gamma'_{n} / \Gamma'_{n-1}\text{ for all }n.
    \]

    Denote $s = t\partial_t$. Let $u' \in \Gamma'_{n}$. We can write $u' = v + (s+n-\alpha)w$ with $v \in \Gamma_{n-1}$ and $w \in \Gamma_{n}$.
    We calculate the action of $Q(s) = \beta(s+n)(s+n-\alpha-1)$:
    \[
        Q(s)u' = \beta(s+n)(s+n-\alpha-1)v + \beta(s+n)(s+n-\alpha-1)(s+n-\alpha)w.
    \]
    For the first term involving $v \in \Gamma_{n-1}$: note that
    \[
        (s+n-\alpha-1)v \in (s+(n-1)-\alpha)\Gamma_{n-1} \subseteq \Gamma'_{n-1}.
    \]
    For the second term involving $w \in \Gamma_{n}$: we use the fact that polynomials in $s$ commute, so
    \begin{align*}
        \beta(s+n)(s+n-\alpha-1)(s+n-\alpha)w & = (s+n-\alpha-1) \beta(s+n)(s+n-\alpha) w \\
                                              & = (s+n-\alpha-1) b(s+n) w.
    \end{align*}
    By the hypothesis on $\Gamma_\bullet$, we have $b(s+n)w \in \Gamma_{n-1}$. Let $z = b(s+n)w$.
    Then
    \[
        (s+n-\alpha-1)z \in (s+(n-1)-\alpha)\Gamma_{n-1} \subseteq \Gamma'_{n-1}.
    \]
    Thus, $Q(s)u' \in \Gamma'_{n-1}$.

    Continue doing this if $\alpha$ has multiplicity, we will finally move a root $\alpha$ of $b(s)$ to $\alpha+1$, increasing its real part.

    For uniqueness, see \cite[2.10.24 The existence of the Kashiwara-Malgrange filtration]{Bjork1993AnalyticDModules}.

\end{proof}

\begin{definition}
    $M$ is a holomonic module, then:

    \begin{enumerate}
        \item  $M$ is called $Q\textbf{-specializable}$ along $Y$ if the $b$-function has all its roots in $Q$, where $Q \subseteq \mathbb{C}$ is a subfield.

              Since any subfield of $\C$ must contain its \textbf{prime subfield} $\mathbb{Q}$, and $b(t\partial_t + k)$ kills $V_kM/V_{k-1}M$,
              the eigenvalues $\lambda$ of $t\partial_t$ on $V_kM/V_{k-1}M$ kills its eigenvectors by multiplying $b(\lambda + k)$.

              So $\lambda$ differs from the roots of $b(s)$ by integers.
              Thus, if all roots of $b(s)$ are in $Q$, then all eigenvalues of $t\partial_t$ are in $Q$ (as $\mathbb{Z} \subseteq \mathbb{Q} \subseteq Q$).

        \item If $M$ is $\mathbb{Q}$-specializable,
              then for all $\beta \in \mathbb{Q}$,
              take $k = \lceil \beta \rceil$,
              then define
              $$V_{\beta} M = \pi^{-1} \left( \bigoplus_{\alpha \leq \beta} \left( \frac{V_k M}{V_{k-1} M} \right)^{\alpha} \right),$$
              and
              $$V_{<\beta} M = \pi^{-1} \left( \bigoplus_{\alpha < \beta} \left( \frac{V_k M}{V_{k-1} M} \right)^{\alpha} \right),$$
              where $\left( \frac{V_k M}{V_{k-1} M} \right)^{\alpha}$ denotes the $\alpha$-generalized eigenspace of $\frac{V_k M}{V_{k-1} M}$
              with respect to the $t\partial_t$-action, and $\pi: V_k M \rightarrow \frac{V_k M}{V_{k-1} M}$;
              Then we get a $\mathbb{Q}$-indexed $V$-filtration s.t.
              $$\mathrm{gr}_{\beta}^{V} M = \frac{V^{\beta} M}{V^{<\beta} M} = \left( \frac{V_k M}{V_{k-1} M} \right)^{\beta}$$
              with $t\partial_t - \beta$ acts on $\mathrm{gr}_{\beta}^{V} M$ nilpotently.
    \end{enumerate}
\end{definition}
\begin{example}
    Let $X=\mathbb{A}^{1}_{\mathbb{C}}$ and $M = \mathbb{C}\left[ t, \frac{1}{t} \right]e^{\frac{1}{t}}$ an irregular holonomic $\mathcal{D}$-mod.

    Consider $V$-filtration (of $\D_X$) along $t = 0$. The $V$-filtration on $M$ exists by Theorem \ref{Theorem 8.7}.

    Recall Rmk.~\ref{rmk:b_function_of_C[x,1/x]e^1/x}. The filtration $\Omega_\bullet M$ is defined by $\Omega_i M := V_i \mathcal{D}_{X} \cdot e^{\frac{1}{t}}$ for $i \in \mathbb{Z}$.
    We have $\Omega_i M = M$ for all $i\in \mathbb{Z}$.

    This filtration gives a $b$-function $b(s) = 1$. Which has all its roots in $[0,1)$, so $\Omega_\bullet M$ is exactly the $V$-filtration on $M$.
\end{example}

\begin{theorem}\label{thm:V-filtration_of_RH_D-module_is_not_trivial}
    If M is a regular holonomic $\mathcal{D}_X$-mod, then its V-filtration is not trivial. In particular, its b-function is not $1$.
\end{theorem}

\begin{proof}[Sketch of proof]

    See \cite{Wu2021CharacteristicCyclesHolonomicDmodules} for details of each step.

    First, each $V_i M$ is a $V_0 \mathcal{D}_X$-module and $V_0 \mathcal{D}_X = \mathcal{D}_X^{\log}$ w.r.t. $(t = 0)$. Since $V_i M$ has no $t$-torsion for all $i < 0$, $\mathrm{Ch}(V_i M)$ has no component over $(t = 0)$, for all $i < 0$.

    Further, we obtain that $\mathrm{Ch}^{\log}\left(\frac{V_i M}{V_{i-1} M}\right) = \mathrm{Ch}^{\log}\left(\frac{V_i M}{t V_i M}\right)= \mathrm{Ch}^{\log}(V_i M)\big|_{(t=0)}$, thus $\frac{V_i M}{V_{i-1} M} \neq 0.$

\end{proof}

And work out its definition when $X$ is a smooth affine $\C$ variety, and the result should be consistent with \cite[discussion above Prop.~1.4]{Budur04}.

\begin{proposition}[Functor of points]\label{prop:functor_of_points}
    We have following \textbf{natural isomorphism}, for $\mathbb{A}^1_\C = \mathbf{G}_{a,\Spec\C}$, and any $\Spec\C$-scheme $T$, we have
    \[
        \Hom_{\mathrm{Sch}_{\Spec\C}}(T , \mathbb{A}^1_\C) \cong \Gamma(T, \mathcal{O}_T).
    \]
\end{proposition}
\begin{proof}
    See \cite[\href{https://stacks.math.columbia.edu/tag/022V}{Tag 022V}]{stacks-project}.
\end{proof}

\begin{definition}[Graph embedding]\label{def:graph_embedding}
    We define \textbf{graph embedding}, which is induced by the universal property of a fiber product $X\times_{\Spec\C} \mathbb{A}^1_{\C}$:
    $$
        \begin{tikzcd}
            X \arrow[drr, "f", bend left=20] \arrow[ddr, "\operatorname{id}_X"', bend right=20] \arrow[dr, "\exists! \Gamma_f" description, dashed] & & \\
            & X \times_{\Spec\C} \mathbb{A}^1_{\C} \arrow[r, "p_2"] \arrow[d, "p_1"] \arrow[dr, phantom, "\lrcorner", very near start] & \mathbb{A}^1_{\C} \arrow[d] \\
            & X \arrow[r] & \Spec\C
        \end{tikzcd}
    $$
    where $f\in\Hom_{\mathrm{Sch}_{\Spec\C}}(X, \mathbb{A}^1_{\C})$ and the dashed arrow $\Gamma_f:X\to X\times_{\Spec\C}\mathbb{A}^1_{\C}$ is the graph embedding.
\end{definition}

\begin{example}\label{example:direct_image_of_graph_embedding}
    Now we begin to calculate the $\D$-module theoretic \textbf{direct image} (see Rmk.~\ref{rem:kashiwara_explicit}) of $\Gamma_f$ when $X$ is a smooth affine $\C$ variety.

    \begin{enumerate}
        \item \textbf{Step 1:} Identify $X$ as a closed subscheme of $X\times_{\Spec\C}\mathbb{A}^1_{\C}$.

              Base change $p_1:X\times_{\Spec\C}\mathbb{A}^1_{\C} \to X$ of $\mathbb{A}^1_\C \to \Spec\C$ is also separated, since $\mathbb{A}^1_\C \to \Spec\C$ is separated.
              As a section of the separated morphism $p_1$ ($p_1\circ\Gamma_f = \mathrm{id}_X$), $\Gamma_f$ is a closed immersion \cite[\href{https://stacks.math.columbia.edu/tag/024T}{Tag 024T}]{stacks-project}.
              So we can use \cite[\href{https://stacks.math.columbia.edu/tag/01HO}{Tag 01HO}]{stacks-project} to identify $X$ as a closed subscheme of $X\times_{\Spec\C}\mathbb{A}^1_{\C}$.

              We identify $f\in \Gamma(X,\Ox) =: R$ by Prop.~\ref{prop:functor_of_points}.
              Since $X$ is affine, we have $\Gamma_f$ is also an affine morphism, on the level of ring map between global sections, we have:
              \begin{align*}
                  R[t] = R \otimes_\C \C[t] & \to R               \\
                  a \otimes b(t)            & \mapsto a\cdot b(f)
              \end{align*}
              So the kernel of this ring map is $\langle t - f \rangle$. $X\cong V(\langle t - f \rangle) \subseteq X\times_{\Spec\C}\mathbb{A}^1_{\C}$.

        \item \textbf{Step 2:} Introduce two coordinate systems on $Y:=X\times_{\Spec\C}\mathbb{A}^1_{\C}$.

              Let $u:=t-f$. Then $Y$ has two coordinate systems $(x_i,t)$ and $(x_i,u)$.
              There is a fact that both $\{\mathrm{d} x_i,\mathrm{d} t\}$ and $\{\mathrm{d} x_i,\mathrm{d} u\}$ are basis of $\Omega_Y = \Omega_{R[t]/\C[t]}$.
              (that is why there is graph embedding, locus of $f$ may not smooth, but $t-f$ must be in a smooth coordinate system.)

              They have corresponding dual basis, say $\{\partial^{(t)}_{x_i},\partial_t\}$ and $\{\partial^{(u)}_{x_i},\partial_u\}$, of $\mathcal{T}_Y\subseteq \D_Y$.

              One should realize that for a vector space without additional structure,
              $V$ and $V^\vee$ DO NOT have a \textbf{canonical} isomorphism between them.
              So $\partial^{(t)}_{x_i}$ and $\partial^{(u)}_{x_i}$ are \underline{truly different}.

              (Due to the same fact, the identification of $\mathcal{T}_Y\hookrightarrow\End_{\C}(\Oy)$, as a subspace of $\D_Y$, should be \textbf{coordinate-free}.)

              We calculate the transformation matrix between the two basis:

              \begin{enumerate}
                  \item \textbf{Differentials (Cotangent bundle):}
                        Since $u = t - f(x)$, we have the relation of differentials:
                        \[
                            \mathrm{d} t = \mathrm{d} u + \sum_{i} (\partial_{x_i} f) \mathrm{d} x_i.
                        \]

                  \item \textbf{For $\partial_u$:}
                        By duality definition, $\langle \mathrm{d} x_i, \partial_u \rangle = 0$ and $\langle \mathrm{d} u, \partial_u \rangle = 1$.
                        Using the differential relation:
                        \[
                            \langle \mathrm{d} t, \partial_u \rangle = \langle \mathrm{d} u + \sum (\partial_{x_k} f) \mathrm{d} x_k, \partial_u \rangle = 1 + 0 = 1.
                        \]
                        And $\langle \mathrm{d} x_k, \partial_u \rangle = 0$.
                        This characterizes $\partial_t$, so we have $\partial_u = \partial_t$.

                  \item \textbf{For $\partial_{x_i}^{(u)}$:}
                        Write $\partial_{x_i}^{(u)} = \sum_j a_j\partial_{x_j}^{(t)} + b \partial_t$.

                        Apply $\mathrm{d} x_k$:
                        \[
                            \delta_{ik} = \langle \mathrm{d} x_k, \partial_{x_i}^{(u)} \rangle
                            = \sum_j a_j \langle \mathrm{d} x_k, \partial_{x_j}^{(t)} \rangle + b \langle \mathrm{d} x_k, \partial_t \rangle
                            = a_k.
                        \]
                        So $\partial_{x_i}^{(u)} = \partial_{x_i}^{(t)} + b \partial_t$.

                        Apply $\mathrm{d} t = \mathrm{d} u + \sum_k (\partial_{x_k} f) \mathrm{d} x_k$:
                        \[
                            b = \langle \mathrm{d} t, \partial_{x_i}^{(u)} \rangle
                            = \langle \mathrm{d} u + \sum_k (\partial_{x_k} f) \mathrm{d} x_k, \partial_{x_i}^{(u)} \rangle
                            = \partial_{x_i} f.
                        \]
                        So the coordinate change is given by:
                        \[
                            \partial_{x_i}^{(u)} = \partial_{x_i}^{(t)} + (\partial_{x_i} f) \partial_t.
                        \]
              \end{enumerate}

              In summary, we have
              \begin{align*}
                  \partial_u           & = \partial_t,                                           \\
                  \partial_{x_i}^{(u)} & = \partial_{x_i}^{(t)} + (\partial_{x_i} f) \partial_t.
              \end{align*}


        \item \textbf{Step 3:} Identify the transfer bimodules.

              Let $Y := X\times_{\Spec\C}\mathbb{A}^1_{\C}$,
              $X$ has coordinate system $(x_i)$,
              and $\mathbb{A}^1_\C$ has coordinate system $(t)$,
              then:

              We have $\D_{X\to Y} := \Ox \otimes_{\Gamma_f^{-1}\Oy} \Gamma_f^{-1}\D_Y = \Ox[\partial^{(t)}_{x_i}, \partial_t]$ as free $\Ox$-module.

              We calculate its left $\D_X$-module structure, using the following formula in Example~\ref{example:side-changing-transfer-bimodules}:
              \begin{align*}
                  a \cdot (b \otimes P)      & := (ab) \otimes P,                                                                  \\
                  \theta \cdot (a \otimes P) & := \theta(a) \otimes P + a \sum_j \theta(\Gamma_f^* y_j) \otimes (\partial_{y_j} P)
              \end{align*}
              Which translates to:
              \begin{align*}
                  x_i \cdot (b \otimes P)            & := (x_i b) \otimes P,                                                                                  \\
                  \partial_{x_i} \cdot (a \otimes P) & := \partial_{x_i}(a) \otimes P + a (1\otimes (\partial_{x_i} P) + \partial_{x_i}f\otimes \partial_t P)
              \end{align*}
              for $a,b\in\Ox, P\in\C[\partial^{(t)}_{x_i}, \partial_t]$.

              And we have the right $\Gamma_f^{-1}\D_Y$-module structure:
              \begin{align*}
                  (b \otimes P) \cdot x_i                  & := (x_i b) \otimes P,                  \\
                  (b \otimes P) \cdot t                    & := (f b) \otimes P,                    \\
                  (a \otimes P) \cdot \partial^{(t)}_{x_i} & := a \otimes (P \partial^{(t)}_{x_i}), \\
                  (a \otimes P) \cdot \partial_t           & := a \otimes (P \partial_t).
              \end{align*}

              The transfer bimodule $\mathcal{D}_{Y \leftarrow X}$ is a side change of $\D_{X\rightarrow Y}$,
              we need the identification in \cite[Example~1.3.5]{hotta2008d} to help us to calculate, that is:
              \[
                  \D_{Y\leftarrow X} = (\Gamma_f^{-1}\D_Y \otimes_{\Gamma_f^{-1}\Oy} \Ox) \otimes_{\Ox} (\Gamma_f^{-1}\omega_Y^{\otimes -1}\otimes_{\Gamma_f^{-1}\Oy} \omega_X)
              \]
              Since $\Gamma_f^{-1}\omega_Y^{\otimes -1}\otimes_{\Gamma_f^{-1}\Oy} \omega_X$ is locally free, and $X$ is \textbf{affine}, so it is free, so we get:
              \[
                  \D_{Y\leftarrow X} = \Gamma_f^{-1}\D_Y \otimes_{\Gamma_f^{-1}\Oy} \Ox.
              \]
              Since $\Gamma_f^{-1}\D_Y = \C[\partial^{(t)}_{x_i}, \partial_t]\otimes_\C \Gamma_f^{-1}\Oy$ is free \textbf{right} $\Gamma_f^{-1}\Oy$-module (see the second point of Lem.~\ref{lem:quasi-coherent-sheaf-and-localization-of-D-modules}),
              so $\D_{Y\leftarrow X}= \C[\partial^{(t)}_{x_i}, \partial_t]\otimes_\C \Ox$ is a free \textbf{right} $\Ox$-module.

              The right $\mathcal{D}_X$-module structure is given by unwinding the 4 Steps at the end of Exercise~\ref{example:side-changing-transfer-bimodules}:
              \begin{align*}
                  (P \otimes b) \cdot x_i            & := P \otimes (b x_i),                                                                                      \\
                  (P \otimes b) \cdot \partial_{x_i} & := ((P \partial_{x_i}) \otimes 1 + (P\partial_t)\otimes \partial_{x_i}f)b + P \otimes (\partial_{x_i}(b)),
              \end{align*}

              And the left $\Gamma_f^{-1}\D_Y$-module structure is given by:
              \begin{align*}
                  x_i \cdot (P \otimes b)                  & := (x_i P) \otimes b             = [x_i, P]\otimes b + P\otimes (x_i b), \\
                  \partial^{(t)}_{x_i} \cdot (P \otimes b) & := (\partial^{(t)}_{x_i} P) \otimes b,                                   \\
                  t \cdot (P \otimes b)                    & := (t P) \otimes b               = [t, P]\otimes b + P\otimes (f b),     \\
                  \partial_t \cdot (P \otimes b)           & := (\partial_t P) \otimes b.
              \end{align*}

        \item \textbf{Step 4:} Identify the direct image.

              For a left $\D_X$-module $M$, we recall the definition of the direct image of $M$ under the graph embedding $\Gamma_f$,
              see Rmk.~\ref{rem:kashiwara_explicit}:
              \[
                  \int_{\Gamma_f} M := R\Gamma_{f*}\left(\mathcal{D}_{Y \leftarrow X} \otimes^L_{\mathcal{D}_X} M\right)
              \]

              Firstly, since $\Gamma_f$ is an affine closed embedding, if we take a new coordinate system $(x_i,t-f)$ on $Y$,
              one can prove $\D_{Y\leftarrow X}$ is a free \textbf{right} $\D_X$-module, so it does not need a resolution.

              Secondly, pushforward by a closed immersion (in this case $\Gamma_f$) is an exact functor between quasi-coherent sheaves,
              see \cite[\href{https://stacks.math.columbia.edu/tag/01AX}{Tag 01AX}]{stacks-project}.

              Combining the above two points, we get
              \[
                  R\Gamma_{f*}\left(\mathcal{D}_{Y \leftarrow X} \otimes^L_{\mathcal{D}_X} M\right) = \Gamma_{f*}\left(\mathcal{D}_{Y \leftarrow X} \otimes_{\mathcal{D}_X} M\right)
              \]
              Through a categorical equivalence, says a quasi-coherent sheaf on an affine scheme is equivalent to its \textbf{global sections}, that is, a module.
              To identify
              \[
                  \Gamma_{f*}\left(\mathcal{D}_{Y \leftarrow X} \otimes_{\mathcal{D}_X} M\right),
              \]
              if $M$ is a quasi-coherent sheaf (see Lem.~\ref{lem:quasi-coherent-sheaf-and-localization-of-D-modules}), we just denote
              \[
                  \mathcal{D}_{Y \leftarrow X} \otimes_{\mathcal{D}_X} M
              \]
              as its global sections.

        \item \textbf{Step 5:} Identification as a left $\D_Y$-module, the result should be compatible with \cite[discussion above Prop.~1.4]{Budur04}.

              Recall, \cite[Example~1.3.5.]{hotta2008d}, for given coordinate $(x_i, u)$ (not $(x_i, t)$) on $Y$, we have
              \[
                  \D_{Y \leftarrow X} = \bigoplus_{n \ge 0} \partial_u^n \D_X.
              \]
              So we have
              \[
                  \D_{Y \leftarrow X} \otimes_{\D_X} M = \bigoplus_{n \ge 0} \partial_u^n \D_X \otimes_{\D_X} M = \bigoplus_{n \ge 0} \partial_u^n M.
              \]

              So we present elements in $\D_{Y \leftarrow X} \otimes_{\D_X} M$ in the form of
              \[
                  P\otimes_\C Q \otimes_{\D_X} m\quad (P\in\C[\partial_t]=\C[\partial_u],\, Q\in\D_X = \C[\partial^{(u)}_{x_i}]\otimes_\C\Ox,\, m\in M),
              \]
              Considering the isomorphism
              \[
                  \bigoplus_{n \ge 0} \partial_u^n \D_X \otimes_{\D_X} M = \bigoplus_{n \ge 0} \partial_u^n M,
              \]
              we can take $Q=1$ in the following calculation.

              The following calculations heavily depend on the identified right $\D_X$-module structure and left $\Gamma_f^{-1}\D_Y$-module structure on $\D_{Y\leftarrow X}$ in Step 3.
              All following Lie bracket are taken in $\D_Y\to \D_Y\otimes_{\Oy}\Ox = \D_{Y\leftarrow X} \to \D_{Y\leftarrow X}\otimes_{\D_X} M$.

              For $g\in \Ox\subseteq\Oy\subseteq\D_Y$, we have
              \[
                  g\cdot(\partial_t^\nu \otimes n) = \partial_t^\nu\otimes (gn)
              \]
              The derivation is as follows:
              \begin{align*}
                  g\cdot(\partial_t^\nu\otimes 1 \otimes n) & := (g\cdot (\partial_t^\nu\otimes 1))\otimes n                                                                                        \\
                                                            & = ([g,\partial_t^\nu]\otimes 1 + \partial_t^\nu\otimes g)\otimes n                                                                    \\
                                                            & = ([g,\partial_t^\nu]\otimes 1)\otimes n + (\partial_t^\nu\otimes 1)\otimes (gn)                                                      \\
                                                            & = (\partial_t^\nu\otimes 1)\otimes (gn)                                          &  & \text{($[g,\partial_t^\nu]=0$, for $g\in\Ox$)}.
              \end{align*}

              For $t\in (\Oy\setminus\Ox) \subseteq \D_Y$, we have
              \[
                  t\cdot(\partial_t^\nu\otimes n) = \partial_t^\nu\otimes (fn) - \nu\cdot(\partial_t^{\nu-1}\otimes n)
              \]
              The derivation is as follows:
              \begin{align*}
                  t\cdot(\partial_t^\nu\otimes 1 \otimes n) & := (t\cdot (\partial_t^\nu\otimes 1))\otimes n                                        \\
                                                            & = ([t,\partial_t^\nu]\otimes 1 + \partial_t^\nu\otimes f)\otimes n                    \\
                                                            & = (-\nu\partial_t^{\nu-1}\otimes 1 + \partial_t^\nu\otimes f)\otimes n                \\
                                                            & = (-\nu\partial_t^{\nu-1}\otimes 1)\otimes n + (\partial_t^\nu\otimes 1)\otimes (fn).
              \end{align*}


              For $\partial^{(t)}_{x_i}$, we have
              \[
                  \partial^{(t)}_{x_i}\cdot(\partial_t^\nu\otimes n) = \partial_t^\nu\otimes(\partial_{x_i}n) - \partial_t^{\nu+1}\otimes\bigl((\partial_{x_i}f)n\bigr)
              \]
              The derivation is as follows, we first consider what $\partial_t^\nu\otimes 1 \otimes (\partial_{x_i}n)$ is:
              \begin{align*}
                  \partial_t^\nu\otimes 1 \otimes (\partial_{x_i}n) & = ((\partial_t^\nu\otimes 1)\cdot\partial_{x_i})\otimes n                                                                                                  \\
                                                                    & = (\partial_t^\nu \otimes \partial^{(t)}_{x_i} + (\partial_t^{\nu+1})\otimes \partial_{x_i}f)1 + \partial_t^\nu \otimes (\partial^{(t)}_{x_i}(1))\otimes n \\
                                                                    & = (\partial_t^\nu \otimes \partial^{(t)}_{x_i} + (\partial_t^{\nu+1})\otimes \partial_{x_i}f)\otimes n                                                     \\
                                                                    & =  (\partial_t^\nu \otimes \partial^{(t)}_{x_i})\otimes n + (\partial_t^{\nu+1}\otimes \partial_{x_i}f)\otimes n.
              \end{align*}
              So we have
              \begin{align*}
                  \partial^{(t)}_{x_i}\cdot(\partial_t^\nu\otimes 1\otimes n) & := (\partial^{(t)}_{x_i}\cdot(\partial_t^\nu\otimes 1))\otimes n                                                        \\
                                                                              & =(\partial_t^\nu \otimes \partial^{(t)}_{x_i})\otimes n                                                                 \\
                                                                              & = \partial_t^\nu\otimes 1 \otimes (\partial_{x_i}n) - (\partial_t^{\nu+1}\otimes \partial_{x_i}f)\otimes n              \\
                                                                              & =\partial_t^\nu\otimes 1 \otimes (\partial_{x_i}n) - \partial_t^{\nu+1}\otimes 1\otimes \bigl((\partial_{x_i}f)n\bigr).
              \end{align*}


              In summary, we have
              \begin{align*}
                  \text{Block A: } & \quad g\cdot(\partial_t^\nu\otimes n) =\partial_t^\nu\otimes(gn),                                                                                         \\
                  \text{Block B: } & \quad \partial_{t}\cdot(\partial_t^\nu\otimes n) =\partial_t^{\nu+1}\otimes n,                                                                            \\
                  \text{Block C: } & \quad \partial_{x_i}^{(t)}\cdot(\partial_t^\nu\otimes n) =\partial_t^\nu\otimes(\partial_{x_i}n)-\partial_t^{\nu+1}\otimes\bigl((\partial_{x_i}f)n\bigr), \\
                  \text{Block D: } & \quad t\cdot(\partial_t^\nu\otimes n) =\partial_t^\nu\otimes fn-\nu_j\,\partial_t^{\nu-1}\otimes n,
              \end{align*}
              Which is consistent with \cite[discussion above Prop.~1.4]{Budur04}.

        \item \textbf{Step 6:} Take $M= R_f$, further identification, the result should be consistent with \cite[the $\D_X$-action on $B_{f,+}$]{Lee2025MultiplicativeThomSebastianiBSato_JAlgebra}.

              Take $\delta_f := 1\otimes 1\in \C[\partial_t]\otimes_{\C}R_f$.

              Define
              \[
                  s:=-\partial_t t.
              \]
              Then one can prove
              \[
                  \C[\partial_t]\otimes_{\C}R_f = R_f[s] \delta_f.
              \]

              To be consistent with \cite[the $\D_X$-action on $B_{f,+}$]{Lee2025MultiplicativeThomSebastianiBSato_JAlgebra},
              we need to verify
              \[
                  \theta\cdot(us^j f^s)
                  =\theta(u)s^j f^s+\frac{\theta(f)}{f}u\,s^{j+1}f^s.
              \]
              Where $\theta\in\mathcal{T}_X\subseteq\D_X$, $u\in R_f$, here $f^s$ just denotes $\delta_f$.


              Recall:
              \[
                  B_{f,+} := \Gamma_{f,+}(R_f)\cong\mathbb{C}[\partial_t]\otimes_{\mathbb{C}}R_f,\qquad
                  \delta_f=1\otimes 1,\qquad u\delta_f=1\otimes u,
              \]
              apply Step~5 Block~C at $\nu=0$:
              \[
                  \partial_{x_i}^{(t)}\cdot(1\otimes u)
                  =1\otimes(\partial_{x_i}u)-\partial_t\otimes\bigl((\partial_{x_i}f)u\bigr),
              \]
              i.e.
              \[
                  \partial_{x_i}^{(t)}\cdot(u\delta_f)
                  =(\partial_{x_i}u)\delta_f-(\partial_{x_i}f)u\,\partial_t\delta_f.
              \]
              For $\theta=\sum_i a_i\partial_{x_i}$ (Step~5 Block~A+C):
              \[
                  \theta\cdot(u\delta_f)
                  =\sum_i a_i(\partial_{x_i}u)\delta_f-\sum_i a_i(\partial_{x_i}f)u\,\partial_t\delta_f
                  =\theta(u)\delta_f-(\theta f)u\,\partial_t\delta_f.
              \]
              Since by Step~5 Block~B+D:
              \[
                  s\delta_f := s\cdot\delta_f=-\partial_t\cdot(t\cdot\delta_f)=-\partial_t\cdot(f\delta_f)=-f\,\partial_t\delta_f,
              \]
              we get
              \[
                  -\partial_t\delta_f=\frac{1}{f}s\delta_f,
              \]
              hence
              \[
                  \theta\cdot(u\delta_f)=\theta(u)\delta_f+\frac{\theta(f)}{f}u\,s\delta_f.
              \]
              Also in $\D_Y$:
              \[
                  [\theta,s]=0
              \]
              so
              \[
                  \theta\cdot(us^j\delta_f)
                  =\theta(u)s^j\delta_f+\frac{\theta(f)}{f}u\,s^{j+1}\delta_f.
              \]
              Replacing $\delta_f$ by $f^s$:
              \[
                  \theta\cdot(us^j f^s)
                  =\theta(u)s^j f^s+\frac{\theta(f)}{f}u\,s^{j+1}f^s.
              \]

              By restricting the action to a $\C$-subalgebra $\D_X[s]\subseteq\D_Y$, $R_f[s]f^s$ becomes a $\D_X[s]$-module.
    \end{enumerate}

\end{example}

\begin{example}\label{Example 8.11}
    The text below is the original treatment of this example in the handwritten lecture note,
    in the latter Remark~\ref{remark:second_treatment_of_example_8.11} we give another treatment the editor thinks more rigorous.

    \medskip

    Let $X = \mathbb{A}_{\mathbb{C}}^n$ and $Y = \mathbb{A}_{\mathbb{C}}^n \times_{\Spec{\C}} \mathbb{A}_{\mathbb{C}}^1$, here $\mathbb{A}^{1}_{\mathbb{C}}$ is with coordinate $t$.
    For $f \in \mathbb{C}[x_1, \cdots, x_n]$, Consider
    \[
        M = \mathbb{C}[x_1, \cdots, x_n]_f[s] f^s.
    \]

    Define $t M = f M$ and then $t^{-1} M = f^{-1} M$.

    Take $s = -t\partial_t$, then $\partial_t = -t^{-1} s$, the $\partial_{t}$-action is defined,
    and further, $M$ is a $\mathcal{D}_Y = \mathcal{D}_X \langle t, \partial_t \rangle$-module.

    Then
    \[
        N := \mathbb{C}[x_1, \cdots, x_n][\partial_t] f^s= \mathbb{C}[x_1, \cdots, x_n][t^{-1} s] f^s,
    \]
    and
    \[
        \partial_{x_i} f^s = \partial_{x_i}(f) \cdot \frac{1}{f} s f^s= \partial_{x_i}(f) \cdot (-\partial_t) f^s,
    \]
    thus $N$ is a $\mathcal{D}_Y$-submodule of $M$.
    Furthermore, we can check that $\mathrm{Ch}(N)$ = the conormal bundle of the graph of $X$,
    which indicates that $N$ is a holonomic $\mathcal{D}_{Y}$-module.

    We consider $V_{i}\mathcal{D}_{Y}\cdot f^{s}$ forming a good filtration of $N$:
    \[
        V_0 \mathcal{D}_Y \cdot f^s = \mathcal{D}_X \langle t, s \rangle f^s = \mathcal{D}_X[s] f^s,
    \]
    Similarly, $V_{-1} \mathcal{D}_Y \cdot f^s = \mathcal{D}_X[s] f^{s+1}$.

    By Theorem \ref{Theorem 8.4}, there exists $b(s) \in \mathbb{C}[s]$ s.t.
    \[
        b(s) \cdot \frac{V_0 \mathcal{D}_Y \cdot f^s}{V_{-1} \mathcal{D}_Y \cdot f^s} = b(s) \cdot \frac{\mathcal{D}_X[s] f^s}{\mathcal{D}_X[s] f^{s+1}} = 0.
    \]
    which is to say there exists $P(s)\in \mathcal{D}_{X}[s]$ such that $b(s) f^s = P(s) \cdot f^{s+1}$

    In fact, $N = \iota_{f,+} (\mathbb{C}[x_1, \ldots, x_n])$, the $\mathcal{D}$-module pushforward, where
    \[
        \iota_f: X \to Y = X \times \mathbb{A}^{1}_{\mathbb{C}}\atop x \mapsto (x, f(x))
    \]
    the graph embedding. So $N$ is regular holonomic, by Thm.~\ref{thm:V-filtration_of_RH_D-module_is_not_trivial}, we get $b(s) \neq 1$.

    If one get interested in why a set theoretic like definition $x \mapsto (x, f(x))$ can be used to define a \textbf{scheme theoretic} morphism,
    see \cite[\href{https://stacks.math.columbia.edu/tag/001O}{Tag 001O}]{stacks-project} for the notion of \textbf{functor of points} and a \textbf{representable functor}.
\end{example}

\begin{remark}
    One can replace $\mathbb{C}[x_1, \ldots, x_n]$ by any regular holonomic $\mathcal{D}_X$-module, to get similar results in general.
\end{remark}

\begin{remark}\label{remark:second_treatment_of_example_8.11}
    We give a second treatment of Example~\ref{Example 8.11} here.

    Given $X$ a smooth affine algebraic variety, $f \in \mathcal{O}_X(X)$, $Y = X \times_{\Spec{\C}} \mathbb{A}^1_{\C}$ with coordinate $t$.

    Define $M:=\Gamma_{f,+}\left(\mathcal{O}_X\left[\frac{1}{f}\right]\right)$, and $N:=\Gamma_{f,+}(\mathcal{O}_X)$.

    By Lemma~\ref{lem:quasi-coherent-sheaf-and-localization-of-D-modules}, or more precisely \cite[\href{https://stacks.math.columbia.edu/tag/0G36}{Tag 0G36}]{stacks-project} or \cite[Lem.~4.1., Lem.~4.2.]{chen2024dmodules}.
    We know as $\D_X$-modules, $\Ox$ is a $\D_X$-submodule of $\Ox\left[\frac{1}{f}\right]$.

    Since the $\D$-module theoretic direct image is a \textbf{functor}, we have $N$ is a $\D_Y$-submodule of $M$.

    By Exercise~\ref{example:direct_image_of_graph_embedding}, we get
    \[
        M = \mathcal{O}_X\left[\frac{1}{f}\right][s]f^s, \quad N = \mathcal{O}_X[s]f^s,
    \]
    but this time we define $s = -t\partial_t$.

    \bigskip

    Since $\Ox$ is an integrable connection, push it forward to a divisor completion $j:X\to Y$ such that $Y\setminus X =: D$.
    We get $(j_*(\Ox))^{\mathrm{an}} = \Oy^{\mathrm{an}}(\ast D)$, which is a regular holonomic $\D^{\mathrm{an}}_Y$-module (WLOG $D$ is normal crossing, and use curve test Def.~\ref{def:analytic_regular_holmonicity}).
    By \cite[Thm.~6.11]{hotta2008d}, $\Ox$ is a regular holonomic $\D_X$-module.

    \bigskip

    We calculate the characteristic variety of $N = \Gamma_{f,+}(\mathcal{O}_X)$, by using Lem.~\ref{prop:charvar-closed-emb}.

    Since $\Gamma_f: X \to Y$ is a closed embedding given by the graph (defined by $t = f(x)$),
    so we can split the algebraic/étale coordinate system $(x_1, \dots , x_n, u:= t - f)$ on $Y$, such that $X = \{ u = 0 \}$.

    In $T^\ast X$, $\Ch(\mathcal{O}_X) = T^\ast_X X$ is the zero section, given by $\xi = 0$ ($\xi_i$ corresponds to $\partial_{x_i}$).

    So in $T^\ast Y\mid_X$, $\rho^{-1}_{\Gamma_f}(\Ch(\mathcal{O}_X))$ is also defined by $\xi = 0$
    (this is the intermediate result ``flat algebra'' in the proof in Lem.~\ref{prop:charvar-closed-emb}).

    Finally in $T^\ast Y$, $\Ch(\Gamma_{f,+}(\mathcal{O}_X)) = \widetilde{\Gamma_f}(\rho^{-1}_{\Gamma_f}(\Ch(\mathcal{O}_X)))$ is defined by pulling back the ideal
    \[
        \langle \xi_1, \dots, \xi_n \rangle \subseteq \mathcal{O}_{T^\ast Y\mid_X} = \Sym^\bullet_{\Ox}(\Omega^1_Y \otimes_{\Oy} \Ox).
    \]
    By the ring map
    \[
        \Sym^\bullet_{\Oy}(\Omega^1_Y)\to \Sym^\bullet_{\Ox}(\Omega^1_Y \otimes_{\Oy} \Ox) \text{ i.e. } \Oy[\xi_1, \dots, \xi_n, \tau] \to \Ox[\xi_1, \dots, \xi_n, \tau]
    \]
    Where $\tau$ cooresponds to $\partial_u$.

    Since $\Ox = \Oy\big/\langle u \rangle$, we get $\Ch(\Gamma_{f,+}(\mathcal{O}_X))$ is defined by $\xi_1 = \xi_2 = \dots =\xi_x = u = 0$,
    by the \hyperref[prop:conormal_bundle_in_local_coordinate]{identification}, we see this is exactly $T^*_X Y \subseteq T^\ast Y$
    if we view $X\subseteq Y$ as the graph defined by $u=0$.

    \bigskip

    $\Ox$ is a \textbf{coherent} (finitely generated) $\D_X$-module,
    $\D$-module direct image preserve coherency, so $\Gamma_{f,+}(\mathcal{O}_X)$ is a coherent $\D_Y$-module.
    By the above calculation of characteristic variety, we get $\Gamma_{f,+}(\mathcal{O}_X)$ is a holonomic $\D_Y$-module.

    By holonomicity, when we take a good filtration $V_i \mathcal{D}_Y \cdot f^s$
    (recall $f^s$ represent canonical section $1\otimes 1 \in N = \C[\partial_t]\otimes_\C \Ox$), by Thm.~\ref{Theorem 8.4}
    there is a polynomial $b(-)$ of $s := -t\partial_t$ such that $b(-t\partial_t)$ annihilates the graded piece
    \[
        \frac{V_0 \mathcal{D}_Y \cdot f^s}{V_{-1} \mathcal{D}_Y \cdot f^s}.
    \]
    To see this, notice that by \hyperref[def:definition_of_V_filtration]{definition},
    \[
        V_0\D_Y = \Oy\langle \partial_x, -t\partial_t \rangle
    \]
    When acting on $\Ox[s]f^s$, by Step 5 Block D in Exercise~\ref{example:direct_image_of_graph_embedding}, we have
    \[
        t \cdot f^s = f f^s = f^{s+1}.
    \]
    So the action of $\Oy\langle \partial_x, -t\partial_t \rangle$ factor through
    \[
        \Oy\big/\langle t-f \rangle \otimes_{\Oy} \langle \partial_x, -t\partial_t \rangle = \Ox\langle \partial_x, -t\partial_t \rangle = \D_X[-t\partial_t].
    \]
    (Notation $\D_X[-t\partial_t]$ using square bracket suggests that $-t\partial_t$ commute with $\D_X$ in $\D_Y$,
    so one can apply the \textbf{universal property} of polynomial ring, justifying the polynomial ring notation $\D_X[-t\partial_t]$.).

    \noindent So we have
    \[
        V_0\D_Y \cdot f^s = \D_X[-t\partial_t]f^s.
    \]
    And
    \[
        V_{-1}\D_Y \cdot f^s = V_0\D_Y \cdot t \cdot f^s = \D_X[-t\partial_t]t f^s = \D_X[-t\partial_t]f^{s+1}.
    \]
    So we have
    \[
        \frac{V_0 \mathcal{D}_Y \cdot f^s}{V_{-1} \mathcal{D}_Y \cdot f^s} = \frac{\mathcal{D}_X[s] f^s}{\mathcal{D}_X[s] f^{s+1}}.
    \]
    Justifying the definition (and existence) of Bernstein-Sato polynomial
    \[
        b(-t\partial_t) f^s = P(-t\partial_t) ff^{s}, \text{ for some polynomial $P(-)$ with coefficient in $\D_X$}.
    \]

\end{remark}
