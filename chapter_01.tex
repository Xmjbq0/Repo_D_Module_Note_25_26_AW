\section{Recall of smooth affine algebraic geometry over $\C$}
Let $R$ be an integral commutative $\C$-algebra of finite type. Set
\[
    X := \Spec R,
\]
the affine algebraic variety associated to $R$, endowed with the Zariski topology.
For an ideal $I\subseteq R$, write
\[
    V(I) := \{\, \mathfrak p\in \Spec R \mid \mathfrak p\supseteq I \,\},
\]
so that a subset $V\subseteq X$ is closed iff $V=V(I)$ for some $I$.

We denote the module of K"ahler differentials of $R$ over $\C$ by
\[
    \Omega_R = \Omega_X = \Omega_{R/\C}.
\]
\begin{definition}[K\"ahler differentials]\label{def:kahler}
    The $R$-module $\Omega_{R/\C}$ is generated by formal symbols $\mathrm df$ with $f\in R$,
    subject to the relations, for all $f,g\in R$ and $a,b\in\C$,
    \begin{align*}
        \mathrm d(fg)    & = f\,\mathrm dg + g\,\mathrm df \qquad\text{(Leibniz rule)}, \\
        \mathrm d(af+bg) & = a\,\mathrm df + b\,\mathrm dg.
    \end{align*}
\end{definition}
\begin{exercise}
    Show that $\mathrm dc=0$ for every constant $c\in\C$.
\end{exercise}
\begin{proof}[Solution to the exercise]
    Using linearity and the Leibniz rule,
    \(\mathrm d(1)=\mathrm d(1\cdot 1)=1\,\mathrm d1+1\,\mathrm d1=2\,\mathrm d1\), hence
    \(\mathrm d1=0\). For $c\in\C$ we have
    \(\mathrm dc=\mathrm d(c\cdot 1)=c\,\mathrm d1+1\,\mathrm dc=\mathrm dc\), and the only possibility is
    \(\mathrm dc=0\).
\end{proof}

Define the $R$-module
\[
    \T_X := \Hom_R\!\big(\Omega_{R/\C},\, R\big) \cong \Der_{\C}(R,R),
\]
called the module of $\C$-derivations on $X$ (the vector fields on $X$).

\begin{definition}[Smoothness]
    We say that $R$ (or $X=\Spec R$) is \emph{smooth} if $\Omega_{R/\C}$ is a free $R$-module of
    finite rank. In this case, the rank
    \[
        n := \operatorname{rank}_R(\Omega_{R/\C})
    \]
    which is equal to the topological dimension of $X$.

    \begin{proof}[Prove that $\mathrm{rank}_R(\Omega_{R/\mathbb{C}}) = \dim X$]
        Let $K = \mathrm{Frac}(R)$ denote the fraction field of $R$. Since $R$ is integral.

        First, recall a fundamental result from field theory \cite[Chapter 2, Proposition 8.6A.]{hartshorne1977algebraic}: for a field extension $K/\mathbb{C}$, the module of Kähler differentials $\Omega_{K/\mathbb{C}}$ is a vector space over $K$, and its dimension is equal to the transcendence degree of the extension:
        \[
            \dim_K (\Omega_{K/\mathbb{C}}) = \mathrm{tr.deg}_{\mathbb{C}}(K).
        \]

        One should note that $\mathbb{C}$ is of characteristic 0, which ensures all extensions are separable, even when extensions are not algebraic.

        By the definition of the dimension of an algebraic variety \cite[Chapter 1, Proposition 1.8A.(a)]{hartshorne1977algebraic}, we have:
        \[
            \dim X = \mathrm{tr.deg}_{\mathbb{C}}(K).
        \]

        Next, we relate the differentials on $R$ to those on $K$ via localization. We have the canonical isomorphism \cite[Chapter 2, Proposition 8.2A.]{hartshorne1977algebraic}:
        \[
            \Omega_{K/\mathbb{C}} \cong \Omega_{R/\mathbb{C}} \otimes_R K.
        \]
        The hypothesis states that $\Omega_{R/\mathbb{C}}$ is a \textbf{free $R$-module} of rank $n$. A key property of free modules is that their rank is preserved under base change to the fraction field. Therefore:
        \[
            \dim_K (\Omega_{R/\mathbb{C}} \otimes_R K) = \mathrm{rank}_R (\Omega_{R/\mathbb{C}}) = n.
        \]

        Combining these equalities, we conclude:
        \[
            n = \dim_K (\Omega_{R/\mathbb{C}} \otimes_R K) = \dim_K (\Omega_{K/\mathbb{C}}) = \mathrm{tr.deg}_{\mathbb{C}}(K) = \dim X.
        \]

    \end{proof}

\end{definition}

If $X$ is smooth, a tuple $(x_1,\dots,x_n)$ with $x_i\in R$ is called an \textbf{algebraic/étale coordinate system} if
\[
    \Omega_{R/\C} \cong \bigoplus_{i=1}^n R\,\mathrm d x_i.
\]
Equivalently, the dual basis gives
\[
    \T_X \cong \bigoplus_{i=1}^n R\,\partial_{x_i},
\]
where $\partial_{x_i}$ are the vector fields dual to $\mathrm d x_i$.

\begin{example}[Polynomial ring]
    Let $R=\C[x_1,\dots,x_n]$. Then
    \[
        \Omega_{R/\C} \cong \bigoplus_{i=1}^n R\,\mathrm d x_i,
    \]
    so $X=\Spec R\simeq \A^n_{\C}$ is smooth and $(x_1,\dots,x_n)$ is an algebraic coordinate system.
\end{example}

From now on assume $X=\Spec R$ is smooth affine and $(x_1,\dots,x_n)$ are algebraic coordinates.
The symmetric algebra of vector fields is
\[
    \Sym^{\bullet}\T_X = \bigoplus_{m\ge 0}\Sym^{m}\T_X \simeq R[\xi_1,\dots,\xi_n],
\]
canonically an $R$-algebra (identify $\xi_i$ with the class of $\partial_{x_i}$). Define the
cotangent bundle
\[
    T^{\ast}X := \Spec\, \Sym^{\bullet}\T_X \xrightarrow{\quad\pi\quad} X.
\]
and $\pi$ is induced by the $R$-algebra map $R = \Sym^{0}\T_X \subset \bigoplus_{m\ge 0}\Sym^{m}\T_X$.


\section{$\D$-modules on smooth affine varieties}
Let $\End_{\C}(R)$ denote the $\C$-algebra of all $\C$-linear maps $R\to R$. There are natural
embeddings
\[
    R\hookrightarrow \End_{\C}(R),\quad f\longmapsto (r\mapsto fr),
\]
and
\[
    \T_X\hookrightarrow \End_{\C}(R),\quad v\longmapsto v':=[f\mapsto v(\mathrm d f)],
    \begin{tikzcd}[row sep=3em, column sep=1.5em]
        R \arrow[rr, "d"] \arrow[dr, "v'"]  % v' 在箭头上方（默认）
        & & \Omega_R \arrow[dl, "v"]       % v 在箭头下方（不使用 ' 换边）
        \\
        & R &
    \end{tikzcd}.
\]

\begin{definition}[Ring of differential operators]
    The ring $\D_X$ of algebraic differential operators on $X$ is the $\C$-subalgebra of
    $\End_{\C}(R)$ generated by $R$ and $\T_X$.
\end{definition}

\begin{example}[Weyl algebra]
    If $X=\A^n_{\C}=\Spec \C[x_1,\dots,x_n]$, then
    \[
        \D_X \cong \C[x_1,\dots,x_n]\langle \partial_{x_1},\dots,\partial_{x_n}\rangle,
    \]
    with relations
    \[
        [\partial_{x_i},x_j]=\delta_{ij},\quad [x_i,x_j]=0,\quad [\partial_{x_i},\partial_{x_j}]=0.
    \]
\end{example}

Thus $\D_X$ is non-commutative. Every operator $P\in\D_X$ can be written in multi-index notation as
\[
    P=\sum_{\underline{\alpha}\in\mathbb N^n} f_{\underline{\alpha}}\,\partial^{\underline{\alpha}},\qquad
    \partial^{\underline{\alpha}}:=\partial_{x_1}^{\alpha_1}\cdots\partial_{x_n}^{\alpha_n},\quad f_{\underline{\alpha}}\in R.
\]

\paragraph{Basic modules.}\mbox{}\par

\noindent 1.  The coordinate ring $R$ is a left $\D_X$-module where$\partial_{x_i}\cdot r = \partial_{x_i}(\mathrm d r)\in R$.

\noindent Equivalently, writing $\mathrm dr=\sum_i (\partial_{x_i}r)\,\mathrm d x_i$, we have $\partial_{x_i}r$ the
$i$-th coefficient.\\

\par\noindent 2.  The top exterior power
\[
    \omega_X := \bigwedge{}^{n}\Omega_{R/\C} \cong R\,\mathrm d x_1\wedge\cdots\wedge\mathrm d x_n
\]
is the \textbf{dualizing module} of $R$. It carries a natural right $\D_X$-module structure via the transpose
anti-involution $t\colon \D_X\to\D_X$ determined by
\[
    \prescript{t}{}{x_i}=x_i,\qquad \prescript{t}{}{\partial_{x_i}}=-\partial_{x_i}.
\]
If $P=\sum_{\underline{\alpha}} f_{\underline{\alpha}}\,\partial^{\underline{\alpha}}$, then
\[
    \prescript{t}{}{P}=\sum_{\underline{\alpha}} (-\partial)^{\underline{\alpha}} f_{\underline{\alpha}},
\]
and the action is \cite[Lemma~1.2.6.]{hotta2008d}
\[
    (r\,\mathrm d x_1\wedge\cdots\wedge\mathrm d x_n)\cdot P := \left(\prescript{t}{}{P}\cdot r\right)\,\mathrm d x_1\wedge\cdots\wedge\mathrm d x_n.
\]

\subsection*{Side change and opposite ring}
Let $A$ be a ring and let $A^{\mathrm{op}}$ denote the opposite ring. i.e. $A^{\op} = \{ a^{\op} \mid a \in A \}$, $a^{\op} \cdot b^{\op} = (ba)^{\op}$. Then right $A$-modules are the
same as left $A^{\mathrm{op}}$-modules.

\begin{remark}
    For an alternative treatment of the same thing below, one can see \cite[Section~3.6.]{mustata2023dmodules}, which is of \cite{hotta2008d}'s style and keeps its notation.

    The editor of this note belives that constructions in algebra only make sense when it is defined globally, i.e., with a coordinate-free style or using a universal property.
\end{remark}

\begin{lemma}\label{lem:DXop}
    There is a natural isomorphism of rings
    \[
        \D_X^{\mathrm{op}} \simeq \omega_X \otimes_R \D_X \otimes_R \omega_X^{\!*},\qquad
        \text{where } \omega_X^{\!*}:=\Hom_R(\omega_X,R) \cong R\,(\mathrm d x_1\wedge\cdots\wedge\mathrm d x_n)^{-1}.
    \]
\end{lemma}
\begin{proof}
    Define
    \[
        \Phi(P^{\mathrm{op}}):=\mathrm d x_1\wedge\cdots\wedge\mathrm d x_n\;\otimes\; \prescript{t}{}{P}\;\otimes\;
        (\mathrm d x_1\wedge\cdots\wedge\mathrm d x_n)^{-1}.
    \]
    Then for $P,Q\in\D_X$ we have
    \begin{align*}
        \Phi(P^{\mathrm{op}}Q^{\mathrm{op}})
         & = \Phi((QP)^{\mathrm{op}})                                                            \\
         & = \mathrm d x\;\otimes \prescript{t}{}{(QP)}\otimes (\mathrm d x)^{-1}                \\
         & = \mathrm d x\;\otimes \prescript{t}{}{P}\prescript{t}{}{Q}\otimes (\mathrm d x)^{-1} \\
         & =: \left(\mathrm d x\;\otimes \prescript{t}{}{P}\otimes (\mathrm d x)^{-1}\right)
        \cdot \left(\mathrm d x\;\otimes \prescript{t}{}{Q}\otimes (\mathrm d x)^{-1}\right)     \\
         & = \Phi(P^{\mathrm{op}})\,\Phi(Q^{\mathrm{op}}),
    \end{align*}

    so $\Phi$ is a ring homomorphism.
\end{proof}

\begin{remark}
    $\D_X$ is a $(R,R)$-bimodule in the natural way. One should note that the left and the right $R$-module actions on $\D_X$ are NOT the same.
\end{remark}

\begin{remark}
    There is a global version of \ref{lem:DXop}, see the discussion lying above\cite[Lemma~1.2.7.]{hotta2008d}.
\end{remark}

Write $\Mod(\D_X^{\mathrm{op}})$ for the category of right $\D_X$-modules and $\Mod(\D_X)$ for the
category of left $\D_X$-modules.

\begin{theorem}\label{thm:left-right-equivalence}
    Tensoring with the canonical bundle yields an equivalence of categories
    \[
        \Mod(\D_X) \xrightarrow{\quad\sim\quad} \Mod(\D_X^{\mathrm{op}}),\qquad M\longmapsto \omega_X\otimes_R M,
    \]
    with quasi-inverse $N\longmapsto \omega_X^{\!*}\otimes_R N = \Hom_R(\omega_X,N)$.

    Here for $N \in \Mod(\D_X)$ and $M', N' \in \Mod(\D_X^{\mathrm{op}})$,
    we define their left/right $\D_X$-module structures as follows,
    for $\theta \in \mathcal{T}_X$, see \cite[Proposition~1.2.9]{hotta2008d} or \cite[Exercise~8.12,~8.13]{SabbahSchnell_MHMChap8_TrainingDModules}:
    \begin{enumerate}
        \item $M' \otimes_{\mathcal{O}_X} N \in \Mod(\D_X^{\mathrm{op}})$;
              \quad $(s' \otimes t)\theta = s'\theta \otimes t - s' \otimes \theta t$,
        \item $\Hom_{\mathcal{O}_X}(M', N') \in \Mod(\D_X)$;
              \quad $(\theta\psi)(s) = -\psi(s)\theta + \psi(s\theta)$.
    \end{enumerate}
\end{theorem}

\begin{proof}
    % 第一部分
    \noindent 1. Let $M \in \Mod(\Dx)$.
    \begin{align*}
        \ox \otimes_R \Dx \otimes_R \oxinv \cdot (\ox \otimes_R M)
         & = \ox \otimes_R (\Dx \cdot M) \\
         & \subseteq \ox \otimes_R M
    \end{align*}
    \[
        \implies \ox \otimes_R M \in \Mod(\Dx^{\mathrm{op}})
    \]

    % 第二部分
    \noindent 2. Let $N \in \Mod(\Dx^{\mathrm{op}}) = \Mod(\ox \otimes_R \Dx \otimes_R \oxstar)$.
    \begin{align*}
        \Dx \cdot (\oxstar \otimes_R N)
         & = (\oxstar \otimes_R \ox \otimes_R \Dx \otimes_R \oxstar) \cdot N \\
         & = \oxstar \otimes_R (\ox \otimes_R \Dx \otimes_R \oxstar \cdot N) \\
         & \subseteq \oxstar \otimes_R N
    \end{align*}
    \[
        \implies \oxstar \otimes_R N \in \Mod(\Dx)
    \]
\end{proof}

In particular, we will henceforth only consider left $\D_X$-modules, written simply as $\D_X$-modules.

\begin{remark}
    As a $R$-module, $\Hom_{R}(M', N') = M'^{\vee} \otimes_R N'$.
    But one should not expect $\D_X$-structure from this tensor product presentation
    \cite[Caveat.~8.24]{SabbahSchnell_MHMChap8_TrainingDModules}.
\end{remark}

\begin{remark}
    We also should give a note that why two definitions of right $\D_X$-module on $\omega_X \otimes_R N$ are equivalent.

    First action via $\Dx^{\mathrm{op}} \xlongrightarrow{\sim} \ox \otimes_R \Dx \otimes_R \oxinv$ \ref{lem:DXop} on $\omega_X \otimes_R N$ means:
    \[
        (\mathrm{vol} \otimes n) \cdot P := \mathrm{vol} \otimes (\prescript{t}{}{P} \cdot n)
    \]
    Where $\mathrm{vol} := \mathrm{d}x_1 \wedge \cdots \wedge \mathrm{d}x_n \in \omega_X$, and $P \in \Dx$.

    Second action by \cite[Proposition~1.2.9]{hotta2008d} on $\omega_X \otimes_R N$ means:
    \begin{align*}
        (\mathrm{vol} \otimes n) \cdot \theta & := \mathrm{vol}\cdot \theta \otimes n - \mathrm{vol} \otimes \theta\cdot n                                \\
                                              & = (\prescript{t}{}{\theta} \cdot 1)\mathrm{vol} \otimes n - \mathrm{vol} \otimes (\theta\cdot n)          \\
                                              & = 0 \mathrm{vol} \otimes n - \mathrm{vol} \otimes (\theta\cdot n) = -\mathrm{vol} \otimes (\theta\cdot n)
    \end{align*}
    Where $\theta \in \mathcal{T}_X \subseteq \Dx$.

    \bigskip

    For the two left $\D_X$-module structures on $\omega_X^{\!*} \otimes_R N'$:

    First action via $\Dx^{\mathrm{op}} \xlongrightarrow{\sim} \ox \otimes_R \Dx \otimes_R \oxinv$ \ref{lem:DXop} on $\omega_X^{\!*} \otimes_R N'$ means:
    \begin{align*}
        P \cdot (\mathrm{vol}^{-1} \otimes n') & := \mathrm{vol}^{-1} \otimes ((\mathrm{vol} \otimes P \otimes \mathrm{vol}^{-1}) \cdot n')                  \\
                                               & = \mathrm{vol}^{-1} \otimes ((\mathrm{vol} \otimes \prescript{tt}{}{P} \otimes \mathrm{vol}^{-1}) \cdot n') \\
                                               & = \mathrm{vol}^{-1} \otimes (n' \cdot \prescript{t}{}{P})
    \end{align*}
    Where $\mathrm{vol}^{-1} := (\mathrm{d}x_1 \wedge \cdots \wedge \mathrm{d}x_n)^{-1} \in \omega_X^{\!*}$, and $P \in \Dx$, and $\psi = \mathrm{vol}^{-1} \otimes n'$ is the map $\mathrm{vol} \mapsto n'$.

    Second action by \cite[Proposition~1.2.9]{hotta2008d} on $\omega_X^{\!*} \otimes_R N' \cong \Hom_R(\omega_X, N')$ means:
    \begin{align*}
        (\theta \cdot \psi)(\mathrm{vol}) & := -\psi(\mathrm{vol})\cdot \theta + \psi(\mathrm{vol} \cdot \theta)     \\
                                          & = -n' \cdot \theta + \psi((\prescript{t}{}{\theta} \cdot 1)\mathrm{vol}) \\
                                          & = -n' \cdot \theta + 0 = -n' \cdot \theta
    \end{align*}
    which corresponds to $\theta \cdot (\mathrm{vol}^{-1} \otimes n') = -\mathrm{vol}^{-1} \otimes (n' \cdot \theta)$.
    Where $\theta \in \mathcal{T}_X \subseteq \Dx$.

    These two actions in both cases are identical, it suffices to check it for $P \in \mathcal{T}_X \subseteq \Dx$.

\end{remark}

% Example: side-changing transfer bimodules via the Hom-description

\begin{example}[Side-changing transfer bimodules]\label{example:side-changing-transfer-bimodules}
    Let $f:X\to Y$ be a morphism of smooth (analytic or algebraic) manifolds/varieties over $\C$.
    Write $\mathcal O_X,\mathcal O_Y$ for the structure sheaves and
    $\mathcal D_X,\mathcal D_Y$ for the sheaves of differential operators.

    \medskip
    \noindent\textbf{Step 1 (the transfer bimodule $\mathcal D_{X\to Y}$).}
    Define
    \[
        \mathcal D_{X\to Y}
        :=\mathcal O_X\otimes_{f^{-1}\mathcal O_Y} f^{-1}\mathcal D_Y,
    \]
    which has a $(\mathcal D_X,f^{-1}\mathcal D_Y)$--bimodule structure; in particular
    the \emph{right} $f^{-1}\mathcal D_Y$--action is induced from right multiplication on the second
    tensor factor:
    \[
        (a\otimes P)\cdot Q := a\otimes (P Q),
        \qquad a\in\mathcal O_X,\ P,Q\in f^{-1}\mathcal D_Y.
    \]
    And the \emph{left} $\mathcal D_X$--action is defined via the map $\mathcal{T}_X \to f^*\mathcal{T}_Y = \mathcal{O}_X \otimes_{f^{-1}\mathcal{O}_Y} f^{-1}\mathcal{T}_Y$ ($\theta \mapsto \widetilde{\theta}$)
    which is dual to $\Omega_{Y} \otimes_{f^{-1}\mathcal O_Y} f^{-1}\mathcal O_X \to \Omega_X$, see Lem.~\ref{lem:second_fundamental_exact_sequence} for a local version.
    \[
        \theta \cdot (a \otimes P) := \theta(a) \otimes P + a \widetilde{\theta} P
    \]
    Coordinate-wise, the map is given by $\theta \mapsto \sum_j \theta(f^* y_j) \otimes \partial_{y_j}$, with $f^*y_j$ denoting the image of $y_j$ under $f^{-1}\mathcal{O}_Y \to \mathcal{O}_X$. Explicitly:
    \begin{align*}
        a \cdot (b \otimes P)      & := (ab) \otimes P,                                                           \\
        \theta \cdot (a \otimes P) & := \theta(a) \otimes P + a \sum_j \theta(f^* y_j) \otimes (\partial_{y_j} P)
    \end{align*}
    (See \cite[Section~1.3]{hotta2008d}.)

    \medskip

    \noindent\textbf{Step 2 (side-changing on the $X$-side: build a right $\mathcal D_X$-module).}
    Let $\omega_X$ be the canonical bundle. It is a natural \emph{right} $\mathcal D_X$--module
    via Lie derivative ($\Lie_\xi(\eta)$ for $\xi\in\mathcal{T}_X$, we have talked it in dualizing module with a coordinate system).
    The coordinate-free definition is given by
    \[
        (\Lie_\xi \eta)(\theta_1, \dots, \theta_n) := \xi(\eta(\theta_1, \dots, \theta_n)) - \sum_{i=1}^n \eta(\theta_1, \dots, [\xi, \theta_i], \dots, \theta_n)
    \]
    where $\theta_i \in \mathcal{T}_X$.

    For any left $\mathcal D_X$--module $M$, the tensor product
    $\omega_X\otimes_{\mathcal O_X}M$ becomes a \emph{right}
    $\mathcal D_X$--module by the standard tensor rule:
    for $\xi\in\mathcal{T}_X\subset \mathcal D_X$,
    \[
        (\eta\otimes m)\cdot \xi
        := (\Lie_\xi \eta)\otimes m \;-\; \eta\otimes(\xi\cdot m).
    \]

    Apply this with $M=\mathcal D_{X\to Y}$ and set
    \[
        B := \omega_X\otimes_{\mathcal O_X}\mathcal D_{X\to Y}.
    \]
    Then $B$ has:
    \begin{itemize}
        \item a \emph{right} $\mathcal D_X$--action given on generators by the tensor rule above;
        \item a \emph{right} $f^{-1}\mathcal D_Y$--action induced from $\mathcal D_{X\to Y}$:
              \[
                  (\eta\otimes P)\cdot Q := \eta\otimes(P\cdot Q),
                  \qquad Q\in f^{-1}\mathcal D_Y,
              \]
    \end{itemize}

    \medskip
    \noindent\textbf{Step 3 (side-changing on the $Y$-side via $\mathcal Hom$).}
    Define the opposite transfer bimodule by the Hom model \cite[Definition~8.6.4]{SabbahSchnell_MHMChap8_TrainingDModules}
    \[
        \mathcal D_{Y\leftarrow X}
        :=\mathcal Hom_{f^{-1}\mathcal O_Y}\!\Bigl(f^{-1}\omega_Y,\; B\Bigr).
    \]
    We now describe its $(f^{-1}\mathcal D_Y,\mathcal D_X)$--bimodule structure.

    \smallskip
    \emph{(a) Left $f^{-1}\mathcal D_Y$--action.}
    The sheaf $\omega_Y$ is a right $\mathcal D_Y$--module via Lie derivative, hence
    $f^{-1}\omega_Y$ is a right $f^{-1}\mathcal D_Y$--module.
    Moreover, $B$ is a right $f^{-1}\mathcal D_Y$--module (Step 2).
    Therefore $\mathcal Hom_{f^{-1}\mathcal O_Y}(f^{-1}\omega_Y,B)$ carries a natural
    \emph{left} $f^{-1}\mathcal D_Y$--action defined on generators as follows:

    \begin{itemize}
        \item for $g\in f^{-1}\mathcal O_Y$ and $\varphi\in \mathcal D_{Y\leftarrow X}$,
              \[
                  (g\cdot \varphi)(\alpha) := g\cdot \varphi(\alpha);
              \]
        \item for $\theta\in f^{-1}\mathcal{T}_Y\subset f^{-1}\mathcal D_Y$ and $\varphi\in \mathcal D_{Y\leftarrow X}$,
              \[
                  (\theta\cdot \varphi)(\alpha)
                  := \varphi(\alpha)\cdot \theta \;+\; \varphi(\Lie_\theta \alpha),
                  \qquad \alpha\in f^{-1}\omega_Y,
              \]
              where $\varphi(\alpha)\cdot\theta$ uses the \emph{right} $f^{-1}\mathcal D_Y$--action on $B$.
    \end{itemize}

    \smallskip
    \emph{(b) Right $\mathcal D_X$--action.}
    Since $B$ is a right $\mathcal D_X$--module (Step 2), define for $P\in\mathcal D_X$
    \[
        (\varphi\cdot P)(\alpha) := \varphi(\alpha)\cdot P,
        \qquad \alpha\in f^{-1}\omega_Y.
    \]
    This makes $\mathcal D_{Y\leftarrow X}$ a right $\mathcal D_X$--module, and the left
    $f^{-1}\mathcal D_Y$--action commutes with it because the two right actions on $B$
    ($f^{-1}\mathcal D_Y$ and $\mathcal D_X$) commute.

    \medskip
    \noindent\textbf{Step 4 (comparison with the tensor model).}
    Since $\omega_Y$ is locally free of rank 1, there is a canonical identification
    \[
        \mathcal Hom_{f^{-1}\mathcal O_Y}(f^{-1}\omega_Y,B)
        \ \cong\ B\otimes_{f^{-1}\mathcal O_Y} f^{-1}\omega_Y^{-1},
    \]
    so that the Hom model recovers the usual tensor expression \cite[Definition~1.3.3]{hotta2008d}
    \[
        \mathcal D_{Y\leftarrow X}
        \ \cong\ \omega_X\otimes_{\mathcal O_X}\mathcal D_{X\to Y}
        \otimes_{f^{-1}\mathcal O_Y} f^{-1}\omega_Y^{-1}
    \]

    \noindent\textbf{Remark (example of a closed embedding).}
    \begin{enumerate}
        \item \textbf{Transfer module $\D_{X\to Y}$:}
              Assume that $i : X \to Y$ is a closed embedding of smooth algebraic (affine) varieties
              (or complex manifolds) defined locally by $y_{r+1} = \cdots = y_n = 0$

              So every step of calculation there is required to be valid on a \textbf{ringed space}, in both cases (algebraic and complex manifolds), we use the result that
              $\D$-modules are \textbf{locally generated by sections} \cite[\href{https://stacks.math.columbia.edu/tag/01AL}{Tag 01AL}, \href{https://stacks.math.columbia.edu/tag/01AN}{Tag 01AN}, \href{https://stacks.math.columbia.edu/tag/01B2}{Tag 01B2}]{stacks-project} \cite[Section~4.1]{hotta2008d}:
              \[
                  \D_X = \mathcal{O}_X [\partial_{x_1}, \dots, \partial_{x_r}],
                  \qquad
                  \D_Y = \mathcal{O}_Y [\partial_{y_1}, \dots, \partial_{y_n}].
              \]
              Since $X$ and $Y$ are affine.

              Firstly, we forget that $i$ is a closed embedding, then $f^{-1}\D_Y = f^{-1}\mathcal{O}_Y [\partial_{y_{1}}, \dots, \partial_{y_n}]$ is a locally free $f^{-1}\mathcal{O}_Y$ module \cite[\href{https://stacks.math.columbia.edu/tag/01AJ}{Tag 01AJ}]{stacks-project}.
              After tensoring with $\mathcal{O}_X$, we get $\mathcal{D}_{X\to Y} = \mathcal{O}_X [\partial_{y_{1}}, \dots, \partial_{y_n}]$ locally free over $\mathcal{O}_X$.

              Recall that the left $\D_X$--action is given by the general formula (with $f=i$), given $\theta \in \T_X$, $a \in \mathcal{O}_X$, $P \in f^{-1}\D_Y$:
              \[
                  \theta \cdot (a \otimes P) := \theta(a) \otimes P + a \sum_{j=1}^n \theta(i^* y_j) \otimes (\partial_{y_j} P).
              \]
              In local coordinates, since $y_{r+1} = \cdots = y_n = 0$ on $X$, we have $i^* y_j = 0$ for $j > r$.
              For $j \le r$, we can identify $x_j = i^* y_j$. If $\theta = \partial_{x_k}$ ($1 \le k \le r$), then $\theta(i^* y_j) = \delta_{jk}$.
              Thus:
              \[
                  \partial_{x_k} \cdot (1 \otimes P) = 1 \otimes (\partial_{y_k} P).
              \]

              Secondly, we investigate its right $f^{-1}\D_Y$-module structure, we see the (global) section:
              \[
                  1\otimes 1\in \mathcal{O}_X \otimes_{f^{-1}\mathcal{O}_Y} f^{-1}\mathcal{D}_Y
              \]
              generate $\mathcal{O}_X \otimes_{f^{-1}\mathcal{O}_Y} f^{-1}\mathcal{D}_Y$ over $f^{-1}\mathcal{D}_Y$ (i.e. induce a surjective morphism between sheaves), since for a closed embedding,
              $f^{-1}\mathcal{O}_Y \to \mathcal{O}_X$ is surjective \cite[\href{https://stacks.math.columbia.edu/tag/01HK}{Tag 01HK}, \href{https://stacks.math.columbia.edu/tag/01B2}{Tag 01B2}]{stacks-project}.
              Since $X$ is (locally) defined by (sections) $y_{r+1} = \cdots = y_n = 0$ on $Y$, we have
              \[
                  \mathcal O_X \otimes_{f^{-1}\mathcal O_Y} f^{-1}\mathcal D_Y \simeq f^{-1}\mathcal D_Y \Big/ \sum_{j=r+1}^n y_j f^{-1}\mathcal D_Y
              \]
              as a right $f^{-1}\mathcal D_Y$-module.

              Thirdly, we investigate the left $\D_X$-module structure. We see the action of $\D_X$ (global) sections:
              \[
                  1\otimes 1 \in \mathcal{O}_X \otimes_{f^{-1}\mathcal{O}_Y} f^{-1}\mathcal{D}_Y,
              \]
              induce an injective morphism between sheaves $\D_X \to \mathcal{D}_{X \to Y}$. Since $\D_X = \mathcal{O}_X [\partial_{x_1}, \dots, \partial_{x_r}]$, and
              \[
                  \partial_{x_k} \cdot (1 \otimes P) = 1 \otimes (\partial_{y_k} P),
              \]
              We have an isomorphism of left $\D_X$-modules:
              \[
                  \mathcal{D}_{X \to Y} \simeq \D_X \otimes_{\C} \C[\partial_{y_{r+1}}, \dots, \partial_{y_n}].
              \]

        \item \textbf{Transfer module $\D_{Y\leftarrow X}$:}
              If we take a trivialization of
              \[
                  \text{d}x_1 \wedge \cdots \wedge \text{d}x_r \in \ox
              \]
              and
              \[
                  (\text{d}y_1 \wedge \cdots \wedge \text{d}y_n)^{-1} \in f^{-1}(\omega_Y^{-1}),
              \]
              then we can have a trivialization:
              \begin{align*}
                  \omega_X \otimes_{\mathcal O_X} \D_{X \to Y} \otimes_{f^{-1}\mathcal O_Y} f^{-1}\omega_Y^{-1}
                   & \cong \mathcal{O}_X\cdot \text{d}x \otimes_{\mathcal{O}_X} \D_{X}[\partial_{y_{r+1}}, \dots, \partial_{y_n}] \otimes_{f^{-1}\mathcal{O}_Y} f^{-1}\mathcal{O}_Y\cdot (\text{d}y)^{-1} \\
                   & \cong \mathcal{D}_X[\partial_{y_{r+1}}, \dots, \partial_{y_n}]                                                                                                                       \\
                   & \cong \C[\partial_{y_{r+1}}, \dots, \partial_{y_n}] \otimes_{\C} \D_{X}
              \end{align*}
              Then the side change gives a right $\D_X$-module structure by, for $\theta \in \mathcal{T}_X \subseteq \D_X$:
              \begin{align*}
                  (1\otimes P)\cdot \theta & = \prescript{t}{}{\theta} \cdot (1\otimes P)  &  & \text{(by side change)}                                   \\
                                           & =  1\otimes (\prescript{t}{}{\theta} \cdot P) &  & \text{(by left $\D_X$-module structure of $\D_{X\to Y}$)}
              \end{align*}
              (The left $f^{-1}\D_Y$-module structure also needs to transpose) But this identification is not standard in the literature.

              \cite{hotta2008d} calculate this side change using a different identification \cite[Lemma~1.3.4]{hotta2008d}
              \[
                  \D_{Y\leftarrow X} \cong f^{-1} (\D_Y \otimes_{\mathcal O_Y} \omega_Y^{-1}) \otimes_{f^{-1}\mathcal O_Y} \mathcal O_X \otimes_{\mathcal O_X} \omega_X,
              \]
              which is not just plainly apply the side change functor.

              In this identification, the left $f^{-1}\D_Y$-module structure is given by left multiplication of $\D_Y$ on $\D_Y$, the right $\D_X$-module structure is given by 4 steps:
              \begin{enumerate}
                  \item Step 1: $\D_Y$ itself is a $(\D_Y, \D_Y)$-bimodule, its left multiplication and right multiplication commute as actions.
                        \begin{enumerate}
                            \item Left multiplication is used to give left $f^{-1}\D_Y$-module structure of $\D_{Y\leftarrow X}$.
                            \item Right multiplication is used to give right $\D_X$-module structure of $\D_{Y\leftarrow X}$.
                        \end{enumerate}
                  \item Step 2: $\D_Y \otimes_{\mathcal O_Y} \omega_Y^{-1}$ is a left $f^{-1}\D_Y$-module, we use right multiplication in step 1.
                  \item Step 3: $f^{-1} (\D_Y \otimes_{\mathcal O_Y} \omega_Y^{-1}) \otimes_{f^{-1}\mathcal O_Y} \otimes \mathcal O_X$ is a left $\D_X$-module, which is
                        \[
                            \theta \cdot (P \otimes a) :=  P \otimes \theta(a) +  (\widetilde{\theta} \cdot P) \otimes a
                        \]
                        Induced by $\mathcal{T}_X \to f^*\mathcal{T}_Y = \mathcal{O}_X \otimes_{f^{-1}\mathcal{O}_Y} f^{-1}\mathcal{T}_Y$,
                        we talked it in the begging of this Example~\ref{example:side-changing-transfer-bimodules}.
                  \item Step 4: $f^{-1} (\D_Y \otimes_{\mathcal O_Y} \omega_Y^{-1}) \otimes_{f^{-1}\mathcal O_Y} \otimes \mathcal O_X \otimes_{\mathcal O_X} \omega_X$ has right $\D_X$-module structure.
              \end{enumerate}
              The Step 4 gives a transpose (in local coordinates) once, but the Step 2 gives a transpose once as well, so they cancel out. \cite{hotta2008d} also gives:
              \[
                  \D_{Y\leftarrow X} \simeq \mathbb{C}[\partial_{y_{r+1}}, \dots, \partial_{y_n}] \otimes_{\mathbb{C}} \D_X.
              \]
              Since there is no transpose in this identification, the right $\D_X$-module structure is just plainly by right multiplication.
              \[
                  (P\otimes 1)\cdot \theta =  (P\theta)\otimes 1.
              \]
    \end{enumerate}

\end{example}
