\section{Two definitions of nearby cycles}\label{sect:two_definitions_of_nearby_cycles}

On $\mathbb{A}_{\mathbb{C}}^1$ with coordinate $t$, given any $\alpha \in \mathbb{C}$, we have a left indecomposable $\D_{{A}^1_\C}$ module
\[
    K^\alpha_m :=  \frac{\D_{\mathbb{A}^1_\C}}{\D_{\mathbb{A}^1_\C}\cdot(t\partial_t - \alpha)^m}
\]
First note that $t\in K^\alpha_m$ has \textbf{both} a left and a right inverse, i.e., action by multiplying $t$ induce a bijection.
So $K^\alpha_m$ not only a (left, see 2. in Lem.~\ref{lem:quasi-coherent-sheaf-and-localization-of-D-modules}) $\C[t]\subseteq \D_{\mathbb{A}^1_\C}$ module, but also a (left) $\C[t,t^{-1}]$-module that extending the $\C[t]$-module structure.

We take sections $e^\alpha_i := \overline{(t\partial_t - \alpha)^{m-1-i}}$ (equivalence class) then we have a presentation:
\[
    K^\alpha_m = \bigoplus_{i=0}^{m-1} \mathbb{C}[t, t^{-1}] e^\alpha_i
\]
with $\partial_t e^\alpha_i = \frac{1}{t}(\alpha e^\alpha_i + e^\alpha_{i-1})$ for all $i$ (note we have claimed that $t$ is invertible), where $e^\alpha_{-1} = 0$,
and the direct sum notation should be treated as \textbf{internal} direct sum.

Then when restricted to $\mathbb{A}_{\mathbb{C}}^1 \setminus \{0\}$, $K^\alpha_m\mid_{\mathbb{A}_{\mathbb{C}}^1 \setminus \{0\}}$
becomes a \textbf{finitely generated}, (locally) free $\mathcal{O}_{\mathbb{A}_{\mathbb{C}}^1 \setminus \{0\}}$-module, i.e., an \textbf{integrable connection} (see \hyperref[def:integrable-connection]{definition}).

Then the matrix (note $K^\alpha_m$ is not a vector space since $\C[t,t^{-1}]$ is not a field) of $t\partial_t$-action on $K_m^{\alpha}$ is $J_{\alpha,m}$, the $m\times m$ Jordan block with eigenvalue $\alpha$:
\[
    J_{\alpha,m} = \begin{bmatrix}
        \alpha & 1      &        &        \\
               & \alpha & \ddots &        \\
               &        & \ddots & 1      \\
               &        &        & \alpha
    \end{bmatrix}_{m \times m}
\]

\begin{remark}
    Here $e_\ell^{\alpha}$ can be understood as the formal symbol
    of the multivalued function
    \[
        t^{\alpha} \cdot \frac{(\log t)^\ell}{\ell!}.
    \]
    And $K^\alpha_{m}$ indicates about the section not flat,
    a \textbf{flat section} means the section is in the kernel of connection
    \[
        \nabla:=t\partial_t \in\End_\C(K^\alpha_m).
    \]

    Indeed in \cite[Ch.~5]{hotta2008d},
    they develop arguments or definitions such as ``moderate growth'' by multivalued functions,
    so it is not only as a heuristic symbol to introduce multivalued function.
\end{remark}

Then locally around a simply connected neighborhood (Euclidean topology) of $(\mathbb{A}_{\mathbb{C}}^1 \setminus \{0\})^{\mathrm{an}}$,
with simply connectedness, we obtain that $\mathrm{\log}\, t$ is a well-defined analytic function (so we should consider the analytification of the module $K^\alpha_m$).
Consider
$$
    v_{\ell} := e^{-J_{\alpha,m} \log t} \cdot e_{\ell}^{\alpha} \in (K^\alpha_m)^{\mathrm{an}}
    = \frac{\D^{\mathrm{an}}_{(\mathbb{A}^1_\C)^\mathrm{an}}}{\D^{\mathrm{an}}_{(\mathbb{A}^1_\C)^\mathrm{an}}\cdot(t\partial_t - \alpha)^m}
    = \bigoplus_{i=0}^{m-1} \mathcal{O}^{\mathrm{an}}_{(\mathbb{A}^1_\C)^\mathrm{an}}\left[\frac{1}{t}\right] e^\alpha_i,
$$
for $\ell = 0, \cdots, m-1$,
we then have, by the Leibniz rule,
\begin{align*}
    t\partial_t \circ \left(e^{-J_{\alpha,m} \log t} \cdot e_{\ell}^{\alpha}\right)
     & := t\partial_t \cdot e^{-J_{\alpha,m} \log t} \cdot e_{\ell}^{\alpha}                                                                                                                                                                                             \\
     & = \left[t\partial_t, e^{-J_{\alpha,m} \log t}\right] \cdot e_{\ell}^{\alpha} + e^{-J_{\alpha,m} \log t} \cdot t\partial_t \cdot e_{\ell}^{\alpha}                                                                                                                 \\
     & = t\partial_t\left(e^{-J_{\alpha,m} \log t}\right) \cdot e_{\ell}^{\alpha} + e^{-J_{\alpha,m} \log t} \cdot t\partial_t \cdot e_{\ell}^{\alpha}                              &  & \text{(see \cite[Prop~2.25(iii)]{mustata2023dmodules})}                         \\
     & = t \left( -J_{\alpha,m} \frac{1}{t} e^{-J_{\alpha,m} \log t} \right) \cdot e_{\ell}^{\alpha} + e^{-J_{\alpha,m} \log t} \cdot \left( J_{\alpha,m} e_{\ell}^{\alpha} \right)                                                                                      \\
     & = -J_{\alpha,m} \left( e^{-J_{\alpha,m} \log t} \cdot e_{\ell}^{\alpha} \right) + J_{\alpha,m} \left( e^{-J_{\alpha,m} \log t} \cdot e_{\ell}^{\alpha} \right)               &  & \text{($\C$ is in the \textbf{center} of $\D^{\an}_{(\mathbb{A}^1_\C)^{\an}}$)} \\
     & = 0
\end{align*}
we use $\cdot$ to denote the multiplication in $\D^{\an}_{(\mathbb{A}^1_\C)^{\an}}$, and we get $v_{\ell}$ are $t\partial_t$-flat.

\bigskip

We have a local system, recall the equivalence given in Prop.~\ref{prop:classification-of-local-system}:
$$
    L_m^{\lambda}=\left(\bigoplus \limits_{i=0}^{m-1}\mathbb{C}\cdot v_{i}, e^{-2\pi iJ_{\alpha,m}}\right)
$$
There are several things to be point out:
\begin{enumerate}
    \item Taking fiber (not stalk) at any given point $p\in(\mathbb{A}_{\mathbb{C}}^1 \setminus \{0\})^{\mathrm{an}}$, $\Ker(\nabla)$ becomes a finite ($m$) dimensional $\C$ vector space.
    \item On a simply connected Euclidean open set $p \in U \subset (\mathbb{A}_{\mathbb{C}}^1 \setminus \{0\})^{\mathrm{an}}$, we can give $\log\, t(p)$ a concreate $\C$-value.
    \item View $e^\alpha_i(p)$ as the canonical basis vector (the $i+1$-th entry is $1$, others are $0$)
          \[
              \begin{pmatrix}
                  0 \\ \vdots \\ 1 \\ \vdots \\ 0
              \end{pmatrix}\in\C^m,
          \]
          where the $1$ is at the $(i+1)$-th row. Then we identify
          \[
              v^\alpha_i(p) = e^{-J_{\alpha,m} \log t}(p) \cdot e^\alpha_i(p)
          \]
          with
          \[
              e^{-J_{\alpha,m} \log t}(p) \cdot \begin{pmatrix}
                  0 \\ \vdots \\ 1 \\ \vdots \\ 0
              \end{pmatrix} \in \C^m.
          \]
    \item So the monodromy action (of a generator of $\pi_1((\mathbb{A}_{\mathbb{C}}^1 \setminus \{0\})^{\mathrm{an}}, p)$) acts on $\C^m$ by left multiplication with $e^{-2\pi i J_{\alpha,m}}$.
\end{enumerate}
We denote $\lambda = e^{-2\pi i \alpha}\in \C$.

\bigskip

Since $K^\alpha_m\mid_{\mathbb{A}_{\mathbb{C}}^1 \setminus \{0\}}$ is an integrable connection on $\mathbb{A}_{\mathbb{C}}^1 \setminus \{0\}$ (note that $\C[t, t^{-1}]$ is also smooth finite type over $\C$), recall by Riemann-Hilbert correspondence Thm.~\ref{thm:basic-RH-corr}, we have \cite[Thm.~7.2.1]{hotta2008d} that in $\D^b(\underline{\C}_{X^\mathrm{an}})$,
$$
    \text{DR}\left((K_m^{\alpha}\mid_{\mathbb{A}_{\mathbb{C}}^1 \setminus \{0\}})^{\mathrm{an}}\right) \xrightarrow{\substack{\text{quasi-isomorphic}\\ \sim}}  L_m^{\lambda}.
$$
Since the direct system (in category of coherent (finitely generated) $\D_{\mathbb{A}_{\mathbb{C}}^1}$-modules)
\begin{equation*}
    \begin{array}{ccccccc}
        K_{m-1}^{\alpha} & \lhook\joinrel\xrightarrow{\quad} & K_m^{\alpha}                              & \lhook\joinrel\xrightarrow{\quad} & K_{m+1}^{\alpha}                            & \lhook\joinrel\xrightarrow{\quad} & \cdots \\[1.5ex]
        \overline P      & \xmapsto{\quad}                   & \overline{P \cdot (t\partial_t - \alpha)} & \xmapsto{\quad}                   & \overline{P \cdot (t\partial_t - \alpha)^2} & \xmapsto{\quad}                   & \cdots
    \end{array}
\end{equation*}
induces a direct system of local systems
(categorical equivalence is given by functors, so it also transports morphisms),
arrows should be viewed as $\pi_1((\mathbb{A}_{\mathbb{C}}^1 \setminus \{0\})^\mathrm{an}, p)$-\textbf{equivariant} $\C$-linear maps by Prop.~\ref{prop:classification-of-local-system}:
\begin{equation*}
    \begin{array}{ccccccc}
        L_{m-1}^{\lambda}=\bigoplus \limits_{i=0}^{m-2}\C\cdot v_i & \lhook\joinrel\xrightarrow{\quad} & L_m^{\lambda}=\bigoplus \limits_{i=0}^{m-1}\C\cdot v_i & \lhook\joinrel\xrightarrow{\quad} & L_{m+1}^{\lambda}=\bigoplus \limits_{i=0}^{m}\C\cdot v_i & \lhook\joinrel\xrightarrow{\quad} & \cdots \\[1.5ex]
        v_i                                                        & \xmapsto{\quad}                   & v_i                                                    & \xmapsto{\quad}                   & v_i                                                      & \xmapsto{\quad}                   & \cdots
    \end{array}
\end{equation*}
To see why the morphism are $\pi_1((\mathbb{A}_{\mathbb{C}}^1 \setminus \{0\})^\mathrm{an}, p)$-equivariant,
since $J_{\alpha, m}$ is an upper-triangular matrix, then $e^{-2\pi iJ_{\alpha,m}}$ is also upper-triangular, so each $L^\lambda_{i}\subseteq L^\lambda_{i+1}$ is a sub $\pi_1((\mathbb{A}_{\mathbb{C}}^1 \setminus \{0\})^\mathrm{an}, p)$-representation.

Let $X = \text{Spec } R$ be a smooth affine algebraic variety over $\mathbb{C}$, and $f \in R$,
a regular function, we have the following commutative diagram:
\begin{center}\phantomsection\label{commutative_diagram_of_hyper_surface}
    \begin{tikzcd}
        {(f=0)\atop \text{divisor}} & X && {\mathbb{A}^1_{\mathbb{C}}} \\
        \\
        & {U=X\backslash(f=0)} && {\mathbb{A}^1_{\mathbb{C}}\backslash \{0\}}
        \arrow["i", hook, from=1-1, to=1-2]
        \arrow["f", from=1-2, to=1-4]
        \arrow["j", hook', from=3-2, to=1-2]
        \arrow["f", from=3-2, to=3-4]
        \arrow[hook', from=3-4, to=1-4]
    \end{tikzcd}
\end{center}
where the morphism $f$ is given by the ring map:
\begin{align*}
    \C[t] & \xrightarrow{\quad} R                       \\
    \C    & \xrightarrow{\mathrm{id}_\C} \C \subseteq R \\
    t     & \xmapsto{\quad} f
\end{align*}
And also note that the locus of $f$ (as an \textbf{analytic set} in complex manifold $X^\mathrm{an}$) can be \textbf{non-smooth}.

We define
$$
    \Psi_{f,\lambda} = \lim_{\substack{\longrightarrow \\ m \to \infty}} i^{-1} Rj_\ast L_m^{\lambda}
$$
which is called the $\lambda\textbf{-nearby cycles}$ of $f$.
And $T$-action ($\pi_1((\mathbb{A}_{\mathbb{C}}^1 \setminus \{0\})^\mathrm{an})$-action) on $L_m^{\lambda}$ induces $T$-action on $\Psi_{f,\lambda}$.



\begin{remark}
    Firstly, note that since $\{f=0\}$ may not smooth, so $i^{-1}$ is the sheaf-theoretic pull-back between topological (ringed) spaces.
    Which is somewhat different from scheme case.

    Secondly, $j_{\ast}$ is the direct image functor,
    and it's a left exact functor,
    we write $Rj_\ast$ the right derived functor of $j_{\ast}$,
    which can be viewed as the complex of sheaves obtained by taking the \textbf{injective resolution} of $L_{m}^{\lambda}$ and applying $j_{\ast}$.
    Since $j_\ast$ maybe have nontrivial cohomology even in a non-$\D$-module-theoretic context, we choose to use a derived functor.
\end{remark}

The above definition is a topological definition, which is a bit differed from Deligne's. And total cycles is defined by combining all information from nearby cycles.


On the other hand, recall Example~\ref{Example 8.11}, we have
$$
    \mathcal{D}_{X}[s]f^{s} \subseteq R\left[\frac{1}{f}\right][s] \cdot f^{s}.
$$

For closed point $\alpha \in \mathbb{A}_{\mathbb{C}}^{1} \simeq \operatorname{Spec} \mathbb{C}[s]$, we use $\mathfrak{m}_{\alpha}$ to denote its maximal ideal. We have proved in Example \ref{Example 8.11}:

\begin{remark}
    We now justify the symbol $R_f[s]f^s$,
    since in Rmk.~\ref{remark:second_treatment_of_example_8.11} we only get the module $R_f[\partial_t]$,
    and $f^s$ represent the canonical section $1\otimes 1$.

    One should note that the notation $R_f[\partial_t]$ (or more precisely $\C[\partial_t]\otimes_\C R_f$) itself doesn't imply any module structure.
    If we want to make it an $R_f$ (left or right) module, we should induce the action from its left $\D_Y$-module structure.

    The Step 5 Block A of Rmk.~\ref{remark:second_treatment_of_example_8.11} gives us an answer.
    We have a projection
    \[
        p: Y = X\times_{\Spec \C}\mathbb{A}^1_{\C} \to X
    \]
    which makes $\Ox$ a subring of $\Oy$.
    So we get a left $R$ module structure by
    \[
        R=\Ox\xhookrightarrow{p^\sharp} \Oy \subseteq \D_Y.
    \]

    Since before we pushforward left $\D_X$-module $R_f$ along $\Gamma_f$,
    $f\in\D_X$ already acts bijectively (left and right) on $R_f$,
    which makes $t\in\D_Y$ act bijectively on $\Gamma_{f,+}(R_f) = \C[\partial_t] \otimes_{\C} R_f$ \cite[Lemma~2.1.5]{PopaDMBG2021}.
    Then by Step 5 Block A+D of Rmk.~\ref{remark:second_treatment_of_example_8.11},
    we discover $f\in R$ also act bijectively on $\Gamma_{f,+}(R_f)$.
    So we finally give a left $R_f$ module structure on $\C[\partial_t] \otimes_{\C} R_f$.

    Then we discover $s = -t\partial_t$ and $R$ commute in $\D_Y$, this is because we have
    \[
        [\partial_t, g] = \partial_t(g) = 0 \quad (\text{see  \cite[Prop~2.25(iii)]{mustata2023dmodules}})
    \]
    By universal property of polynomial ring, we have $R_f[s]$ embedded in $\D_Y$ (as a $\C$-subalgebra),
    if we map formal symbol $s$ to $-t\partial_t \in \D_Y$.

    \medskip

    Then we can act $R_f[s]$ onto $1\otimes 1 \in \Gamma_{f,+}(R_f)$, then we claim:
    \begin{enumerate}
        \item this action is faithful, so we get a free rank 1 $R_f[s]$ submodule in $\Gamma_{f,+}(R_f)$.
        \item This submodule is the whole of $\Gamma_{f,+}(R_f)$.
    \end{enumerate}
    We denote it as $R_f[s]f^s$.
    Then we can give the abstractly constructed $R_f[s]$ a left $\D_Y$ module structure by pulling back the one on $\Gamma_{f,+}(R_f)$.
\end{remark}

\begin{remark}\label{rmk:base_change_of_R_f[s]_f^s}
    We then try to define $R_f(s)f^s$ and $R_f[s]_{\frak{m}_\alpha}f^s$, the construction should satisfy:
    \begin{enumerate}
        \item From (the proof in) the corollary~\ref{cor:bernstein_beilinson} below, which suggests that this construction should be functorial.
        \item Be consistent with \cite[The $\D_{U_s}$ module $\mathcal{M}$ defined in Proof of Theorem 2.2.1.]{PopaDMBG2021}.
    \end{enumerate}

    This is equivalent to say that we have a ring ($\C$-algebra) map
    \[
        \D_Y \to \End_\C(R_f[s])
    \]
    And we have a composition
    \[
        \D_X\otimes_\C \C[s] \to \D_Y \to \End_\C(R_f[s])
    \]
    by assigning $s$ to $-t\partial_t\in\D_Y$

    \medskip

    We define $R_f(s) := R_f[s]\otimes_{\C[s]}\C(s)$ and $R_f[s]_{\frak{m}_\alpha} := R_f[s]\otimes_{\C[s]}\C[s]_{\frak{m}_\alpha}$.
    Before giving it a left $\D$ module structure, we discover that this construction is functorial (in the category of $\C[s]$-modules).

    \medskip

    Then we give it a left $\D$ module structure, this structure should also be functorial.
    \begin{enumerate}
        \item $R_f(s)$ should have a left $\D_{R(s)/\C(s)} = \D_{R/\C}\otimes_\C \C[s]\otimes_{\C[s]}\C(s)$ module structure, see Lem.~\ref{lem:base_change_of_D_modules}.
        \item $R_f[s]_{\frak{m}_\alpha}$ should have a left $\D_{R[s]_{\frak{m}_\alpha}/\C[s]_{\frak{m}_\alpha}} = \D_{R/\C}[s]\otimes_{\C[s]} \C[s]_{\frak{m}_\alpha}$ module structure, but one should note $\C[s]_{\frak{m}_\alpha}$ is not a field.
    \end{enumerate}
    Thanks to the construction in the proof in Lem.~\ref{lem:base_change_of_D_modules}, the base change of the \textbf{ring of \underline{relative} differential operator} is functorial.

    \medskip

    To give a (left) module structure is equivalent to give the ring map
    \begin{align*}
        \D_{R/\C}[s]\otimes_{\C[s]}\C(s) & \to \End_\C(R_f[s]\otimes_{\C[s]}\C(s))                    \\
        P \otimes a                      & \mapsto \left( u \otimes b \mapsto P(u) \otimes ab \right)
    \end{align*}
    and
    \begin{align*}
        \D_{R/\C}[s]\otimes_{\C[s]}\C[s]_{\frak{m}_\alpha} & \to \End_\C(R_f[s]\otimes_{\C[s]}\C[s]_{\frak{m}_\alpha})  \\
        P \otimes a                                        & \mapsto \left( u \otimes b \mapsto P(u) \otimes ab \right)
    \end{align*}
    From the two formulas above, one see this definition of ring action (on module) is functorial. Where $u\mapsto P(u)$ is just
    \[
        \D_{R/\C} \otimes_\C \C[s] = \D_X\otimes_\C \C[s] \to \D_Y \to \End_\C(R_f[s])
    \]
    mentioned above.
\end{remark}

\begin{theorem}\label{thm:existence_of_b_function}
    There exists $b(s) \in \mathbb{C}[s]$ s.t.
    \[
        b(s) \quad \mathrm{ kills } \quad \frac{\mathcal{D}_{X}[s]f^{s}}{\mathcal{D}_{X}[s]f^{s+1}}
    \]
    see Thm.~\ref{Theorem 8.4} or Ex.~\ref{Example 8.11} or Rmk.~\ref{remark:second_treatment_of_example_8.11}.
\end{theorem}



\begin{corollary}[Bernstein-Beilinson]\label{cor:bernstein_beilinson}
    Let $\mathbb{C}(s)$ be the fractional field of $\mathbb{C}[s]$, and $\mathcal{D}_{X}(s) = \mathcal{D}_{X} \otimes_{\mathbb{C}} \mathbb{C}(s)$; for all $\alpha \in \mathbb{A}_{\mathbb{C}}^{1}$ closed point, we obtain that:
    \begin{enumerate}
        \item $\mathcal{D}_{X}(s) f^{s+k} \simeq \mathcal{D}_{X}(s) f^{s}, $ for all $  k \in \mathbb{Z}$.
        \item There exists $k_{0} \geqslant 0$ s.t.
              \begin{itemize}
                  \item $\mathcal{D}_{X}[s]_{\mathfrak{m}_{\alpha}} f^{s-k} = \mathcal{D}_{X}[s]_{\mathfrak{m}_{\alpha}} f^{s-k_{0}}, $ for all $ k > k_{0}$
                  \item $\mathcal{D}_{X}[s]_{\mathfrak{m}_{\alpha}} f^{s+k} = \mathcal{D}_{X}[s]_{\mathfrak{m}_{\alpha}} f^{s+k_{0}},$ for all $  k > k_{0}$
              \end{itemize}
    \end{enumerate}

\end{corollary}

\begin{remark}\label{rmk:D_X(s)f^s=R_f(s)f^s_and_D_X[s]_m_alpha_f^s=R_f[s]_m_alpha_f^s}
    Noted that here  $\mathcal{D}_{X}(s) f^{s} = R[\frac{1}{f}](s) f^{s},\ \mathcal{D}_{X}[s]_{\mathfrak{m}_{\alpha}} f^{s} = R[\frac{1}{f}][s]_{m_{\alpha}} f^{s}$.
\end{remark}

\begin{proof}
    \mbox{}

    $1)$ Since $\mathbb{C}(s)$ can be view as the localization at $(0)$ of $\mathbb{C}[s]$,
    we have $\mathbb{C}(s)$ is flat over $\mathbb{C}[s]$,
    thus
    \[
        \frac{\mathcal{D}_{X}[s]f^{s}}{\mathcal{D}_{X}[s]f^{s+1}}\otimes \mathbb{C}(s)=\frac{\mathcal{D}_{X}(s)f^{s}}{\mathcal{D}_{X}(s)f^{s+1}}
    \]
    Note that
    \begin{enumerate}
        \item To commute tensor product with quotient, the ring map $\mathbb{C}[s]\to \mathbb{C}(s)$ is not needed to be flat, all tensor product is right exact.
        \item To make the incuced $\D_X(s)$-structure the same on LHS and RHS, the $\D_X(s)$-structure should be functorial, which is required in Rmk.~\ref{rmk:base_change_of_R_f[s]_f^s}.
    \end{enumerate}

    From the $\D_X(s)$-structure mentioned in Rmk.~\ref{rmk:base_change_of_R_f[s]_f^s}, we have
    \[
        b(s+k) \quad\mathrm{kills} \quad\frac{\mathcal{D}_{X}(s)f^{s}}{\mathcal{D}_{X}(s)f^{s+1}} \implies \D_{R/\C}[s]\otimes_{\C[s]} \C(s) \ni b(s+k)\otimes 1  \quad\mathrm{kills } \quad\frac{\mathcal{D}_{X}[s]f^{s}}{\mathcal{D}_{X}[s]f^{s+1}} \otimes_{\C[s]} \C(s)
    \]
    And $b(s+k) \neq 0$ is invertible in $\mathbb{C}(s)$, so
    \[
        b(s+k) \quad\mathrm{kills} \quad\frac{\mathcal{D}_{X}(s)f^{s}}{\mathcal{D}_{X}(s)f^{s+1}} \implies \frac{\mathcal{D}_X(s) f^{s+k}}{\mathcal{D}_X(s) f^{s+k-1}} = 0.
    \]

    $2)$  Note that Bernstein-Sato polynomial $b(s)\in\D_X[s]$ has coefficients in $\C$, see Thm.~\ref{Theorem 8.4}.

    Since the number of roots of $b(s)$ is finite,
    there exists $k_0 \ge 0$, $b(\alpha \pm k) \neq 0 \in\mathbb{C}$ for all $k > k_0$,
    then $b(s \pm k) \notin \mathfrak{m}_\alpha$ for all $k > k_0$.

    So $b(s \pm k)$ is invertible in $\mathbb{C}[s]_{\mathfrak{m}_\alpha}$ for all $k > k_0$,
    similarly, we get
    \[
        \frac{\mathcal{D}_X[s]_{\mathfrak{m}_\alpha} f^{s \pm k}}{\mathcal{D}_X[s]_{\mathfrak{m}_\alpha} f^{s \pm k + 1}} = 0.
    \]
\end{proof}

\begin{definition}
    For all $\alpha \in \mathbb{C}$:
    \begin{enumerate}
        \item $\mathcal{D}_X[s]_{\mathfrak{m}_\alpha} f^{s+k},  k \gg 0$ is stabled, called the $\textbf{minimal extension}$ of $R[\frac{1}{f}][s]_{\mathfrak{m}_\alpha} f^s$.
        \item $\Psi_{f,\alpha} := \frac{\mathcal{D}_X[s]_{\mathfrak{m}_\alpha} f^{s-k}}{\mathcal{D}_X[s]_{\mathfrak{m}_\alpha} f^{s+k}}$ is called the $\alpha\textbf{-nearby cycles}$ of $R$ along $f$.

              \cite[Definition~3.3.2.]{PopaDMBG2021} defines that $\alpha$-nearby cycles is a $\mathcal{D}_X$-module, NOT a $\mathcal{D}_X[s]$-module.
        \item $\Lambda := \mathbb{Z} + \{\text{roots of } b_{f}(s)\}$.
    \end{enumerate}
\end{definition}

\begin{remark}
    For one wonder why here we \emph{define} the minimal extension,
    see \cite[Theorem~2.10]{Wu2020ComparisonNearbyCyclesBfunctions}, which use
    \cite[Def~3.4.1]{hotta2008d} to define the minimal extension and claim the equivalence of the two.

    In $\mathcal{D}$-module theory,
    an extension is usually about given a module over an open subset $U$,
    and extending it to the whole space $X$ to get a $\mathcal{D}_{X}$-module,
    and we shall pay attention on how it extends around $f=0$.
    Take $U=\mathrm{Spec}\, R_{f}$,
    if the module over $U=X\backslash(f=0)$ can be viewed as $R$-module,
    then it's an extension,
    and we call $R_{f}[s]_{\mathfrak{m}_{\alpha}}f^{s}$  $\textbf{maximal extension}$.
\end{remark}

\begin{proposition}\label{prop:properties_of_alpha_nearby_cycles}
    \quad

    \begin{enumerate}
        \item $s - \alpha$ acts on $\Psi_{f,\alpha}$ nilpotently
        \item $\Psi_{f,\alpha} \cong \Psi_{f,\alpha+1}$ (Note: a $\D_X$-module homomorphism, not a $\D_X[s]$-module homomorphism)
        \item $\Psi_{f,\alpha} \neq 0$ if and only if $\alpha \in \Lambda$
        \item $\Psi_{f,\alpha}$ is a regular holonomic $D_X$-module.
    \end{enumerate}
\end{proposition}
\begin{proof}
    \mbox{}

    \begin{enumerate}
        \item The descending chain of submodules
              \[
                  \mathcal{D}_{X}[s]_{\mathfrak{m}_{\alpha}}f^{s+k_{0}}\subseteq\mathcal{D}_{X}[s]_{\mathfrak{m}_{\alpha}}f^{s+k_{0}-1}\subseteq\dots\subseteq \mathcal{D}_{X}[s]_{\mathfrak{m}_{\alpha}}f^{s-k_{0}}
              \]
              induces a finite filtration on the quotient module
              \[
                  \frac{\mathcal{D}_{X}[s]_{\mathfrak{m}_{\alpha}}f^{s-k_{0}}}{\mathcal{D}_{X}[s]_{\mathfrak{m}_{\alpha}}f^{s+k_{0}}},
              \]
              whose successive quotients are of the form
              \[
                  \frac{\mathcal{D}_{X}[s]_{\mathfrak{m}_{\alpha}}f^{s+i}}{\mathcal{D}_{X}[s]_{\mathfrak{m}_{\alpha}}f^{s+i+1}}.
              \]
              For each such quotient, there exists a polynomial $b_{i}(s)$ that annihilates it (before taking localization). Because $s-\alpha$ is the only non-invertible irreducible polynomial in the local ring $\mathbb{C}[s]_{\mathfrak{m}_{\alpha}}$, each $b_{i}(s)$ takes the form $u_{i}(s-\alpha)^{n_{i}}$ for some unit $u_{i} \in \mathbb{C}[s]_{\mathfrak{m}_{\alpha}}$.
              By taking $n = \sum n_{i}$, the element $(s-\alpha)^{n}$ annihilates the entire module $\frac{\mathcal{D}_{X}[s]_{\mathfrak{m}_{\alpha}}f^{s-k_{0}}}{\mathcal{D}_{X}[s]_{\mathfrak{m}_{\alpha}}f^{s+k_{0}}}$. Hence, $s-\alpha$ acts nilpotently on it.

        \item Recall that $t$ acts on $R_f[s]f^s$ bijectively. Consider the $\mathcal{D}_{X}$-linear automorphism of $\mathcal{D}_{X}(s)f^{s}$ defined by the shift
              \[
                  T_{-1}\big(P(s)f^{s}\big) = P(s-1)f^{s-1}.
              \]
              For any element $m$, we have
              \[
                  T_{-1}(s \cdot m) = (s-1) \cdot T_{-1}(m), \quad\text{and generally}\quad T_{-1}(P(s) \cdot m) = P(s-1) \cdot T_{-1}(m).
              \]
              This means $T_{-1}$ is not $\D_X[s]$-linear.

              When localizing at $\mathfrak{m}_{\alpha}$,
              a polynomial $g(s)$ becomes invertible if $g(\alpha) \neq 0$.
              Under the shift, the action of $g(s)$ becomes the action of $g(s-1)$, evaluated at $s = \alpha+1$ is also non-zero, making it invertible in $\mathbb{C}[s]_{\mathfrak{m}_{\alpha+1}}$.
              So we can safely define the map below:
              \begin{align*}
                  T_{-1}: \mathcal{D}_{X}[s]_{\mathfrak{m}_{\alpha}}f^{s-k} & \stackrel{\sim}{\longrightarrow} \mathcal{D}_{X}[s]_{\mathfrak{m}_{\alpha+1}}f^{s-k-1} \\
                  \frac{m}{g(s)}                                            & \longmapsto \frac{T_{-1}(m)}{g(s-1)}
              \end{align*}
              for any integer $k$. And one should verify that it is indeed an isomorphism.

              For $k \gg 0$, the sequence of modules stabilizes. Passing to the respective quotients, this induces the desired $\mathcal{D}_X$-module isomorphism $\Psi_{f,\alpha} \xrightarrow{\sim} \Psi_{f,\alpha+1}$.

        \item Since $\alpha \not\in\Lambda$, $\alpha$ is not a root of $b(s+k)$, then $b(s+k)$ is a unit for all $k$, thus $\Psi_{f,\alpha}=0$.

              If $\Psi_{f,\alpha}\neq 0$,
              there exists $i$ such that $\frac{\mathcal{D}_{X}[s]_{\mathfrak{m}_{\alpha}}f^{s+i}}{\mathcal{D}_{X}[s]_{\mathfrak{m}_{\alpha}}f^{s+i+1}}\neq 0$,
              and then $\alpha$ is a root of $b(s+i)$ such that $b(s+i)$ is not invertible in $\mathbb{C}[s]_{\mathfrak{m}_{\alpha}}$. Thus $\alpha\in\Lambda$.

        \item $M_{\alpha} = \mathbb{C}[t, \frac{1}{t}] \cdot t^{\alpha}$ (see Ex.~\ref{ex:first-examples-of-d-modules}) is regular
              (at least analytically at the stalk of $0$, see Rmk.~\ref{rmk:indecomposable_objects_in_RH_D} and Prop.~\ref{prop:classification-unipotent-indecomposable-regular-holonomic},
              to be verified algebraically it should be embedded into $\mathbb{P}^1_\C$, see Rmk.~\ref{rmk:algebraic_KK_regularity_holomonic})
              holonomic $\mathcal{D}_{\mathbb{A}^{1}_{\mathbb{C}}}$-module.

              Consider
              \begin{align*}
                  f: X   & \longrightarrow \mathbb{A}^{1}_{\mathbb{C}} \\
                  \alpha & \longmapsto t=f(\alpha)
              \end{align*}
              which is induced by
              \begin{align*}
                  \mathbb{C}[t] & \longrightarrow R \\
                  t             & \longmapsto f
              \end{align*}
              We admit the fact that the $\D$-module theoretical (derived) \textbf{inverse image} functor \cite[Sect.~1.5, It is the derived one that comptible with Thm.~6.1.5]{hotta2008d} preserves (algebraic we use here) regular holonomicity \cite[Thm.~6.1.5]{hotta2008d}, then
              \[
                  f^{*} M_{\alpha}:=M_{\alpha}\otimes_{\mathbb{C}[t]} R\cong R_{f}\cdot f^\alpha= R\left[\frac{1}{f}\right] f^{\alpha}
              \]
              makes $R_{f}\cdot f^\alpha$ a regular holonomic $\mathcal{D}_{X}$-module.

              With $(2)$, we can assume $\alpha\neq 0$, otherwise, we just shifts $\alpha$ to $\alpha+1$.
              We have a short exact sequence:
              \[
                  0 \longrightarrow \frac{R_f[s]_{\mathfrak{m}_{\alpha}} f^{s}}{(s-\alpha) R_f[s]_{\mathfrak{m}_{\alpha}} f^{s}} \xrightarrow{\;(s-\alpha)\cdot\;} \frac{R_f[s]_{\mathfrak{m}_{\alpha}} f^{s}}{(s-\alpha)^2 R_f[s]_{\mathfrak{m}_{\alpha}} f^{s}} \xrightarrow{\;\text{proj}\;} \frac{R_f[s]_{\mathfrak{m}_{\alpha}} f^{s}}{(s-\alpha) R_f[s]_{\mathfrak{m}_{\alpha}} f^{s}} \longrightarrow 0
              \]
              where the injective mapping is given by \textbf{left} multiplication by $(s-\alpha)$ (i.e., sending the residue class $\overline{m}$ to $\overline{(s-\alpha)m}$), and the surjective mapping is the natural canonical projection.

              Here, the isomorphism
              \[
                  \frac{R_f[s]_{\mathfrak{m}_{\alpha}} f^{s}}{(s-\alpha) R_f[s]_{\mathfrak{m}_{\alpha}} f^{s}} \cong R_{f} \cdot f^{\alpha}
              \]
              arises because quotienting out the $(s-\alpha)$-multiples algebraically corresponds to taking the evaluation map $s \mapsto \alpha$.
              \begin{enumerate}
                  \item The ring $\mathbb{C}[s]_{\mathfrak{m}_{\alpha}}$ is a local ring with (its unique) maximal ideal $(s-\alpha)$ reduces canonically to $\mathbb{C}$, collapsing the coefficients from $R_f[s]_{\mathfrak{m}_{\alpha}}$ down to $R_{f}$.
                  \item And the formal symbol $f^{s}$ becomes $f^{\alpha}$, this step didn't actually do evaulation, since they are \textbf{both} the canonical section $1\otimes 1$ in their respective presentations. So the real thing goes:
                        \[
                            1\otimes 1 \mapsto 1\otimes 1.
                        \]
              \end{enumerate}
              Thus, this quotient module is precisely the regular holonomic $\mathcal{D}_X$-module $R_{f} \cdot f^{\alpha}$.

              Since regular holonomic $\mathcal{D}_{X}$-modules form an abelian category \cite[Combine Prop.~3.1.2 and Def.~6.1.1]{hotta2008d}, we have $\frac{R_{f}[s]_{\mathfrak{m}_{\alpha}}f^{s}}{(s-\alpha)^{2}R_{f}[s]_{\mathfrak{m}_{\alpha}}f^{s}}$ is regular holonomic.

              And further, we obtain the generalized short exact sequence:
              \[
                  0 \longrightarrow \frac{R_f[s]_{\mathfrak{m}_{\alpha}} f^{s}}{(s-\alpha) R_f[s]_{\mathfrak{m}_{\alpha}} f^{s}} \xrightarrow{\;(s-\alpha)^N\cdot\;} \frac{R_f[s]_{\mathfrak{m}_{\alpha}} f^{s}}{(s-\alpha)^{N+1} R_f[s]_{\mathfrak{m}_{\alpha}} f^{s}} \xrightarrow{\;\text{proj}\;} \frac{R_f[s]_{\mathfrak{m}_{\alpha}} f^{s}}{(s-\alpha)^N R_f[s]_{\mathfrak{m}_{\alpha}} f^{s}} \longrightarrow 0
              \]
              By induction, $\frac{R_f[s]_{\mathfrak{m}_{\alpha}} f^{s}}{(s-\alpha)^{N} R_f[s]_{\mathfrak{m}_{\alpha}} f^{s}}$ is regular holonomic for all $N$.

              By $(1)$, the action of $(s-\alpha)$ on $\Psi_{f,\alpha} = \frac{\mathcal{D}_{X}[s]_{\mathfrak{m}_{\alpha}}f^{s-k}}{\mathcal{D}_{X}[s]_{\mathfrak{m}_{\alpha}}f^{s+k}}$ is nilpotent, meaning there exists some large enough integer $N \gg 0$ such that $(s-\alpha)^N$ entirely annihilates $\Psi_{f,\alpha}$.
              Because of this annihilation, the natural surjective $\mathcal{D}_X[s]_{\mathfrak{m}_{\alpha}}$-module morphism
              \[
                  \mathcal{D}_{X}[s]_{\mathfrak{m}_{\alpha}}f^{s-k} \xrightarrow{\;\text{proj}\;} \frac{\mathcal{D}_{X}[s]_{\mathfrak{m}_{\alpha}}f^{s-k}}{\mathcal{D}_{X}[s]_{\mathfrak{m}_{\alpha}}f^{s+k}} = \Psi_{f,\alpha}
              \]
              factors through the quotient by the submodule generated by $(s-\alpha)^N$. This explicitly means that the map descends to a well-defined surjective homomorphism:
              \[
                  \frac{\mathcal{D}_X[s]_{\mathfrak{m}_{\alpha}} f^{s-k}}{(s-\alpha)^N \mathcal{D}_X[s]_{\mathfrak{m}_{\alpha}} f^{s-k}} \twoheadrightarrow \Psi_{f,\alpha}.
              \]
              Since one can verify
              \begin{enumerate}
                  \item $\mathcal{D}_{X}[s]_{\mathfrak{m}_{\alpha}}f^{s-k} \cong \mathcal{D}_{X}[s]_{\mathfrak{m}_{\alpha}}f^{s}$, even in the case one is a proper subset of the other, since then the isomorphism will not be the inclusion.
                  \item Rmk.~\ref{rmk:D_X(s)f^s=R_f(s)f^s_and_D_X[s]_m_alpha_f^s=R_f[s]_m_alpha_f^s},
                        i.e.,
                        $\mathcal{D}_X[s]_{\mathfrak{m}_{\alpha}} f^{s}$ as a submodule $R_f[s]_{\mathfrak{m}_{\alpha}} f^{s}$ actually equals it.
              \end{enumerate}
              Thus, we can view $\Psi_{f,\alpha}$ as a quotient of the $\mathcal{D}_{X}$-module $\frac{R_f[s]_{\mathfrak{m}_{\alpha}} f^{s}}{(s-\alpha)^N R_f[s]_{\mathfrak{m}_{\alpha}} f^{s}}$.
              So $\Psi_{f,\alpha}$ itself must also be a regular holonomic $\mathcal{D}_{X}$-module.
    \end{enumerate}

\end{proof}


\section{Comparison theorem}
Our next goal is to prove the following theorem:
\begin{theorem}[Comparison] \label{Theorem 9.5}
    For all $\alpha \in \mathbb{C}$, $\lambda = e^{2\pi i\alpha}$, $$\mathrm{DR}(\Psi_{f,-\alpha}) \simeq i_* \Psi_{f,\lambda}[-1]$$such that $e^{2\pi is} = \lambda^{-1}\cdot e^{2\pi i(s+\alpha)}$ is the corresponding $T$-action.
\end{theorem}
\begin{remark}
    With $(s-\alpha)$ acts on $\Psi_{f,\alpha}$ nilpotently, (the action of) $e^{2\pi i(s+\alpha)}$ is essentially a finite sum, hence it is a well-defined operator for $\Psi_{f,\alpha}$.
\end{remark}

\begin{definition}\label{def:N_m_alpha_k}
    Recall we have a graph embedding~\ref{def:graph_embedding} $\Gamma_f: X \hookrightarrow Y = X\times_{\Spec \C} \mathbb{A}^1_\C$,
    we define $\D_Y$-module $\mathcal{N}_m^{\alpha,k}$ by several steps:
    \begin{enumerate}
        \item We denote $j: U = X \setminus (f=0) \hookrightarrow X$ be the affine open inclusion, $\mathcal{O}_U = R_f$,
              Open embedding makes $j_+$ reduce to $Rj_*$;
              Affine-ness of $j$ makes $Rj_*$ reduce to $j_*$ \cite[Sect.~3.4]{hotta2008d};
              Using Ex.~\ref{example:direct_image_of_graph_embedding}, we get $\Gamma_{f,+}(j_*(\mathcal{O}_U)) = R_f[s]f^s$ a $\D_Y$-module.

        \item Define projection $p_2: Y \to \mathbb{A}^1_\C$ be the projection to the second factor.
              Recall in we have defined $\D_{\mathbb{A}^1_\C}$-module $K^\alpha_m$ in the begging of Sect.~\ref{sect:two_definitions_of_nearby_cycles},
              through the pullback $p_2^*$ (automatically non-derived \cite[Prop.~1.5.13]{hotta2008d}),
              we get $p_2^* K_m^{\alpha}$ a $\D_Y$-module.
              By the isomorphism under \cite[Def.~1.3.1]{hotta2008d},
              says $\D$-module theoretic inverse image is the same with the sheaf theoretic one,
              we get
              \[
                  p_2^* K_m^{\alpha} = R[t] \otimes_{\C[t]} K_m^{\alpha} = \bigoplus_{i=0}^{m-1} R[t, t^{-1}] e^\alpha_i.
              \]
              And the left $\D_Y$-module structure of $p_2^* K_m^{\alpha}$ is explicitly given by
              \[
                  \theta(\psi \otimes s) = \theta(\psi) \otimes s + \psi \cdot \theta(t) \otimes \partial_t s
              \]
              Where $\theta \in \mathcal{T}_Y\subset\D_Y$,
              section $\psi \in \mathcal{O}_Y = R[t]$, and $s \in K_m^{-\alpha}$, using the fact that $t$ is the coordinate of $\mathbb{A}^1_\C$ and $p_2^*(t) = t$.

        \item Then we apply \cite[Proposition~1.2.9.(i)]{hotta2008d},
              get a (left) $\D_Y$-module
              \[
                  \Gamma_{f,+}(j_*(\mathcal{O}_U)) \otimes_{\Oy} p_2^* K_m^{\alpha} = R_f[s]f^s \otimes_{R[t]} \bigoplus_{i=0}^{m-1} R[t, t^{-1}] e^\alpha_i
              \]
              Recall the left $\Oy$-mudule structure of $R_f[s]f^s$ is given be the inclusion $\Oy \hookrightarrow \D_Y$,
              here we use it as right $\Oy$-module structure.

        \item Recall $\D_X[s]\subset \D_Y$ is a $\C$-subalgebra, then we define the $\D_X[s]$-submodule, see \cite[Sect.~3]{Wu2020ComparisonNearbyCyclesBfunctions}
              \[
                  \mathcal{N}_m^{\alpha,k}\subset \Gamma_{f,+}(j_*(\mathcal{O}_U)) \otimes_{\Oy} p_2^* K_m^{\alpha}
              \]
              generated be sections $f^{s-k} \otimes e_\ell^{\alpha}$ for $\ell=0,\dots,m-1$:
              \begin{align*}
                  \mathcal{N}_m^{\alpha,k} := \sum_{\ell=0}^{m-1} \D_X[s] (f^{s-k} \otimes e_\ell^{\alpha}) = \bigoplus_{\ell=0}^{m-1} \D_X[s] (f^{s-k} \otimes e_\ell^{\alpha})
              \end{align*}
              The last equality is also been claimed in this article.
    \end{enumerate}
\end{definition}

Now we can identify $s$-action on $\mathcal{N}_m^{\alpha,k}$ now, note $s = -t\partial_t \in \D_Y$, and $\partial_t\in\mathcal{T}_Y$ is a derivation:
\begin{align*}
    s \cdot (\eta \otimes e_\ell^{\alpha}) & = (s \cdot \eta) \otimes e_\ell^{\alpha} + \eta \cdot (s \cdot e_\ell^{\alpha})       &  & \text{(\cite[Proposition~1.2.9.(i)]{hotta2008d})}  \\
                                           & = (s\eta) \otimes e_\ell^{\alpha}+\eta(-\alpha e^{\alpha}_{\ell}-e^{\alpha}_{\ell-1}) &  & \text{(There is a minus sign in $s=-t\partial_t$)} \\
                                           & =(s-\alpha)\eta \otimes e^{\alpha}_{\ell}-\eta \otimes e^{\alpha}_{\ell-1}
\end{align*}

Since in Sect.~\ref{sect:two_definitions_of_nearby_cycles}, we have
\begin{equation*}
    \begin{array}{ccccccc}
        K_{m-1}^{\alpha} & \lhook\joinrel\xrightarrow{\quad} & K_m^{\alpha}                              & \lhook\joinrel\xrightarrow{\quad} & K_{m+1}^{\alpha}                            & \lhook\joinrel\xrightarrow{\quad} & \cdots \\[1.5ex]
        \overline P      & \xmapsto{\quad}                   & \overline{P \cdot (t\partial_t - \alpha)} & \xmapsto{\quad}                   & \overline{P \cdot (t\partial_t - \alpha)^2} & \xmapsto{\quad}                   & \cdots
    \end{array}
\end{equation*}
Tensor with $R_f[s]f^s$, inducing:
\begin{equation*}\phantomsection\label{construction_of_direct_system_of_N_m_alpha_k}
    \begin{array}{ccccccc}
        R_f[s]f^s \otimes_{\Oy} p_2^* K_{m-1}^{\alpha} & \lhook\joinrel\xrightarrow{\quad} & R_f[s]f^s \otimes_{\Oy} p_2^* K_m^{\alpha} & \lhook\joinrel\xrightarrow{\quad} & R_f[s]f^s \otimes_{\Oy} p_2^* K_{m+1}^{\alpha} & \lhook\joinrel\xrightarrow{\quad} & \cdots \\[1.5ex]
        \eta \otimes e^{\alpha}_i                      & \xmapsto{\quad}                   & \eta \otimes e^{\alpha}_i                  & \xmapsto{\quad}                   & \eta \otimes e^{\alpha}_i                      & \xmapsto{\quad}                   & \cdots
    \end{array}
\end{equation*}
So we can restrict to a direct system of $\mathcal{D}_X[s]$-modules:
\[
    \cdots \lhook\joinrel\longrightarrow \mathcal{N}_{m-1}^{\alpha,k} \lhook\joinrel\longrightarrow \mathcal{N}_m^{\alpha,k} \lhook\joinrel\longrightarrow \mathcal{N}_{m+1}^{\alpha,k} \lhook\joinrel\longrightarrow \cdots .
\]

Since by Thm.~\ref{thm:existence_of_b_function}, when we view $\D_X[s]$ module as $\C[s]$ module
through $\C[s]\hookrightarrow \D_X[s]$, then
\[
    \mathrm{Ann}_{\mathbb{C}[s]}\left(\frac{\mathcal{D}_X[s] f^s}{\mathcal{D}_X[s] f^{s+1}}\right)\neq 0
\]
we define $b_f(s)$ to be the monic polynomial of least degree among $b(s)$ such that
\[
    b(s) \cdot \frac{\mathcal{D}_X[s] f^s}{\mathcal{D}_X[s] f^{s+1}} = 0
\]
called the \textbf{b-function} or \textbf{Bernstein-Sato polynomial of} $f$.

\begin{lemma}\label{lemma 10.6}
    Recall the \hyperref[commutative_diagram_of_hyper_surface]{diagram}. In the (derived) category of (quasi-coherent) $\underline{\C}_{X^{\mathrm{an}}}$-modules (so the chain maps here only guarantees isomorphism of cohomology), we have:

    \begin{enumerate}
        \item $\mathrm{DR}\left(\left(R_f[s] f^s \otimes^L_{\mathbb{C}[s]} \mathbb{C}_\alpha\right)^{\mathrm{an}}\right) \simeq \mathrm{DR}\left(\left(R_f[s] f^{s+\alpha} \otimes^L_{\mathbb{C}[s]} \mathbb{C}_\alpha\right)^{\mathrm{an}}\right)\simeq R j_* (f^{-1} L_1^\lambda)$
        \item $\mathrm{DR}\left(\left(\mathcal{D}_X[s] f^{s+k} \otimes^{L}_{\mathbb{C}[s]} \mathbb{C}_\alpha\right)^{\mathrm{an}}\right) \simeq \mathrm{DR}\left(\left(\mathcal{D}_X[s] f^{s+k+\alpha} \otimes^{L}_{\mathbb{C}[s]} \mathbb{C}_\alpha\right)^{\mathrm{an}}\right)\simeq j_! (f^{-1} L_1^\lambda), \quad k \gg |\alpha|$
    \end{enumerate}
    where $\simeq$ stand for quasi isomorphism, $\mathbb{C}_\alpha := \mathbb{C}[s]_{\mathfrak{m}_\alpha}\big/\mathfrak{m}_\alpha\mathbb{C}[s]_{\mathfrak{m}_\alpha}$ is the residue field of $\alpha \in \mathbb{A}_{\mathbb{C}}^1$, and $\lambda = e^{2\pi i \alpha}$.
\end{lemma}
\begin{remark}
    We defer the explanation of the symbol $R_f[s]f^{s+\alpha}$.
    Intuitively, however, addition in the exponent corresponds to the multiplication of formal powers,
    and this multiplication will be formalized via a tensor product construction.
\end{remark}
\begin{proof}[Sketch of Proof]

    Take a $\C[s]$-free resolution of $\C_\alpha$:
    \begin{equation*}
        0\longrightarrow \mathbb{C}[s]\xrightarrow{s-\alpha} \mathbb{C}[s]\longrightarrow \frac{\mathbb{C}[s]}{(s-\alpha)}\stackrel{\sim}{\longrightarrow}\frac{\mathbb{C}[s]_{\mathfrak{m}_\alpha}}{\mathfrak{m}_\alpha\mathbb{C}[s]_{\mathfrak{m}_\alpha}}=\mathbb{C}_{\alpha}
    \end{equation*}
    then we obtain that
    \begin{align*}
        R_f[s]f^s \otimes^L_{\mathbb{C}[s]} \mathbb{C}_\alpha & \xleftarrow{\substack{\text{quasi-isomorphic}}}\left[ R_f[s]f^s \xrightarrow{s-\alpha} R_f[s]f^s \right] &  & \text{(Tensor with the resolution above)}                                           \\
                                                              & \xrightarrow{\substack{\text{quasi-isomorphic}}} R_f f^\alpha.                                           &  & \text{(see the proof of (4) of Prop.~\ref{prop:properties_of_alpha_nearby_cycles})}
    \end{align*}

    Recall the diagram we have a map $U\xrightarrow{f}\mathbb{A}^{1}_{\mathbb{C}}\backslash\{0\}$.
    In the proof of (4) of Prop.~\ref{prop:properties_of_alpha_nearby_cycles}, we have shown that
    \[
        R_f \cdot f^\alpha = f^* K_1^\alpha
    \]
    Then we recall Sect.~\ref{sect:two_definitions_of_nearby_cycles}, $K_1^\alpha$, restricted to $\mathbb{A}^1_{\mathbb{C}}\backslash\{0\}$, becomes an integrable connection (a locally free $\mathcal{O}$-module of finite rank).

    \medskip

    The next step is to verify a condition called \textbf{Non-characteristic} \cite[Definition~2.4.2]{hotta2008d}.

    Recall an integrable connection has its characteristic variety is exactly the zero section:
    \[
        \Ch\left(K_1^\alpha\mid_{\mathbb{A}^1_{\mathbb{C}}\backslash\{0\}}\right) = T^\ast_{\mathbb{A}^1_{\mathbb{C}}\backslash\{0\}} (\mathbb{A}^1_{\mathbb{C}}\backslash\{0\}) \subset T^\ast (\mathbb{A}^1_{\mathbb{C}}\backslash\{0\})
    \]
    We have the canonical maps $\varpi$ and $\rho$ (see Lem.~\ref{prop:charvar-closed-emb} for their general construction, which is valid for arbitrary morphisms, not just closed embeddings):
    \begin{align*}
        \varpi: S := U \times_{\mathbb{A}^1_{\mathbb{C}}\backslash\{0\}} T^\ast (\mathbb{A}^1_{\mathbb{C}}\backslash\{0\}) & \longrightarrow T^\ast (\mathbb{A}^1_{\mathbb{C}}\backslash\{0\}) \\
        \rho: S                                                                                                            & \longrightarrow T^\ast U
    \end{align*}
    Recall $X = \Spec R$ with $U = \Spec R_f$,
    $\mathbb{A}^1_{\mathbb{C}}\backslash\{0\} = \Spec \mathbb{C}[t, t^{-1}]$.
    The structure morphism is given by
    \begin{align*}
        \phi: \mathbb{C}[t, t^{-1}] & \longrightarrow R_f \\
        t                           & \longmapsto f
    \end{align*}
    Then $T^\ast(\mathbb{A}^1_{\mathbb{C}}\backslash\{0\}) = \Spec \mathbb{C}[t, t^{-1}, \tau]$.
    The zero section $T^\ast_{\mathbb{A}^1_{\mathbb{C}}\backslash\{0\}}(\mathbb{A}^1_{\mathbb{C}}\backslash\{0\})$
    is defined by the ideal
    \[
        (\tau) \subset \mathbb{C}[t, t^{-1}, \tau].
    \]
    Similarly, $T^\ast U = \Spec \Sym^\bullet_{R_f}(\mathcal{T}_{R_f})$
    The zero section $T^\ast_U U$ is defined by the ideal of positive degree elements
    \[
        I_U := \Sym_{R_f}^{\ge 1}(\mathcal{T}_{R_f}) \subset \Sym^\bullet_{R_f}(\mathcal{T}_{R_f}).
    \]
    The fiber product $S$ has coordinate ring
    \[
        \mathcal{O}_S = R_f \otimes_{\mathbb{C}[t,t^{-1}]} \mathbb{C}[t,t^{-1},\tau] \cong R_f[\tau]
    \]
    The map $\varpi$ corresponds to
    \begin{align*}
        \varpi^\#: \mathbb{C}[t,t^{-1},\tau] & \longrightarrow R_f[\tau] \\
        t                                    & \longmapsto f             \\
        \tau                                 & \longmapsto \tau
    \end{align*}
    The map $\rho$ corresponds to
    \begin{align*}
        \rho^\#: \Sym^\bullet_{R_f}(\mathcal{T}_{R_f}) & \longrightarrow R_f[\tau]        \\
        \theta                                         & \longmapsto \tau \cdot \theta(f)
    \end{align*}
    for $\theta \in \Der_\mathbb{C}(R_f)$.

    We then need to calculate the intersection $\varpi^{-1}\left(\Ch(K_1^\alpha\mid_{\mathbb{A}^1_{\mathbb{C}}\backslash\{0\}})\right) \bigcap \rho^{-1}\left(T^\ast_U U\right)$,
    The core formula is to utilize:
    \[
        V(I)\cap V(J) = V(I \cup J) = V((I \cup J)) = V(I+J)  \quad  \text{ (see \cite[\href{https://stacks.math.columbia.edu/tag/00E0}{Tag 00E0}]{stacks-project})}
    \]
    So we:
    \begin{align*}
        I_{\varpi^{-1}\left(\Ch(K_1^\alpha\mid_{\mathbb{A}^1_{\mathbb{C}}\backslash\{0\}})\right)\bigcap \rho^{-1}\left(T^\ast_U U\right)}
         & = \varpi^\#((\tau))\mathcal{O}_S + \rho^\#(I_U)\mathcal{O}_S         \\
         & = (\tau) + (\tau \cdot \theta(f) \mid \theta \in \Der_\mathbb{C}(A)) \\
         & = (\tau) \subset A[\tau].
    \end{align*}
    Thus, the intersection is exactly the zero space $V((\tau)) \subset \Spec A[\tau]$,
    which is $U \times_{\mathbb{A}^1_{\mathbb{C}}\backslash\{0\}} T^\ast_{\mathbb{A}^1_{\mathbb{C}}\backslash\{0\}} (\mathbb{A}^1_{\mathbb{C}}\backslash\{0\})$.
    We verified that $f$ is non-characteristic with respect to $K_1^\alpha\mid_{\mathbb{A}^1_\C\setminus\{0\}}$.

    \medskip

    By \cite[Proposition~4.7.6]{hotta2008d} (Non-characteristic base change for De Rham complex), we have
    \begin{align*}
        \mathrm{DR}^{\mathrm{Hotta}}_{U}(f^* K_1^\alpha) \simeq f^{-1} \mathrm{DR}^{\mathrm{Hotta}}_{\mathbb{A}^1_{\mathbb{C}}\backslash\{0\}}(K_1^\alpha)[d_U - d_{\mathbb{A}^1_{\mathbb{C}}\backslash\{0\}}].
    \end{align*}
    One must be very careful with the convention of the De Rham functor here. In \cite[Sect.~4.7]{hotta2008d},
    the De Rham functor $\mathrm{DR}^{\mathrm{Hotta}}_X(M) := \Omega_X \otimes_{\mathcal{D}_X}^L M$ is concentrated in degrees $[-d_X, 0]$
    (The dualizing sheaf $\Omega_X$ is already at the highest degree).

    However, in our notes Note.~\ref{note 6.2},
    $\mathrm{DR}_{X^{\mathrm{an}}}(M^\mathrm{an})$ is defined as the standard De Rham complex $0 \to M \to M \otimes \Omega^1 \to \dots $ sitting in degrees $[0, d_X]$.
    Therefore,
    \[
        \mathrm{DR}^{\mathrm{Hotta}}_X(\cdot) = \mathrm{DR}_{X^\mathrm{an}}((\cdot)^\mathrm{an})[d_X]
    \]
    Substituting this into the above formula perfectly cancels the dimension shifts:
    \begin{align*}
        \mathrm{DR}_{U^\mathrm{an}}((f^* K_1^\alpha)^\mathrm{an})[d_U] & \simeq f^{-1} \mathrm{DR}_{(\mathbb{A}^1_{\mathbb{C}}\backslash\{0\})^\mathrm{an}}((K_1^\alpha)^\mathrm{an})[d_{\mathbb{A}^1_{\mathbb{C}}\backslash\{0\}}] [d_U - d_{\mathbb{A}^1_{\mathbb{C}}\backslash\{0\}}] \\
                                                                       & \simeq f^{-1} \mathrm{DR}_{(\mathbb{A}^1_{\mathbb{C}}\backslash\{0\})^\mathrm{an}}((K_1^\alpha)^\mathrm{an})[d_U]
    \end{align*}
    Yielding
    \[
        \mathrm{DR}_{U^\mathrm{an}}((f^* K_1^\alpha)^\mathrm{an}) \simeq f^{-1} \mathrm{DR}_{(\mathbb{A}^1_{\mathbb{C}}\backslash\{0\})^\mathrm{an}}((K_1^\alpha)^\mathrm{an})
    \]
    exactly, without any residual dimension shift!

    Combining with the fact that its analytic De Rham complex is exactly the local system
    \[
        \mathrm{DR}\left((K_1^\alpha\mid_{\mathbb{A}^1_{\mathbb{C}}\backslash\{0\}})^{\mathrm{an}}\right) = L_1^\lambda,
    \]
    we deduce:
    \begin{align*}
        \mathrm{DR}((R_f \cdot f^\alpha)^{\mathrm{an}})|_{U^{\mathrm{an}}} & \simeq \mathrm{DR}( (f^* K_1^\alpha)^{\mathrm{an}} )  \\
                                                                           & \simeq f^{-1} \mathrm{DR}((K_1^\alpha)^{\mathrm{an}}) \\
                                                                           & \simeq f^{-1} L_1^\lambda.
    \end{align*}


    \medskip

    Using log resolution of $(X,f=0)$, one can prove $\mathrm{DR}((R_f \cdot f^\alpha)^{\mathrm{an}}) \simeq R j_*(f^{-1} L_1^\lambda)$.

    \medskip

    Using resolution of singularities and duality, one can prove 2.
\end{proof}
\begin{remark}
    We explain the last two sentence in the proof.
    \begin{enumerate}
        \item Using log resolution of $(X,f=0)$, one can prove $\mathrm{DR}((R_f \cdot f^\alpha)^{\mathrm{an}}) \simeq R j_*(f^{-1} L_1^\lambda)$.
        \item Using resolution of singularities and duality, one can prove 2.
    \end{enumerate}
    We resolve the actual meaning in the following steps.
    \begin{enumerate}
        \item \textbf{Step1: Notation and dictionary}

              Let $X$ be a smooth complex algebraic variety (or complex manifold).
              Let $f\in \mathcal O_X(X)$, $D:=f^{-1}(0)$, $U:=X\setminus D$ and $j:U\hookrightarrow X$.
              Fix $\alpha\in \mathbb C$ and set $\lambda:=e^{2\pi i \alpha}$.
              Let $L_1^\lambda$ be the rank-one Kummer local system on $\mathbb{C}^\ast := \left(\mathbb{A}^1_\C \setminus \{0\}\right)^\mathrm{an}$ with monodromy
              $\lambda$ around $0$, and denote by $f^{-1}L_1^\lambda$ its pullback to $U$.

              For an effective Cartier divisor $D$, since it is a closed set, so we have its corresponding ideal sheaf $\mathcal I_D$
              \cite[\href{https://stacks.math.columbia.edu/tag/01WR}{Tag 01WR}]{stacks-project}.
              \begin{enumerate}
                  \item This definition don't guarantee $\mathcal I_D$ is reduced.
                        But (the editor thinks) if one suggests that his divisor is just a closed set, then we should assume $\mathcal I_D$ is reduced.
                  \item For ringed space $(X,\Ox)$, the stalk $\mathcal{O}_{X,x}$ should be a local ring
                        to guarantee that $\mathcal I_D$ is principal \cite[\href{https://stacks.math.columbia.edu/tag/0B8M}{Tag 0B8M}]{stacks-project}.
              \end{enumerate}
              And we define:
              \[
                  \Ox(D) := \mathcal Hom_{\mathcal O_X}(\mathcal I_D, \mathcal O_X).
              \]
              And $\mathcal O_X(kD)\simeq \mathcal O_X(D)^{\otimes k}$ \cite[\href{https://stacks.math.columbia.edu/tag/02OP}{Tag 02OP}]{stacks-project},
              this also suggests we have a direct system, so we can define:
              \[
                  \mathcal O_X(*D):=\varinjlim_{k\to\infty}\mathcal O_X(kD).
              \]
              One can verify that if $D=(f)$ is principal then $\mathcal O_X(*D)=\mathcal O_X[1/f]$.

              On $U$ the multi-valued function $f^\alpha$ defines a rank-one integrable connection
              $\mathcal O_U\cdot f^\alpha$ (recall we defined it in the proof of (4) in Prop.~\ref{prop:properties_of_alpha_nearby_cycles}) and hence a rank-one local system $f^{-1}L_1^\lambda$.

              $\mathcal O_U\cdot f^\alpha := f^\ast \big(\C[t,t^{-1}]t^{\alpha}\big)$ is regular holonomic on $U$ \cite[Theorem~6.1.5]{hotta2008d},
              provided we prove that $\C[t,t^{-1}]t^{\alpha}$ is regular holonomic on $\mathbb{A}^1_\C \setminus \{0\}$.
              To prove this one should embed it into $\mathbb{P}^1_\C$.

              Define the meromorphic $\mathcal D_X$-module
              \[
                  M^\alpha := \mathcal O_X(*D)\cdot f^\alpha .
              \]
        \item \textbf{Step 2: Log resolution and smooth log pairs}

              In general $D$ need not be a normal crossing divisor.
              Choose a log resolution (principalization) $\sigma:X'\to X$ of the ideal $(f)$ so that:
              \begin{enumerate}
                  \item $X'$ is smooth,
                  \item $D':=\sigma^{-1}(D)_{\mathrm{red}}$ is a reduced normal crossing divisor,
                  \item $\sigma$ is an isomorphism over $U$.
              \end{enumerate}
              (See $\mathcal{BP}$ and $\mathcal{BR}$ functors in Rmk.~\ref{rmk:algebraic_KK_regularity_holomonic})
              Write $U'=X'\setminus D'$ and $j':U'\hookrightarrow X'$.
              Then $U'\simeq U$ and $j=\sigma\circ j'$.
              Set $f':=f\circ \sigma$.

              Then we call we have a smooth log pair $(X',D')$, let $\mathcal D_{X',D'}$ be the sheaf of logarithmic
              differential operators preserving the ideal of $D'$ (see Def.~\ref{def:definition_of_V_filtration}).

              A $\mathcal D_{(X')^\mathrm{an},(D')^\mathrm{an}}$-\textbf{lattice} $N \subset (M^\alpha(*D'))^\mathrm{an}$ is a coherent
              $\mathcal D_{(X')^\mathrm{an},(D')^\mathrm{an}}$-submodule restricting to $(M^\alpha)^\mathrm{an}|_{(U')^\mathrm{an}}$ \cite[\S3.3]{Wu2021CharacteristicCyclesHolonomicDmodules},
              existence in analytic case is guaranteed by \cite[Theorem~5.2.17]{hotta2008d}.

              Given any $\mathcal O_{X'}$-module $\mathcal F$, define the twist
              \[
                  \mathcal F(kD'):=\mathcal F\otimes_{\mathcal O_{X'}}\mathcal O_{X'}(kD').
              \]
              Then for such $\overline M$ one defines the logarithmic de Rham complex
              \begin{align*}
                   & \mathrm{DR}_{\left((X')^\mathrm{an},(D')^\mathrm{an}\right)}\left((\overline M)^\mathrm{an}\right)                                                                                                                                                                                            \\
                   & :=\omega_{(X')^\mathrm{an}}((D')^\mathrm{an})\otimes^{\mathbf L}_{\mathcal D_{(X')^\mathrm{an}, (D')^\mathrm{an}}}(\overline M)^\mathrm{an}                                                                                                                                                   \\
                   & \simeq [(\overline M)^\mathrm{an}\to \Omega^1_{(X')^\mathrm{an}}(\log (D')^\mathrm{an})\otimes_{\mathcal{O}_{(X')^\mathrm{an}}} (\overline M)^\mathrm{an}\to \cdots \to \Omega^n_{(X')^\mathrm{an}}(\log (D')^\mathrm{an})\otimes_{\mathcal{O}_{(X')^\mathrm{an}}} (\overline M)^\mathrm{an}] \\
                   & \text{(Last isomorphism claimed in \cite[\S3.1]{Wu2021CharacteristicCyclesHolonomicDmodules})}.
              \end{align*}


        \item \textbf{Step 3: [Black box] stabilization of log de Rham for lattices}

              Let $M$ be a regular holonomic $\mathcal D_{X'}$-module and $\overline M$ a $\mathcal D_{X',D'}$-lattice.
              Then for all integers $k\gg 0$ there are canonical quasi-isomorphisms
              \begin{align*}
                   & \mathrm{DR}_{(X')^\mathrm{an},(D')^\mathrm{an}}((\overline M(kD'))^\mathrm{an})   \simeq Rj'_*\mathrm{DR}(M^\mathrm{an}|_{(U')^\mathrm{an}}), \\
                   & \mathrm{DR}_{(X')^\mathrm{an},(D')^\mathrm{an}}((\overline M(-kD'))^\mathrm{an})  \simeq j'_!\mathrm{DR}(M^\mathrm{an}|_{(U')^\mathrm{an}}).
              \end{align*}
              \cite[Thm.~1.2]{Wu2021CharacteristicCyclesHolonomicDmodules}.

              Moreover, for an analytic integrable logarithmic connection $\mathcal E$ on $((X')^\mathrm{an},(D')^\mathrm{an})$ with local system $L$,
              the following are equivalent:
              \begin{enumerate}
                  \item $\Omega^\bullet_{(X')^\mathrm{an}}(\log (D')^\mathrm{an})(\mathcal E)\to Rj'_*L$
                  \item the inclusion
                        $\Omega^\bullet_{(X')^\mathrm{an}}(\log (D')^\mathrm{an})(\mathcal E)\hookrightarrow \Omega^\bullet_{(X')^\mathrm{an}}(\mathcal E(*(D')^\mathrm{an}))$
                        is a quasi-isomorphism
              \end{enumerate}
              \cite[Thm.~2.1.1]{CMNM06}.

        \item \textbf{Step 4: Proof of sentence (1): $\mathrm{DR}((M^\alpha)^\mathrm{an})\simeq Rj_*(f^{-1}L_1^\lambda)$}

                  On $U$ we have $\mathrm{DR}\big((\mathcal O_U\cdot f^\alpha)^\mathrm{an}\big)\simeq f^{-1}L_1^\lambda$, see the proof of \ref{lemma 10.6}.

                  On $X'$ consider the meromorphic connection
                  \[
                      M'^\alpha := \mathcal O_{X'}(*D')\cdot (f')^\alpha .
                  \]
                  ($f'$ is a regular function on $X'$, by Prop.~\ref{prop:functor_of_points})
                  Choose a $\mathcal D_{X',D'}$-lattice $N \subset (M'^\alpha)^\mathrm{an}$;
                  (Note: $N$ may not be induced by analytification of some \textbf{algebraic} $\mathcal D_{X',D'}$-module)
                  The choice can be arbitrary, and the existence is mentioned in Step 2.

                  Apply \cite[Thm.~1.2]{Wu2021CharacteristicCyclesHolonomicDmodules} to $M=(M'^\alpha)^\mathrm{an}$ and $\overline M=N$:
                  for $k\gg 0$,
                  \[
                      \mathrm{DR}_{(X')^\mathrm{an},(D')^\mathrm{an}}(N(k(D')^\mathrm{an})) \simeq Rj'_*\, f^{-1}L_1^\lambda .
                  \]
                  By \cite[Thm.~2.1.1]{CMNM06}, the inclusion
              $\mathrm{DR}_{(X')^\mathrm{an},(D')^\mathrm{an}}(N(k(D')^\mathrm{an}))\hookrightarrow \mathrm{DR}_{(X')^\mathrm{an}}((M'^\alpha)^\mathrm{an})$
                  is a quasi-isomorphism, hence
                  \[
                      \mathrm{DR}_{(X')^\mathrm{an}}((M'^\alpha)^\mathrm{an})\simeq Rj'_*(f^{-1}L_1^\lambda).
                  \]

                  Finally, since $\sigma$ is proper, de Rham commutes with proper pushforward
                  \cite[Proposition~4.7.5]{hotta2008d}:
                  \[
                      \mathrm{DR}_{(X)^\mathrm{an}}((\sigma_+M'^\alpha)^\mathrm{an})\simeq R\sigma_*\,\mathrm{DR}_{(X')^\mathrm{an}}((M'^\alpha)^\mathrm{an}).
                  \]
                  Using $j=\sigma\circ j'$ and $Rj_*\simeq R\sigma_*Rj'_*$, we obtain
                  \[
                      \mathrm{DR}_{(X)^\mathrm{an}}((\sigma_+M'^\alpha)^\mathrm{an})\simeq Rj_*(f^{-1}L_1^\lambda).
                  \]
                  Since $\sigma$ is an isomorphism over $U$.
                  We have $M'^\alpha = \mathcal O_{X'}(*D')\cdot (f')^\alpha \cong j'_+(\mathcal O_{U}\cdot f^\alpha)$.

              So we have
              \[
                  \sigma_+ M'^\alpha \simeq \sigma_+ (j'_+(\mathcal O_{U}\cdot f^\alpha)) = j_+ (\mathcal O_U\cdot f^\alpha) = M^\alpha.
              \]
              Substituting this into the previous equation, we finally get
              \[
                  \mathrm{DR}_{X^\mathrm{an}}\big((M^\alpha)^\mathrm{an}\big)\simeq Rj_*(f^{-1}L_1^\lambda).
              \]

        \item \textbf{Step 5: Proof of sentence (2): $\mathrm{DR}(\mathcal D_X[s]f^{s+k}\otimes \mathbb C_\alpha)\simeq j_!(f^{-1}L_1^\lambda)$}

              Recall we have proved in Lem.~\ref{lemma 10.6} that:
              \[
                  \DR_{U^\mathrm{an}}\bigl((\mathcal M_U^\alpha)^{\an}\bigr)\ \simeq\ f^{-1}L_1^\lambda.
              \]

              The lecturer (of this course) gives his construction of localization \cite[Definition~2.4]{Wu2020ComparisonNearbyCyclesBfunctions},
              which is same as what is in Rmk.~\ref{rmk:base_change_of_R_f[s]_f^s}, for details:
              \begin{align*}
                  \mathcal D_X[s]_{\mathfrak{m}_\alpha}\cdot \mathcal O_X\cdot f^{s+k} & := \mathcal D_X[s] \cdot \mathcal O_X\cdot f^{s+k} \otimes_{\C[s]} \C[s]_{\mathfrak{m}_\alpha}                                                                       \\
                                                                                       & =\mathcal D_X[s] \cdot \mathcal O_X\cdot f^{s+k} \otimes_{\C[s]} \big(\C[s]_{\mathfrak{m}_\alpha} \otimes_{\C[s]_{\mathfrak{m}_\alpha}} \C[s]_{\mathfrak{m}_\alpha}) \\
                                                                                       & = \big(\mathcal D_X[s] \otimes_{\C[s]} \C[s]_{\mathfrak{m}_\alpha}\big) \cdot \big(\mathcal O_X\cdot f^{s+k} \otimes_{\C[s]} \C[s]_{\mathfrak{m}_\alpha})            \\
                                                                                       & = \mathcal D_X[s]_{\mathfrak{m}_\alpha}\cdot \mathcal{O}_{X,\mathfrak{m}_\alpha}\cdot f^{s+k}
              \end{align*}

              For the open immersion $j$, he defines the $\mathcal D$-module ``$j_!$-extension'' by
              \[
                  j_!^{\mathcal D}(-)\ :=\ \mathbb D\circ j_*^{\mathcal D}\circ \mathbb D(-),
              \]
              which is just similar to what in \cite[Definition~3.2.13]{hotta2008d}.

              Now apply \cite[Theorem~2.10(2) and (3)]{Wu2020ComparisonNearbyCyclesBfunctions}
              to the case $M_U=\mathcal O_U$ and $M_0=\mathcal O_X$:
              for every $\alpha\in\C$ there exists $k_0$ such that for all integers $k>k_0$ we have a
              quasi-isomorphism of complexes of holonomic $\mathcal D_X$-modules
              \begin{align*}
                  j_!^{\mathcal D}(\mathcal O_U\cdot f^\alpha) & \stackrel{{\mathrm{q.i.}}}{\simeq} \bigl(\mathcal D_X[s]_{\mathfrak{m}_\alpha}\cdot \mathcal O_X\cdot f^{s+k}\bigr) \otimes^L_{\C[s]} \C_\alpha  \\
                                                               & = \Bigl(\bigl(\mathcal D_X[s]\cdot \mathcal O_X\cdot f^{s+k}\bigr)\otimes_{\C[s]} \C[s]_{\mathfrak{m}_\alpha}\Bigr) \otimes^L_{\C[s]} \C_\alpha.
              \end{align*}
              Since $\C[s]_{\mathfrak{m}_\alpha}$ is a localization, it is a flat $\C[s]$-algebra. Thus, the ordinary tensor product is exact and coincides with the derived tensor product:
              \[
                  - \otimes_{\C[s]} \C[s]_{\mathfrak{m}_\alpha} \simeq - \otimes^L_{\C[s]} \C[s]_{\mathfrak{m}_\alpha}.
              \]
              By the associativity of the derived tensor product, we can regroup the terms as:
              \[
                  \Bigl( \bigl(\mathcal D_X[s]\cdot \mathcal O_X\cdot f^{s+k}\bigr)\otimes^L_{\C[s]} \C[s]_{\mathfrak{m}_\alpha} \Bigr) \otimes^L_{\C[s]} \C_\alpha
                  \simeq \bigl(\mathcal D_X[s]\cdot \mathcal O_X\cdot f^{s+k}\bigr) \otimes^L_{\C[s]} \Bigl( \C[s]_{\mathfrak{m}_\alpha} \otimes^L_{\C[s]} \C_\alpha \Bigr).
              \]
              Because $\C[s]_{\mathfrak{m}_\alpha}$ is flat (over $\C[s]$), computing the derived tensor product $\C[s]_{\mathfrak{m}_\alpha} \otimes^L_{\C[s]} \C_\alpha$
              reduces to the ordinary tensor product $\C[s]_{\mathfrak{m}_\alpha} \otimes_{\C[s]} (\C[s]/(s-\alpha))$, which is just $\C_\alpha$,
              because in derived category of $\C[s]$-modules we have (quasi-)isomorphism:
              \[
                  \C[s]\big/(s-\alpha) \stackrel{\sim}{\to}\C[s]_{\mathfrak{m}_\alpha} \big/\mathfrak{m}_\alpha\C[s]_{\mathfrak{m}_\alpha}.
              \]
              Substituting this back, the complex simplifies cleanly to:
              \begin{align*}
                  j_!^{\mathcal D}(\mathcal O_U\cdot f^\alpha) & \stackrel{\mathrm{q.i.}}{\simeq} \bigl(\mathcal D_X[s]\cdot \mathcal O_X\cdot f^{s+k}\bigr)\otimes^L_{\C[s]} \C_\alpha.
              \end{align*}

              Apply the analytic de Rham functor $\DR_X$.
              Because $\DR_X$ (algebraic, in sense of \cite[Sect.~4.7]{hotta2008d}) is functorial on the derived category \cite[Sect.~4.7]{hotta2008d}, it sends quasi-isomorphisms to
              quasi-isomorphisms:
              \[
                  \DR_{X^\mathrm{an}}\Bigl(\bigl(\mathcal D_X[s]\cdot f^{s+k}\otimes^L_{\C[s]}\C_\alpha\bigr)^{\an}\Bigr)
                  \simeq
                  \DR_{X^\mathrm{an}}\Bigl(\bigl(j_!^{\mathcal D}(\mathcal O_U\cdot f^\alpha)\bigr)^{\an}\Bigr).
              \]

              By the Riemann--Hilbert correspondence and its six-functor compatibility \cite[Theorem 7.1.1]{hotta2008d},
              $\DR$ commutes with the exceptional direct image for open immersions:
              \[
                  \DR_{X^\mathrm{an}}\Bigl(\bigl(j_!^{\mathcal D}(\mathcal N)\bigr)^{\an}\Bigr)
                  \simeq
                  j_!^{\mathrm{sh}}\Bigl(\DR_{U^\mathrm{an}}(\mathcal N^{\an})\Bigr)
                  \qquad (\mathcal N\ \text{regular holonomic on }U).
              \]
              Apply this to $\mathcal N=\mathcal O_U\cdot f^\alpha$:
              \[
                  \DR_{X^\mathrm{an}}\Bigl(\bigl(j_!^{\mathcal D}(\mathcal O_U\cdot f^\alpha)\bigr)^{\an}\Bigr)
                  \simeq
                  j_!^{\mathrm{sh}}\Bigl(\DR_{U^\mathrm{an}}\bigl((\mathcal O_U\cdot f^\alpha)^{\an}\bigr)\Bigr)
                  \simeq
                  j_!^{\mathrm{sh}}(f^{-1}L_1^\lambda).
              \]

              Combining the last two displays yields the desired statement:
              \[
                  \DR_{X^\mathrm{an}}\Bigl(\bigl(\mathcal D_X[s]\cdot f^{s+k}\ \otimes^L_{\C[s]}\ \C_\alpha \bigr)^{\an}\Bigr)
                  \simeq j_!(f^{-1}L_1^\lambda),
                  \qquad k\gg 0.
              \]
              \qed

    \end{enumerate}

\end{remark}

\begin{lemma} \label{lemma 10.7}
    $$
        (b_f(s-k+\alpha))^m \,\mathrm{kills}\, \mathcal{N}_m^{\alpha,k}\big/\mathcal{N}_m^{\alpha,k-1}
    $$
\end{lemma}
\begin{proof}
    By \hyperref[construction_of_direct_system_of_N_m_alpha_k]{construction}, $\mathcal{N}_m^{\alpha,k}$ is generated by $f^{s-k} \otimes e_0^\alpha, \dots, f^{s-k} \otimes e_{m-1}^\alpha$.
    We can identify $\mathcal{N}_{m-1}^{\alpha,k}$ with the submodule generated by the first $m-1$ elements.
    This yields a short exact sequence of $\mathcal{D}_X[s]$-modules:
    $$
        0\longrightarrow \mathcal{N}_{m-1}^{\alpha,k} \longrightarrow \mathcal{N}_m^{\alpha,k} \longrightarrow Q \longrightarrow 0
    $$
    where the quotient module $Q := \mathcal{N}_m^{\alpha,k} \big/ \mathcal{N}_{m-1}^{\alpha,k}$
    is a cyclic $\mathcal{D}_X[s]$-module generated by the image of $f^{s-k} \otimes e_{m-1}^\alpha$,
    which we denote by $f^{s-k} \otimes \overline{e}_{m-1}^{\alpha}$.

    Recall the $s$-action on the basis elements gives:
    $$
        s \cdot (f^{s-k} \otimes e^{\alpha}_{m-1}) = (s-\alpha) (f^{s-k} \otimes e^{\alpha}_{m-1}) - f^{s-k} \otimes e^{\alpha}_{m-2}
    $$
    In the quotient $Q$, the term belonging to $\mathcal{N}_{m-1}^{\alpha,k}$ vanishes, yielding
    $$
        s \cdot (f^{s-k} \otimes \overline{e}^\alpha_{m-1})=(s-\alpha)(f^{s-k} \otimes \overline{e}^\alpha_{m-1})
    $$
    We want to show that $(b_f(s-k-\alpha))^m$ kills $\mathcal{N}_m^{\alpha,k} \big/ \mathcal{N}_m^{\alpha,k-1}$.
    Taking the above exact sequence modulo the submodule $\mathcal{N}_{m}^{\alpha, k-1}$, we get the exact sequence:
    $$
        0\longrightarrow \frac{\mathcal{N}_{m-1}^{\alpha,k}}{\mathcal{N}_{m-1}^{\alpha,k} \cap \mathcal{N}_m^{\alpha,k-1}} \longrightarrow \frac{\mathcal{N}_m^{\alpha,k}}{\mathcal{N}_m^{\alpha,k-1}} \longrightarrow \overline{Q} \longrightarrow 0
    $$
    where $\overline{Q} = Q \big/ (Q \cap N_m^{\alpha,k-1})$ is a cyclic $\mathcal{D}_X[s]$-module generated by the image of $f^{s-k} \otimes \overline{e}_{m-1}^{\alpha}$.

    Rather than identifying $\overline{Q}$ with a quotient of $\mathcal{D}_X[s]$, we can calculate the action of $b_f(s-k+\alpha)$ directly.
    Recall we have:
    $$
        s \cdot (f^{s-k} \otimes \overline{e}^\alpha_{m-1}) = (s f^{s-k}) \otimes \overline{e}^\alpha_{m-1} - \alpha (f^{s-k} \otimes \overline{e}^\alpha_{m-1}) = ((s-\alpha) f^{s-k}) \otimes \overline{e}^\alpha_{m-1}
    $$
    But what about the action of $\mathcal{D}_X \subset \mathcal{D}_X[s]$?

    Recall the Step 2 of Def.~\ref{def:N_m_alpha_k}, we have:
    \[
        \theta(\psi \otimes s) = \theta(\psi) \otimes s + \psi \cdot \theta(t) \otimes \partial_t s
    \]
    Where $\theta \in \mathcal{T}_Y\subset\D_Y$,
    section $\psi \in \mathcal{O}_Y = R[t]$, and $s \in K_m^{-\alpha}$, using the fact that $t$ is the coordinate of $\mathbb{A}^1_\C$ and $p_2^*(t) = t$.

    Given $D\in\D_X$, since any vector field $\theta \in \mathcal{T}_X\subseteq\mathcal{T}_Y$ has $\theta(t)=0$, so:
    $$
        D \cdot (\eta \otimes \overline{e}^\alpha_{m-1}) = (D \cdot \eta) \otimes \overline{e}^\alpha_{m-1}.
    $$
    So for any polynomial $P(s) \in \mathcal{D}_X[s]$ with coefficients in $\mathcal{D}_X$, we have:
    $$
        P(s) \cdot (f^{s-k} \otimes \overline{e}^\alpha_{m-1}) = (P(s-\alpha) f^{s-k}) \otimes \overline{e}^\alpha_{m-1}
    $$
    We want to calculate the action of $b_f(s-k+\alpha)$.
    Since by Thm~.\ref{thm:existence_of_b_function}, we have $\widetilde{P}(s) \in \mathcal{D}_X[s]$ such that
    $$
        b_f(s) f^s = \widetilde{P}(s) f^{s+1} \in R_f[s]f^s
    $$
    Shifting the variable $s \mapsto s-k$, we get $b_f(s-k)f^{s-k} = \widetilde{P}(s-k)f^{s-k+1}$.
    Using this relation, we calculate:
    \begin{align*}
        b_f(s-k+\alpha) \cdot (f^{s-k} \otimes \overline{e}^\alpha_{m-1})
         & = (b_f((s-\alpha)-k+\alpha) f^{s-k}) \otimes \overline{e}^\alpha_{m-1}          \\
         & = (b_f(s-k) f^{s-k}) \otimes \overline{e}^\alpha_{m-1}                          \\
         & = (\widetilde{P}(s-k) f^{s-k+1}) \otimes \overline{e}^\alpha_{m-1}              \\
         & = \widetilde{P}(s-k+\alpha) \cdot (f^{s-k+1} \otimes \overline{e}^\alpha_{m-1})
    \end{align*}
    Thus $b_f(s-k+\alpha)$ exactly kills $\overline{Q}$.

    Now we proceed by induction on $m$.
    For the base case $m=1$, the submodule $\mathcal{N}_0^{\alpha,k}$ is 0 by definition.
    Therefore $Q \simeq \mathcal{N}_1^{\alpha,k}$, and the quotient $\overline{Q}$ is exactly $\mathcal{N}_1^{\alpha,k} \big/ \mathcal{N}_1^{\alpha,k-1}$.
    Since we just showed $b_f(s-k+\alpha)$ kills $\overline{Q}$, the base case holds.

    For the induction step, we assume $(b_f(s-k+\alpha))^{m-1}$ kills $\mathcal{N}_{m-1}^{\alpha,k} \big/ \mathcal{N}_{m-1}^{\alpha,k-1}$. Recall the exact sequence:
    $$
        0\longrightarrow \frac{\mathcal{N}_{m-1}^{\alpha,k}}{\mathcal{N}_{m-1}^{\alpha,k} \cap \mathcal{N}_m^{\alpha,k-1}} \longrightarrow \frac{\mathcal{N}_m^{\alpha,k}}{\mathcal{N}_m^{\alpha,k-1}} \longrightarrow \overline{Q} \longrightarrow 0
    $$
    Since the left term in the sequence is a quotient of $\mathcal{N}_{m-1}^{\alpha,k} \big/ \mathcal{N}_{m-1}^{\alpha,k-1}$ (as $\mathcal{N}_{m-1}^{\alpha,k-1} \subset \mathcal{N}_{m-1}^{\alpha,k} \cap \mathcal{N}_m^{\alpha,k-1}$), $(b_f(s-k+\alpha))^{m-1}$ kills the left term as well.
    Since $b_{f}(s-k+\alpha)$ kills the right term $\overline{Q}$, their product $(b_f(s-k+\alpha))^{m}$ kills the middle term $\mathcal{N}_m^{\alpha,k} \big/ \mathcal{N}_m^{\alpha,k-1}$, completing the proof.

\end{proof}

\begin{proposition}
    There exists $k_0 > 0$ such that for all $k \geq k_0$, we obtain that (in derived category):
    \begin{enumerate}
        \item $\mathrm{DR}(N_m^{-\alpha,k} \xrightarrow{s} N_{m}^{-\alpha,k}) \simeq R j_*\left(f^{-1} L_m^{\lambda}\right)$;
        \item $\mathrm{DR}(N_m^{-\alpha,-k} \xrightarrow{s} N_{m}^{-\alpha,-k})\simeq j_!\left(f^{-1} L_m^{\lambda}\right)$.
    \end{enumerate}
\end{proposition}
\begin{proof}
    First we explain the notation, how can we apply De Rham to a complex?
    We should use the derived version defined in \cite[Sect.~4.7]{hotta2008d}:
    \[
        \mathrm{DR}_X(M^\bullet) := \mathrm{DR}_{X^\mathrm{an}}\big((M^\bullet)^\mathrm{an}\big) = \Omega_{X^\mathrm{an}} \otimes^L_{\mathcal{D}_{X^\mathrm{an}}} (M^\bullet)^\mathrm{an}.
    \]
    We use this \cite{hotta2008d}'s version here. Note that it is different from our naive one by a shift of degree, see proof of Lem.~\ref{lemma 10.6}.

\begin{lemma}
    Let $W$ be a $\mathbb{C}$-vector space, and let $\varphi$ be a $\mathbb{C}$-linear operator on $W$ admitting a minimal polynomial.
    Set $W_{\infty} = \bigoplus_{k=0}^{\infty} W \otimes e_k$ and define $$\varphi_{\infty}(w \otimes e_i) = (\varphi - \alpha) \cdot w \otimes e_i - w \otimes e_{i-1} \quad \text{for all }  w \in W, \; (e_{-1} = 0).$$Then $\varphi_{\infty}$ is surjective and $\ker(\varphi_{\infty}) = W_{\alpha}$ is the generalized $\alpha$-eigenspace.
\end{lemma}
\begin{proof}
    Define a map
    \begin{equation}
        W_{\alpha} \longrightarrow \ker(\varphi_{\infty})\atop w \longmapsto \sum\limits_{i \geqslant 0} (\varphi - \alpha)^i w \otimes e_{i}.
        \tag{$\ast$}
        \label{eq}
    \end{equation}
    This map is an isomorphism.

    Then we prove surjectivity: If $w\in W_{\alpha}$, then $\varphi_{\infty}( -\sum\limits_{i>j} (\varphi-\alpha)^{i-j-1}w\otimes e_{i})=w\otimes e_{j}$; If $w\in W_{\alpha}^{\perp}$, then $\varphi_{\infty}( \sum\limits_{i=1}^{j}(\varphi-\alpha)^{-i}w\otimes e_{j-i+1} )=w\otimes e_{j}$.
\end{proof}
